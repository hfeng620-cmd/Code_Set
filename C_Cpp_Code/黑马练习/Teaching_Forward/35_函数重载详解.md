# 35 - 函数重载详解

## 一、函数重载的概念

### 1.1 什么是函数重载

函数重载（Overloading）是指在**同一作用域**内，可以定义多个**同名函数**，但它们的**参数列表必须不同**。编译器会根据调用时传入的参数自动选择合适的函数。

```cpp
#include <iostream>
using namespace std;

// 函数重载：同一个函数名，不同参数

// 版本1：两个int相加
int add(int a, int b) {
    cout << "调用 int 版本: ";
    return a + b;
}

// 版本2：两个double相加
double add(double a, double b) {
    cout << "调用 double 版本: ";
    return a + b;
}

// 版本3：三个int相加
int add(int a, int b, int c) {
    cout << "调用 三个参数 版本: ";
    return a + b + c;
}

int main() {
    // 编译器根据参数类型和个数自动选择合适的函数
    cout << add(1, 2) << endl;         // 调用版本1 (int, int)
    cout << add(1.5, 2.5) << endl;     // 调用版本2 (double, double)
    cout << add(1, 2, 3) << endl;      // 调用版本3 (int, int, int)

    return 0;
}

/*
输出结果：
调用 int 版本: 3
调用 double 版本: 4
调用 三个参数 版本: 6
*/
```

### 1.2 为什么需要函数重载

```cpp
#include <iostream>
using namespace std;

// 没有函数重载：需要记住不同的函数名
int addInt(int a, int b) {
    return a + b;
}

double addDouble(double a, double b) {
    return a + b;
}

int addThreeInt(int a, int b, int c) {
    return a + b + c;
}

// 有函数重载：统一用同一个名字，更直观
int add(int a, int b) {
    return a + b;
}

double add(double a, double b) {
    return a + b;
}

int add(int a, int b, int c) {
    return a + b + c;
}

int main() {
    // 没有重载时
    cout << addInt(1, 2) << endl;
    cout << addDouble(1.5, 2.5) << endl;
    cout << addThreeInt(1, 2, 3) << endl;

    // 有重载时（更简洁）
    cout << add(1, 2) << endl;
    cout << add(1.5, 2.5) << endl;
    cout << add(1, 2, 3) << endl;

    return 0;
}
```

## 二、构成重载的条件

### 2.1 参数类型不同

```cpp
#include <iostream>
using namespace std;

// 参数类型不同：int vs double
int maxValue(int a, int b) {
    cout << "int版本: ";
    return (a > b) ? a : b;
}

double maxValue(double a, double b) {
    cout << "double版本: ";
    return (a > b) ? a : b;
}

// 参数类型不同：int vs string
void print(int num) {
    cout << "打印整数: " << num << endl;
}

void print(string str) {
    cout << "打印字符串: " << str << endl;
}

// 参数类型不同：int vs char
void display(int x) {
    cout << "int: " << x << endl;
}

void display(char c) {
    cout << "char: " << c << endl;
}

int main() {
    cout << maxValue(10, 20) << endl;       // 调用int版本
    cout << maxValue(3.14, 2.72) << endl;   // 调用double版本

    print(42);           // 调用int版本
    print("Hello");      // 调用string版本

    display(65);         // 调用int版本
    display('A');        // 调用char版本

    return 0;
}
```

### 2.2 参数个数不同

```cpp
#include <iostream>
using namespace std;

// 一个参数
void print(int a) {
    cout << "一个参数: " << a << endl;
}

// 两个参数
void print(int a, int b) {
    cout << "两个参数: " << a << ", " << b << endl;
}

// 三个参数
void print(int a, int b, int c) {
    cout << "三个参数: " << a << ", " << b << ", " << c << endl;
}

int main() {
    print(1);          // 调用一个参数的版本
    print(1, 2);       // 调用两个参数的版本
    print(1, 2, 3);    // 调用三个参数的版本

    return 0;
}
```

### 2.3 参数顺序不同

```cpp
#include <iostream>
using namespace std;

// 参数顺序不同：(int, double) vs (double, int)
void print(int a, double b) {
    cout << "int, double: " << a << ", " << b << endl;
}

void print(double a, int b) {
    cout << "double, int: " << a << ", " << b << endl;
}

// 另一个例子：(int, string) vs (string, int)
void show(int num, string str) {
    cout << "数字: " << num << ", 字符串: " << str << endl;
}

void show(string str, int num) {
    cout << "字符串: " << str << ", 数字: " << num << endl;
}

int main() {
    print(10, 3.14);       // 调用(int, double)版本
    print(3.14, 10);       // 调用(double, int)版本

    show(42, "hello");     // 调用(int, string)版本
    show("hello", 42);     // 调用(string, int)版本

    return 0;
}
```

## 三、不构成重载的情况

### 3.1 仅返回值类型不同（编译报错）

```cpp
#include <iostream>
using namespace std;

// 错误！仅返回值类型不同，不构成重载
// int add(int a, int b) {
//     return a + b;
// }

// double add(int a, int b) {  // 编译错误：重定义
//     return a + b;
// }

// 正确做法：函数名不同
int addInt(int a, int b) {
    return a + b;
}

double addDouble(int a, int b) {
    return a + b;
}

int main() {
    cout << addInt(1, 2) << endl;
    cout << addDouble(1, 2) << endl;
    return 0;
}
```

### 3.2 仅参数名不同（不是重载，是重复定义）

```cpp
#include <iostream>
using namespace std;

// 错误！仅参数名不同，不构成重载
// void print(int x) {
//     cout << x << endl;
// }

// void print(int y) {  // 编译错误：重定义
//     cout << y << endl;
// }

// 正确理解：参数名不影响函数签名
void print(int x) {
    cout << "x = " << x << endl;
}

int main() {
    print(42);
    return 0;
}
```

### 3.3 仅const修饰不同（特殊情况）

```cpp
#include <iostream>
using namespace std;

// 对于值传递，const修饰不构成重载
// void func(int a) {
//     cout << "int" << endl;
// }

// void func(const int a) {  // 编译错误：重定义
//     cout << "const int" << endl;
// }

// 但是，对于指针和引用，const可以构成重载
void print(int &x) {
    cout << "int &: " << x << endl;
}

void print(const int &x) {
    cout << "const int &: " << x << endl;
}

int main() {
    int a = 10;
    const int b = 20;

    print(a);  // 调用(int &)版本
    print(b);  // 调用(const int &)版本
    print(42); // 调用(const int &)版本，因为42是常量

    return 0;
}
```

## 四、函数重载的调用规则

### 4.1 编译器如何选择

```cpp
#include <iostream>
using namespace std;

void print(int x) {
    cout << "int: " << x << endl;
}

void print(double x) {
    cout << "double: " << x << endl;
}

void print(int x, int y) {
    cout << "int, int: " << x << ", " << y << endl;
}

int main() {
    // 精确匹配
    print(42);         // 匹配 int
    print(3.14);       // 匹配 double
    print(1, 2);       // 匹配 int, int

    // 类型转换匹配
    print('A');        // char -> int，匹配 int版本
    print(true);       // bool -> int，匹配 int版本
    print(2.5f);       // float -> double，匹配 double版本

    return 0;
}
```

### 4.2 匹配优先级

```cpp
#include <iostream>
using namespace std;

void func(int x) {
    cout << "int版本: " << x << endl;
}

void func(double x) {
    cout << "double版本: " << x << endl;
}

int main() {
    // 1. 精确匹配优先
    func(42);      // 精确匹配 int
    func(3.14);    // 精确匹配 double

    // 2. 类型提升匹配
    func('A');     // char提升为int，匹配int版本
    func(true);    // bool提升为int，匹配int版本

    // 3. 类型转换匹配
    func(2.5f);    // float转换为double，匹配double版本

    // 4. 可能产生二义性的情况
    // func(1.5);  // 如果只有func(int)和func(long)，可能会有二义性

    return 0;
}
```

### 4.3 二义性问题

```cpp
#include <iostream>
using namespace std;

// 两个版本都可以匹配某些调用，产生二义性
void print(int x) {
    cout << "int: " << x << endl;
}

void print(long x) {
    cout << "long: " << x << endl;
}

int main() {
    print(42);     // 精确匹配int
    print(42L);    // 精确匹配long

    // print(3.14); // 错误！二义性：double可以转换为int或long

    // 解决方案：显式类型转换
    print((int)3.14);    // 明确调用int版本
    print((long)3.14);   // 明确调用long版本

    return 0;
}
```

## 五、函数重载与默认参数的冲突

### 5.1 可能产生二义性

```cpp
#include <iostream>
using namespace std;

// 函数重载
void func(int a) {
    cout << "一个参数: " << a << endl;
}

// 带默认参数的重载
void func(int a, int b = 10) {
    cout << "两个参数: " << a << ", " << b << endl;
}

int main() {
    // func(5);  // 错误！二义性：可以匹配func(int)或func(int, int=10)

    // 解决方案1：显式调用
    func(5, 20);  // 明确调用两个参数的版本

    // 解决方案2：删除其中一个
    // 保留 func(int a, int b = 10)，删除 func(int a)

    return 0;
}
```

### 5.2 如何避免冲突

```cpp
#include <iostream>
using namespace std;

// 方案1：只保留一个版本（推荐）
void print(int a, int b = 0, int c = 0) {
    cout << a;
    if (b != 0) cout << ", " << b;
    if (c != 0) cout << ", " << c;
    cout << endl;
}

// 方案2：使用不同的函数名
void printOne(int a) {
    cout << "一个参数: " << a << endl;
}

void printTwo(int a, int b) {
    cout << "两个参数: " << a << ", " << b << endl;
}

int main() {
    print(1);        // 使用默认参数
    print(1, 2);     // 使用默认参数
    print(1, 2, 3);  // 不使用默认参数

    printOne(1);     // 明确的函数名
    printTwo(1, 2);  // 明确的函数名

    return 0;
}
```

## 六、函数重载的实际应用

### 6.1 打印函数重载

```cpp
#include <iostream>
#include <string>
using namespace std;

// 打印不同类型的数据
void print(int x) {
    cout << "整数: " << x << endl;
}

void print(double x) {
    cout << "浮点数: " << x << endl;
}

void print(string x) {
    cout << "字符串: " << x << endl;
}

void print(bool x) {
    cout << "布尔值: " << (x ? "true" : "false") << endl;
}

void print(int arr[], int size) {
    cout << "数组: [";
    for (int i = 0; i < size; i++) {
        cout << arr[i];
        if (i < size - 1) cout << ", ";
    }
    cout << "]" << endl;
}

int main() {
    print(42);
    print(3.14);
    print("Hello");
    print(true);

    int arr[] = {1, 2, 3, 4, 5};
    print(arr, 5);

    return 0;
}
```

### 6.2 计算函数重载

```cpp
#include <iostream>
#include <cmath>
using namespace std;

// 计算圆的面积
double area(double radius) {
    return M_PI * radius * radius;
}

// 计算矩形的面积
double area(double width, double height) {
    return width * height;
}

// 计算三角形的面积（底和高）
double area(double base, double height, bool isTriangle) {
    if (isTriangle) {
        return 0.5 * base * height;
    }
    return base * height;
}

int main() {
    cout << "圆的面积（半径5）: " << area(5.0) << endl;
    cout << "矩形面积（4x6）: " << area(4.0, 6.0) << endl;
    cout << "三角形面积（底6高4）: " << area(6.0, 4.0, true) << endl;

    return 0;
}
```

### 6.3 构造函数重载（类中常用）

```cpp
#include <iostream>
#include <string>
using namespace std;

class Student {
public:
    string name;
    int age;
    double score;

    // 无参构造
    Student() {
        name = "未知";
        age = 0;
        score = 0.0;
        cout << "无参构造" << endl;
    }

    // 一个参数构造
    Student(string n) {
        name = n;
        age = 0;
        score = 0.0;
        cout << "一个参数构造" << endl;
    }

    // 两个参数构造
    Student(string n, int a) {
        name = n;
        age = a;
        score = 0.0;
        cout << "两个参数构造" << endl;
    }

    // 三个参数构造
    Student(string n, int a, double s) {
        name = n;
        age = a;
        score = s;
        cout << "三个参数构造" << endl;
    }

    void display() {
        cout << "姓名: " << name << ", 年龄: " << age << ", 分数: " << score << endl;
    }
};

int main() {
    Student s1;
    s1.display();

    Student s2("张三");
    s2.display();

    Student s3("李四", 20);
    s3.display();

    Student s4("王五", 21, 95.5);
    s4.display();

    return 0;
}
```

## 七、常见错误

### 7.1 忽略const的重载

```cpp
#include <iostream>
using namespace std;

// 错误示例：对于值传递，const不构成重载
// void func(int x) { }
// void func(const int x) { }  // 错误！重复定义

// 正确示例：对于引用，const构成重载
void print(int &x) {
    cout << "int &: " << x << endl;
}

void print(const int &x) {
    cout << "const int &: " << x << endl;
}

int main() {
    int a = 10;
    const int b = 20;

    print(a);  // 调用(int &)版本
    print(b);  // 调用(const int &)版本
    print(42); // 调用(const int &)版本

    return 0;
}
```

### 7.2 类型转换导致的二义性

```cpp
#include <iostream>
using namespace std;

void func(int x) {
    cout << "int: " << x << endl;
}

void func(float x) {
    cout << "float: " << x << endl;
}

int main() {
    func(42);     // 匹配int
    func(3.14f);  // 匹配float

    // func(3.14); // 错误！二义性：double可以转换为int或float

    // 解决方案：显式转换
    func((int)3.14);    // 调用int版本
    func((float)3.14);  // 调用float版本

    return 0;
}
```

### 7.3 默认参数与重载冲突

```cpp
#include <iostream>
using namespace std;

// 冲突示例
void print(int a) {
    cout << "一个参数: " << a << endl;
}

void print(int a, int b = 10) {
    cout << "两个参数: " << a << ", " << b << endl;
}

int main() {
    // print(5);  // 错误！二义性

    // 解决方案：删除其中一个，或修改函数名
    print(5, 20);  // OK

    return 0;
}
```

## 八、注意事项总结

1. **构成重载的条件**：参数类型、个数、顺序不同
2. **不构成重载**：仅返回值类型不同、仅参数名不同
3. **const修饰**：对于值传递不构成重载，对于引用和指针可以构成重载
4. **二义性**：当编译器无法确定调用哪个函数时，会产生二义性错误
5. **默认参数冲突**：重载函数带默认参数可能产生二义性
6. **类型转换**：编译器会尝试自动类型转换来匹配函数
7. **建议**：函数重载应该用于功能相似的函数，不要滥用
