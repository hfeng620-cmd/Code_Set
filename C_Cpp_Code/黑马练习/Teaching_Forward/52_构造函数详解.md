# 构造函数详解

## 一、构造函数的作用

构造函数在**创建对象时自动调用**，用于**初始化成员属性**。

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
public:
    string name;
    int age;

    // 构造函数：创建对象时自动调用
    Person() {
        name = "未命名";
        age = 0;
        cout << "Person 构造函数被调用了！" << endl;
    }
};

int main() {
    cout << "准备创建对象..." << endl;
    Person p;       // 创建对象时，构造函数自动调用
    cout << "对象创建完成！" << endl;

    cout << "姓名：" << p.name << endl;   // 输出：未命名
    cout << "年龄：" << p.age << endl;     // 输出：0

    return 0;
}
```

**输出：**
```
准备创建对象...
Person 构造函数被调用了！
对象创建完成！
姓名：未命名
年龄：0
```

## 二、构造函数的特点

1. **函数名与类名相同**
2. **没有返回值**（连 `void` 都没有）
3. **可以有参数**（可以重载）
4. **创建对象时自动调用**，不需要手动调用

```cpp
class Person {
public:
    string name;
    int age;

    // 特点1：函数名与类名相同（Person）
    // 特点2：没有返回值（没有 void，什么都没有）
    // 特点3：可以有参数
    Person(string n, int a) {
        name = n;
        age = a;
    }

    // 可以重载多个构造函数
    Person() {
        name = "默认";
        age = 0;
    }
};
```

## 三、三种构造函数

### 3.1 无参构造（默认构造）

```cpp
class Person {
public:
    string name;
    int age;

    // 无参构造（默认构造）
    Person() {
        name = "未命名";
        age = 0;
    }
};

int main() {
    Person p;           // 调用无参构造
    cout << p.name;     // 输出：未命名
    return 0;
}
```

### 3.2 有参构造

```cpp
class Person {
public:
    string name;
    int age;

    // 有参构造
    Person(string n, int a) {
        name = n;
        age = a;
    }
};

int main() {
    Person p("张三", 25);   // 调用有参构造
    cout << p.name;          // 输出：张三
    return 0;
}
```

### 3.3 拷贝构造

```cpp
class Person {
public:
    string name;
    int age;

    // 有参构造
    Person(string n, int a) {
        name = n;
        age = a;
    }

    // 拷贝构造：参数是同类对象的引用
    Person(const Person &p) {
        name = p.name;      // 把 p 的 name 复制到当前对象
        age = p.age;        // 把 p 的 age 复制到当前对象
        cout << "拷贝构造被调用！" << endl;
    }
};

int main() {
    Person p1("张三", 25);      // 调用有参构造
    Person p2(p1);               // 调用拷贝构造

    cout << p2.name << endl;     // 输出：张三
    cout << p2.age << endl;      // 输出：25

    return 0;
}
```

## 四、构造函数的三种调用方式

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
public:
    string name;
    int age;

    // 无参构造
    Person() {
        name = "默认";
        age = 0;
        cout << "无参构造被调用" << endl;
    }

    // 有参构造
    Person(string n, int a) {
        name = n;
        age = a;
        cout << "有参构造被调用" << endl;
    }

    // 拷贝构造
    Person(const Person &p) {
        name = p.name;
        age = p.age;
        cout << "拷贝构造被调用" << endl;
    }
};

int main() {
    // ========== 1. 括号法（最常用）==========
    cout << "--- 括号法 ---" << endl;
    Person p1;                   // 无参构造（注意：不要加括号！）
    Person p2("张三", 25);      // 有参构造
    Person p3(p2);               // 拷贝构造

    cout << "p1: " << p1.name << ", " << p1.age << endl;
    cout << "p2: " << p2.name << ", " << p2.age << endl;
    cout << "p3: " << p3.name << ", " << p3.age << endl;

    // ========== 2. 显式法 ==========
    cout << "\n--- 显式法 ---" << endl;
    Person p4 = Person();                // 无参构造
    Person p5 = Person("李四", 30);     // 有参构造
    Person p6 = Person(p5);              // 拷贝构造

    cout << "p4: " << p4.name << ", " << p4.age << endl;
    cout << "p5: " << p5.name << ", " << p5.age << endl;
    cout << "p6: " << p6.name << ", " << p6.age << endl;

    // ========== 3. 隐式法（不推荐，代码可读性差）==========
    cout << "\n--- 隐式法 ---" << endl;
    Person p7 = "王五";        // 相当于 Person p7("王五"); （需要单参数构造函数）
    Person p8 = p7;             // 相当于 Person p8(p7); （拷贝构造）

    return 0;
}
```

**输出：**
```
--- 括号法 ---
无参构造被调用
有参构造被调用
拷贝构造被调用
p1: 默认, 0
p2: 张三, 25
p3: 张三, 25

--- 显式法 ---
无参构造被调用
有参构造被调用
拷贝构造被调用
p4: 默认, 0
p5: 李四, 30
p6: 李四, 30

--- 隐式法 ---
有参构造被调用（单参数）
拷贝构造被调用
```

## 五、匿名对象

匿名对象是没有名字的对象，**当前行执行结束后立即销毁**。

```cpp
#include <iostream>
using namespace std;

class Person {
public:
    string name;

    Person() {
        cout << "无参构造" << endl;
    }

    Person(string n) {
        name = n;
        cout << "有参构造：" << name << endl;
    }

    ~Person() {
        cout << "析构：" << name << endl;
    }
};

int main() {
    cout << "=== 匿名对象演示 ===" << endl;

    // 匿名对象：当前行结束后立即销毁
    Person("张三");       // 创建匿名对象，这一行结束后立即析构

    cout << "=== 继续执行 ===" << endl;

    // 普通对象：离开作用域才销毁
    Person p("李四");

    cout << "=== main 即将结束 ===" << endl;
    return 0;
}
```

**输出：**
```
=== 匿名对象演示 ===
有参构造：张三
析构：张三            ← 匿名对象立即被销毁
=== 继续执行 ===
有参构造：李四
=== main 即将结束 ===
析析：李四            ← 普通对象在 main 结束时才销毁
```

## 六、常见陷阱：Person p1() 不是创建对象

```cpp
class Person {
public:
    Person() {
        cout << "构造函数" << endl;
    }
};

int main() {
    Person p1;      // ✅ 这才是创建对象，调用无参构造
    // Person p1(); // ❌ 这不是创建对象！这是函数声明！
                     //    编译器认为：声明了一个名为 p1 的函数，
                     //    返回类型是 Person，没有参数

    return 0;
}
```

**记忆方法：** 创建无参对象时，不要加括号！

## 七、编译器默认提供的构造函数

### 7.1 默认无参构造

如果你没有写任何构造函数，编译器会自动提供一个空的无参构造函数。

```cpp
class Person {
public:
    string name;
    int age;

    // 编译器自动提供：
    // Person() {}   // 空实现，什么都不做
};

int main() {
    Person p;       // ✅ 可以创建对象（调用编译器默认的无参构造）
    cout << p.name; // 输出：空字符串（string 的默认值）
    cout << p.age;  // 输出：0（int 的默认值是未定义的，可能为0）
    return 0;
}
```

### 7.2 默认拷贝构造

编译器还会自动提供一个默认的拷贝构造函数，进行**浅拷贝**（逐字节复制）。

```cpp
class Person {
public:
    string name;
    int age;
};

int main() {
    Person p1;
    p1.name = "张三";
    p1.age = 25;

    Person p2(p1);   // 调用编译器默认的拷贝构造（浅拷贝）
    cout << p2.name;  // 输出：张三
    cout << p2.age;   // 输出：25

    return 0;
}
```

### 7.3 重要规则

```cpp
// 规则1：如果你写了有参构造，编译器不再提供默认无参构造
class A {
public:
    int x;
    A(int val) {    // 写了有参构造
        x = val;
    }
    // 编译器不再提供 A() {}
};

int main() {
    // A a;          // ❌ 错误！没有无参构造可用
    A a(10);         // ✅ 有参构造
}

// 规则2：如果你写了拷贝构造，编译器不再提供默认构造函数
class B {
public:
    int x;
    B(const B &other) {   // 写了拷贝构造
        x = other.x;
    }
    // 编译器不再提供 B() {} 和 B(其他参数)
};

int main() {
    // B b;          // ❌ 错误！没有无参构造
    B b1;            // 假设上面能编译...
    // B b2(b1);    // ✅ 用自己写的拷贝构造
}
```

## 八、完整综合示例

```cpp
#include <iostream>
#include <string>
using namespace std;

class Student {
public:
    string name;
    int age;
    double score;

    // 无参构造（默认构造）
    Student() {
        name = "未知";
        age = 0;
        score = 0.0;
        cout << "[无参构造] 创建了学生：" << name << endl;
    }

    // 有参构造（全参数）
    Student(string n, int a, double s) {
        name = n;
        age = a;
        score = s;
        cout << "[有参构造] 创建了学生：" << name << endl;
    }

    // 有参构造（只传姓名）
    Student(string n) {
        name = n;
        age = 18;
        score = 0.0;
        cout << "[单参构造] 创建了学生：" << name << endl;
    }

    // 拷贝构造
    Student(const Student &s) {
        name = s.name;
        age = s.age;
        score = s.score;
        cout << "[拷贝构造] 复制了学生：" << name << endl;
    }

    // 显示信息
    void show() {
        cout << "  姓名：" << name
             << "，年龄：" << age
             << "，分数：" << score << endl;
    }
};

int main() {
    cout << "====== 括号法 ======" << endl;
    Student s1;                     // 无参构造
    Student s2("张三", 20, 95.5);  // 有参构造
    Student s3(s2);                 // 拷贝构造

    s1.show();
    s2.show();
    s3.show();

    cout << "\n====== 显式法 ======" << endl;
    Student s4 = Student();                     // 无参构造
    Student s5 = Student("李四", 22, 88.0);    // 有参构造
    Student s6 = Student(s5);                   // 拷贝构造

    s4.show();
    s5.show();
    s6.show();

    cout << "\n====== 隐式法 ======" << endl;
    Student s7 = "王五";            // 单参构造
    Student s8 = s7;                // 拷贝构造

    s7.show();
    s8.show();

    cout << "\n====== 匿名对象 ======" << endl;
    Student("赵六", 19, 92.0).show();   // 匿名对象，用完即销毁

    return 0;
}
```

## 九、常见错误总结

### 错误1：无参加括号

```cpp
Person p();     // ❌ 这是函数声明，不是创建对象！
Person p;       // ✅ 正确
```

### 错误2：写了有参构造，忘记写无参构造

```cpp
class Person {
public:
    Person(string n) { name = n; }  // 写了有参构造
};

int main() {
    // Person p;   // ❌ 编译错误！编译器不再提供默认无参构造
    Person p("张三");  // ✅
}
```

### 错误3：拷贝构造中忘记加 const

```cpp
class Person {
public:
    // Person(Person &p) { ... }          // 也能用，但不规范
    Person(const Person &p) { ... }       // ✅ 加 const，防止意外修改原对象
};
```

### 注意事项

1. 构造函数**没有返回值**，连 void 都没有
2. 构造函数**可以重载**（多个同名不同参数的构造函数）
3. 创建对象时**自动调用**，不需要手动调用
4. 如果自己写了构造函数，编译器**不再提供默认构造**
5. **建议**：写了有参构造，就手动写一个无参构造
