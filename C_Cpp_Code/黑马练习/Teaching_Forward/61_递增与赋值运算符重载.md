# 递增与赋值运算符重载

## 一、递增 ++ 运算符重载

递增运算符 `++` 有两种形式：
- **前置 ++**：`++a`，先自增，再返回自身
- **后置 ++**：`a++`，先保存旧值，再自增，返回旧值

---

### 1.1 前置 ++ 重载

**语法：** `类名& operator++()`

**特点：** 先自增，再返回自身的**引用**（支持链式操作）

```cpp
#include <iostream>
using namespace std;

class MyInteger {
    // 声明友元，让全局函数 operator<< 能访问私有成员
    friend ostream& operator<<(ostream &out, const MyInteger &myInt);

private:
    int m_Value;

public:
    // 构造函数
    MyInteger(int value = 0) : m_Value(value) {}

    // 前置 ++ 重载
    // 没有参数，操作数就是 this
    // 返回引用（MyInteger&），支持链式操作
    // 调用方式：++myInt  →  myInt.operator++()
    MyInteger& operator++() {
        // 第1步：先自增
        this->m_Value++;
        // 第2步：返回自身引用
        return *this;
    }
};

// 重载 << 运算符，用于输出
ostream& operator<<(ostream &out, const MyInteger &myInt) {
    out << myInt.m_Value;
    return out;
}

int main() {
    MyInteger myInt(10);

    cout << "初始值: " << myInt << endl;       // 输出：10

    // 前置 ++：先自增，再返回
    cout << "++myInt = " << ++myInt << endl;   // 输出：11
    cout << "自增后: " << myInt << endl;        // 输出：11

    // 链式操作（因为返回的是引用）
    cout << "++++myInt = " << ++ ++myInt << endl;  // 输出：13（连续自增两次）
    cout << "最终值: " << myInt << endl;            // 输出：13

    return 0;
}
```

**前置 ++ 的执行流程：**

```
++myInt
    ↓ 编译器转换
myInt.operator++()
    ↓ 函数内部
m_Value++  →  m_Value 从 10 变成 11
    ↓ 返回
return *this  →  返回 myInt 自身的引用
    ↓ 输出
cout << myInt  →  输出 11
```

### 1.2 后置 ++ 重载

**语法：** `类名 operator++(int)`

**特点：**
- 参数中的 `int` 是**占位参数**，仅用于区分前置和后置
- 先保存旧值，再自增，返回旧值（**值返回**，不是引用）
- 效率比前置低（因为有额外的拷贝）

```cpp
#include <iostream>
using namespace std;

class MyInteger {
    friend ostream& operator<<(ostream &out, const MyInteger &myInt);

private:
    int m_Value;

public:
    MyInteger(int value = 0) : m_Value(value) {}

    // 前置 ++ 重载（无参数）
    MyInteger& operator++() {
        this->m_Value++;
        return *this;
    }

    // 后置 ++ 重载（int 是占位参数，不是真正的参数）
    // 调用方式：myInt++  →  myInt.operator++(0)  编译器自动传入 0
    // 返回值类型是 MyInteger（不是引用），返回的是旧值的拷贝
    MyInteger operator++(int) {
        // 第1步：保存当前值（旧值）
        MyInteger temp = *this;  // 拷贝构造，temp 保存了自增前的值

        // 第2步：自增
        this->m_Value++;

        // 第3步：返回旧值
        return temp;  // 返回的是 temp 的拷贝（不是引用）
    }
};

ostream& operator<<(ostream &out, const MyInteger &myInt) {
    out << myInt.m_Value;
    return out;
}

int main() {
    MyInteger myInt(10);

    cout << "初始值: " << myInt << endl;       // 输出：10

    // 后置 ++：先返回旧值，再自增
    cout << "myInt++ = " << myInt++ << endl;   // 输出：10（返回的是旧值）
    cout << "自增后: " << myInt << endl;        // 输出：11（已经自增了）

    return 0;
}
```

**后置 ++ 的执行流程：**

```
myInt++
    ↓ 编译器转换
myInt.operator++(0)  // 编译器自动传入 0 作为占位参数
    ↓ 函数内部
MyInteger temp = *this;  →  temp 保存 m_Value = 10（旧值）
this->m_Value++;         →  m_Value 从 10 变成 11
return temp;             →  返回 temp（值为 10 的拷贝）
    ↓ 输出
cout << temp  →  输出 10（旧值）
    ↓ 此时 myInt 的 m_Value 已经是 11
```

### 1.3 前置和后置的区别

```cpp
#include <iostream>
using namespace std;

class MyInteger {
    friend ostream& operator<<(ostream &out, const MyInteger &myInt);

private:
    int m_Value;

public:
    MyInteger(int value = 0) : m_Value(value) {}

    // 前置 ++：返回引用
    MyInteger& operator++() {
        m_Value++;
        return *this;
    }

    // 后置 ++：返回值（不是引用）
    MyInteger operator++(int) {
        MyInteger temp = *this;
        m_Value++;
        return temp;
    }
};

ostream& operator<<(ostream &out, const MyInteger &myInt) {
    out << myInt.m_Value;
    return out;
}

int main() {
    MyInteger a(10);
    MyInteger b(10);

    // 前置 ++：返回引用，支持链式操作
    cout << "前置: " << ++ ++a << endl;  // 输出：12（连续自增两次）
    cout << "a = " << a << endl;          // 输出：12

    // 后置 ++：返回值（不是引用），不支持链式操作
    // cout << "后置: " << b++ ++ << endl;  // 错误！b++ 返回的是临时对象，不能对临时对象再 ++
    // 因为 b++ 返回的是值（不是引用），是一个临时对象，临时对象不能作为左值

    // 后置 ++ 可以单独使用
    b++;
    cout << "b = " << b << endl;  // 输出：11

    return 0;
}
```

**前置 vs 后置 对比表：**

| 对比项 | 前置 ++ (++a) | 后置 ++ (a++) |
|--------|-------------|-------------|
| 语法 | `operator++()` | `operator++(int)` |
| 参数 | 无 | int 占位参数 |
| 返回值类型 | `MyInteger&`（引用） | `MyInteger`（值） |
| 返回内容 | 自增后的自身 | 自增前的旧值副本 |
| 链式操作 | 支持 | 不支持 |
| 效率 | 高（无额外拷贝） | 低（有额外拷贝） |
| 建议 | 优先使用 | 除非需要旧值否则不用 |

### 1.4 前置效率更高的原因

```cpp
#include <iostream>
using namespace std;

class MyInteger {
    friend ostream& operator<<(ostream &out, const MyInteger &myInt);

private:
    int m_Value;

public:
    MyInteger(int value = 0) : m_Value(value) {}

    // 拷贝构造函数（用于观察拷贝次数）
    MyInteger(const MyInteger &other) {
        m_Value = other.m_Value;
        cout << "  [拷贝构造被调用]" << endl;
    }

    // 前置 ++：没有额外拷贝
    MyInteger& operator++() {
        m_Value++;
        return *this;  // 直接返回自身引用，无拷贝
    }

    // 后置 ++：有一次额外拷贝（创建 temp）
    MyInteger operator++(int) {
        MyInteger temp = *this;  // 第1次拷贝：拷贝构造创建 temp
        m_Value++;
        return temp;             // 第2次拷贝：返回时拷贝 temp
    }
};

ostream& operator<<(ostream &out, const MyInteger &myInt) {
    out << myInt.m_Value;
    return out;
}

int main() {
    MyInteger a(10);

    cout << "=== 前置 ++ ===" << endl;
    ++a;  // 没有拷贝构造被调用

    cout << "\n=== 后置 ++ ===" << endl;
    a++;  // 会调用拷贝构造

    return 0;
}
```

**输出：**
```
=== 前置 ++ ===

=== 后置 ++ ===
  [拷贝构造被调用]
```

**结论：** 前置 `++` 效率更高，因为它不需要创建临时对象。在循环中大量使用时，差距会很明显。

### 1.5 递减 -- 运算符重载（同理）

递减运算符 `--` 的重载方式和 `++` 完全一样，只是把自增改成自减。

```cpp
#include <iostream>
using namespace std;

class MyInteger {
    friend ostream& operator<<(ostream &out, const MyInteger &myInt);

private:
    int m_Value;

public:
    MyInteger(int value = 0) : m_Value(value) {}

    // 前置 --：先自减，再返回自身引用
    MyInteger& operator--() {
        m_Value--;
        return *this;
    }

    // 后置 --：先保存旧值，再自减，返回旧值
    MyInteger operator--(int) {
        MyInteger temp = *this;
        m_Value--;
        return temp;
    }

    // 前置 ++（也加上，方便对比）
    MyInteger& operator++() {
        m_Value++;
        return *this;
    }

    // 后置 ++
    MyInteger operator++(int) {
        MyInteger temp = *this;
        m_Value++;
        return temp;
    }
};

ostream& operator<<(ostream &out, const MyInteger &myInt) {
    out << myInt.m_Value;
    return out;
}

int main() {
    MyInteger a(10);

    cout << "初始值: " << a << endl;       // 10
    cout << "++a = " << ++a << endl;       // 11
    cout << "a++ = " << a++ << endl;       // 11（返回旧值）
    cout << "现在 a = " << a << endl;      // 12
    cout << "--a = " << --a << endl;       // 11
    cout << "a-- = " << a-- << endl;       // 11（返回旧值）
    cout << "现在 a = " << a << endl;      // 10

    return 0;
}
```

## 二、赋值 = 运算符重载

### 2.1 为什么需要重载赋值运算符？

**问题：浅拷贝 vs 深拷贝**

```cpp
#include <iostream>
#include <cstring>
using namespace std;

class Person {
public:
    char *m_Name;  // 用指针管理动态内存
    int m_Age;

    Person(const char *name, int age) {
        m_Age = age;
        // 深拷贝：分配新内存，复制字符串内容
        m_Name = new char[strlen(name) + 1];
        strcpy(m_Name, name);
        cout << "构造函数被调用: " << m_Name << endl;
    }

    // 析构函数：释放动态内存
    ~Person() {
        cout << "析构函数被调用: " << m_Name << endl;
        if (m_Name != nullptr) {
            delete[] m_Name;
            m_Name = nullptr;
        }
    }
};

int main() {
    Person p1("张三", 18);

    // 问题：编译器默认的赋值操作是浅拷贝（逐字节复制）
    // 浅拷贝的结果：p2.m_Name 和 p1.m_Name 指向同一块内存
    Person p2("李四", 20);
    p2 = p1;  // 浅拷贝！p2.m_Name 现在指向 p1.m_Name 的内存

    // 当 p1 和 p2 都被销毁时，同一块内存会被释放两次 → 程序崩溃！
    return 0;
}
```

**浅拷贝的问题图示：**

```
浅拷贝前：
p1.m_Name → [张三\0]   （p1 拥有的内存）
p2.m_Name → [李四\0]   （p2 拥有的内存）

浅拷贝后（p2 = p1）：
p1.m_Name → [张三\0] ← p2.m_Name  （两个指针指向同一块内存！）
             ↑
             这块内存会被 delete 两次 → 崩溃！
```

---

### 2.2 正确的赋值运算符重载

```cpp
#include <iostream>
#include <cstring>
using namespace std;

class Person {
public:
    char *m_Name;
    int m_Age;

    // 构造函数
    Person(const char *name = "", int age = 0) {
        m_Age = age;
        m_Name = new char[strlen(name) + 1];
        strcpy(m_Name, name);
        cout << "构造: " << m_Name << endl;
    }

    // 析构函数
    ~Person() {
        cout << "析构: " << m_Name << endl;
        if (m_Name != nullptr) {
            delete[] m_Name;
            m_Name = nullptr;
        }
    }

    // 拷贝构造函数（深拷贝）
    Person(const Person &other) {
        m_Age = other.m_Age;
        m_Name = new char[strlen(other.m_Name) + 1];
        strcpy(m_Name, other.m_Name);
        cout << "拷贝构造: " << m_Name << endl;
    }

    // 赋值运算符重载
    // 返回值是 Person&，支持链式赋值（p3 = p2 = p1）
    // 参数是 const Person&，避免不必要的拷贝
    Person& operator=(const Person &p) {
        cout << "赋值运算符被调用" << endl;

        // 第1步：防止自赋值（p = p 的情况）
        if (this == &p) {
            return *this;
        }

        // 第2步：先释放原有内存（避免内存泄漏）
        if (this->m_Name != nullptr) {
            delete[] this->m_Name;
            this->m_Name = nullptr;
        }

        // 第3步：深拷贝（分配新内存，复制内容）
        this->m_Age = p.m_Age;
        this->m_Name = new char[strlen(p.m_Name) + 1];
        strcpy(this->m_Name, p.m_Name);

        // 第4步：返回自身引用（支持链式赋值）
        return *this;
    }

    // 显示信息
    void showInfo() {
        cout << "姓名: " << m_Name << ", 年龄: " << m_Age << endl;
    }
};

int main() {
    cout << "=== 创建对象 ===" << endl;
    Person p1("张三", 18);
    Person p2("李四", 20);
    Person p3("王五", 25);

    cout << "\n=== 赋值操作 ===" << endl;
    // 单次赋值
    p2 = p1;
    cout << "p2 赋值后: ";
    p2.showInfo();  // 姓名: 张三, 年龄: 18

    cout << "\n=== 链式赋值 ===" << endl;
    // 链式赋值：p3 = p2 = p1
    // 等价于：p3 = (p2 = p1)
    // p2 = p1 返回 p2 的引用，然后 p3 = p2的引用
    p3 = p2 = p1;
    cout << "p2: ";
    p2.showInfo();  // 姓名: 张三, 年龄: 18
    cout << "p3: ";
    p3.showInfo();  // 姓名: 张三, 年龄: 18

    cout << "\n=== 自赋值测试 ===" << endl;
    p1 = p1;  // 自赋值，不会出错
    cout << "p1 自赋值后: ";
    p1.showInfo();  // 姓名: 张三, 年龄: 18

    return 0;
}
```

### 2.3 赋值运算符重载的步骤详解

```
Person& operator=(const Person &p)
    ↓
第1步：防止自赋值
    if (this == &p) return *this;
    ↓
第2步：释放原有内存
    delete[] this->m_Name;
    ↓
第3步：深拷贝
    this->m_Name = new char[strlen(p.m_Name) + 1];
    strcpy(this->m_Name, p.m_Name);
    ↓
第4步：返回自身引用
    return *this;
```

**为什么要返回引用？**

```cpp
// 支持链式赋值
p3 = p2 = p1;

// 等价于：
// p2.operator=(p1)  →  返回 p2 的引用
// p3.operator=(p2的引用)  →  返回 p3 的引用

// 如果返回 void，就无法链式赋值
```

**为什么要防止自赋值？**

```cpp
Person p1("张三", 18);
p1 = p1;  // 自赋值

// 如果不检查自赋值：
// 第1步：delete[] this->m_Name;  →  删除了 "张三"
// 第2步：strcpy(this->m_Name, p.m_Name);  →  p.m_Name 已经被删除了！崩溃！

// 加了自赋值检查：
// if (this == &p) return *this;  →  直接返回，不做任何操作
```

### 2.4 不需要手动释放内存时的简化版本

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
public:
    string m_Name;  // 使用 string 而不是 char*，不需要手动管理内存
    int m_Age;

    Person(string name = "", int age = 0) : m_Name(name), m_Age(age) {}

    // 赋值运算符重载（简化版）
    // 当成员变量不涉及动态内存时，编译器默认的赋值操作就是正确的
    // 但为了养成好习惯，可以显式重载
    Person& operator=(const Person &p) {
        // 防止自赋值
        if (this == &p) {
            return *this;
        }

        // 直接赋值（string 的赋值操作符已经处理了内存管理）
        this->m_Name = p.m_Name;
        this->m_Age = p.m_Age;

        return *this;
    }

    void showInfo() {
        cout << "姓名: " << m_Name << ", 年龄: " << m_Age << endl;
    }
};

int main() {
    Person p1("张三", 18);
    Person p2("李四", 20);

    p2 = p1;
    p2.showInfo();  // 姓名: 张三, 年龄: 18

    // 链式赋值
    Person p3;
    p3 = p2 = p1;
    p3.showInfo();  // 姓名: 张三, 年龄: 18

    return 0;
}
```

### 2.5 返回引用 vs 返回值的区别

```cpp
#include <iostream>
using namespace std;

class MyInt {
    friend ostream& operator<<(ostream &out, const MyInt &obj);

private:
    int m_Value;

public:
    MyInt(int v = 0) : m_Value(v) {}

    // 返回引用：可以作为左值
    MyInt& operator=(const MyInt &other) {
        if (this == &other) return *this;
        this->m_Value = other.m_Value;
        return *this;
    }
};

ostream& operator<<(ostream &out, const MyInt &obj) {
    out << obj.m_Value;
    return out;
}

int main() {
    MyInt a(10), b(20), c(30);

    // 链式赋值
    a = b = c;  // 等价于 a = (b = c)
    cout << "a=" << a << ", b=" << b << ", c=" << c << endl;  // a=30, b=30, c=30

    // 返回引用意味着赋值表达式的结果是左值引用
    // (a = b) = c;  // 先把 b 赋给 a，再把 c 赋给 (a = b 的结果，即 a)
    // 这种写法虽然合法但不推荐，容易让人困惑

    return 0;
}
```

## 三、综合示例

```cpp
#include <iostream>
#include <cstring>
using namespace std;

class MyString {
    friend ostream& operator<<(ostream &out, const MyString &str);

private:
    char *m_Data;   // 字符串数据
    int m_Length;   // 字符串长度

public:
    // 构造函数
    MyString(const char *data = "") {
        m_Length = strlen(data);
        m_Data = new char[m_Length + 1];
        strcpy(m_Data, data);
    }

    // 析构函数
    ~MyString() {
        if (m_Data != nullptr) {
            delete[] m_Data;
            m_Data = nullptr;
        }
    }

    // 拷贝构造函数（深拷贝）
    MyString(const MyString &other) {
        m_Length = other.m_Length;
        m_Data = new char[m_Length + 1];
        strcpy(m_Data, other.m_Data);
    }

    // 赋值运算符重载
    MyString& operator=(const MyString &other) {
        if (this == &other) return *this;

        // 释放原有内存
        if (m_Data != nullptr) {
            delete[] m_Data;
            m_Data = nullptr;
        }

        // 深拷贝
        m_Length = other.m_Length;
        m_Data = new char[m_Length + 1];
        strcpy(m_Data, other.m_Data);

        return *this;
    }
};

// 重载 << 运算符
ostream& operator<<(ostream &out, const MyString &str) {
    out << str.m_Data;
    return out;
}

int main() {
    MyString s1("Hello");
    MyString s2("World");
    MyString s3;

    cout << "s1 = " << s1 << endl;  // Hello
    cout << "s2 = " << s2 << endl;  // World

    // 测试赋值
    s3 = s1;
    cout << "s3 = s1 后, s3 = " << s3 << endl;  // Hello

    // 测试链式赋值
    s3 = s2 = s1;
    cout << "链式赋值后:" << endl;
    cout << "s1 = " << s1 << endl;  // Hello
    cout << "s2 = " << s2 << endl;  // Hello
    cout << "s3 = " << s3 << endl;  // Hello

    // 测试自赋值
    s1 = s1;
    cout << "自赋值后, s1 = " << s1 << endl;  // Hello

    return 0;
}
```

## 四、常见错误

### 错误1：后置 ++ 返回引用

```cpp
// 错误：后置 ++ 不应该返回引用
MyInteger& operator++(int) {
    MyInteger temp = *this;
    m_Value++;
    return temp;  // 错误！temp 是局部变量，函数结束后就销毁了
    // 返回的是一个已经销毁的对象的引用 → 未定义行为！
}

// 正确：后置 ++ 返回值（不是引用）
MyInteger operator++(int) {
    MyInteger temp = *this;
    m_Value++;
    return temp;  // 正确！返回的是 temp 的拷贝
}
```

### 错误2：赋值时忘记释放原有内存

```cpp
// 错误：没有释放原有内存，导致内存泄漏
Person& operator=(const Person &p) {
    if (this == &p) return *this;

    // 错误：直接分配新内存，没有先释放旧内存
    this->m_Name = new char[strlen(p.m_Name) + 1];  // 旧内存泄漏了！
    strcpy(this->m_Name, p.m_Name);

    return *this;
}

// 正确：先释放旧内存，再分配新内存
Person& operator=(const Person &p) {
    if (this == &p) return *this;

    // 先释放旧内存
    if (this->m_Name != nullptr) {
        delete[] this->m_Name;
        this->m_Name = nullptr;
    }

    // 再分配新内存并拷贝
    this->m_Name = new char[strlen(p.m_Name) + 1];
    strcpy(this->m_Name, p.m_Name);

    return *this;
}
```

### 错误3：赋值时忘记自赋值检查

```cpp
// 错误：没有自赋值检查
Person& operator=(const Person &p) {
    delete[] this->m_Name;  // 如果是自赋值，p.m_Name 也被删除了！
    this->m_Name = new char[strlen(p.m_Name) + 1];  // p.m_Name 已经是野指针！
    strcpy(this->m_Name, p.m_Name);  // 崩溃！

    return *this;
}
```

## 五、总结

| 运算符 | 语法 | 返回值 | 要点 |
|--------|------|--------|------|
| 前置 ++ | `operator++()` | `MyInteger&` | 先自增，返回引用，效率高 |
| 后置 ++ | `operator++(int)` | `MyInteger` | int 占位区分，先保存旧值再自增 |
| 前置 -- | `operator--()` | `MyInteger&` | 同前置 ++ |
| 后置 -- | `operator--(int)` | `MyInteger` | 同后置 ++ |
| 赋值 = | `operator=(const T&)` | `T&` | 防自赋值、释放旧内存、深拷贝、返回引用 |

**关键记忆点：**
1. 前置 `++` 返回**引用**，后置 `++` 返回**值**
2. 后置 `++` 的 `int` 参数是**占位参数**，编译器自动传入 0
3. 赋值 `=` 必须处理**自赋值**和**深拷贝**问题
4. 赋值 `=` 返回**引用**，支持链式赋值
5. 优先使用前置 `++`，效率更高
