# 08 字符串型详解

## 一、C++ 中两种字符串

| 类型 | 来源 | 头文件 | 推荐度 |
|------|------|--------|--------|
| C 风格字符串 | C 语言 | 无需额外头文件 | 不推荐 |
| C++ string 类 | C++ STL | `#include <string>` | 强烈推荐 |

---

## 二、C 风格字符串（char 数组）

### 2.1 基本语法

```cpp
// 本质是一个 char 类型的数组，以 '\0'（空字符）结尾
// 语法：char 变量名[] = "字符串内容";

char str1[] = "Hello";     // 自动在末尾加 '\0'，实际占 6 字节
char str2[] = {'H', 'i', '\0'};  // 手动加 '\0'

cout << str1 << endl;  // 输出: Hello
cout << str2 << endl;  // 输出: Hi
```

### 2.2 '\0' 结尾的坑（非常重要！）

```cpp
// '\0' 是字符串结束标志，cout 输出时遇到 '\0' 就停止
char str[] = "Hello";
// 内存中实际存储: ['H']['e']['l']['l']['o']['\0']
// 大小: 6 字节（5个可见字符 + 1个 '\0'）

// 坑1：忘记 '\0' 会导致输出乱码
char bad[] = {'H', 'e', 'l', 'l', 'o'};
// 没有 '\0'！cout 会继续往后读内存，直到碰到某个 '\0'，输出不可预测
cout << bad << endl;  // 可能输出 Hello 后面跟一堆乱码

// 坑2：数组大小不够
// char small[3] = "Hello";  // 错误！"Hello" 需要 6 字节，数组只有 3 字节

// 坑3：sizeof 和 strlen 的区别
char s[] = "Hello";
cout << sizeof(s) << endl;  // 6（包含 '\0'）
cout << strlen(s) << endl;  // 5（不包含 '\0'，需要 #include <cstring>）
```

### 2.3 C 风格字符串的常见操作

```cpp
#include <iostream>
#include <cstring>  // C 风格字符串操作函数
using namespace std;

int main()
{
    char str1[20] = "Hello";
    char str2[] = " World";

    // 1. 求长度（不含 '\0'）
    cout << "长度: " << strlen(str1) << endl;  // 5

    // 2. 拼接（目标数组要足够大！）
    strcat(str1, str2);  // str1 变成 "Hello World"
    cout << "拼接后: " << str1 << endl;

    // 3. 比较（逐字符比较 ASCII 码）
    char a[] = "abc";
    char b[] = "abd";
    int result = strcmp(a, b);  // 返回负数（因为 'c' < 'd'）
    if (result == 0)
        cout << "相等" << endl;
    else if (result < 0)
        cout << a << " < " << b << endl;
    else
        cout << a << " > " << b << endl;

    // 4. 拷贝
    char dest[20];
    strcpy(dest, "Copy this");
    cout << "拷贝后: " << dest << endl;

    return 0;
}
```

### 2.4 C 风格字符串的问题

```cpp
// 问题1：容易越界（数组大小不够）
char name[5] = "Hello";  // 危险！'\0' 会写到数组外面

// 问题2：不能直接用 = 赋值
char str[20];
// str = "Hello";       // 错误！数组不能整体赋值
strcpy(str, "Hello");   // 只能用 strcpy 函数

// 问题3：不能直接用 == 比较
char a[] = "Hello";
char b[] = "Hello";
// if (a == b)  // 错误！这是比较两个数组的地址，不是比较内容
if (strcmp(a, b) == 0)  // 正确：用 strcmp 函数比较内容
{
    cout << "内容相同" << endl;
}

// 问题4：操作不直观，代码量大
```

## 三、C++ string 类（推荐！）

### 3.1 基本用法

```cpp
#include <iostream>
#include <string>  // 必须包含这个头文件
using namespace std;

int main()
{
    // 创建字符串
    string s1 = "Hello";
    string s2("World");
    string s3 = string("C++");

    // 输出
    cout << s1 << endl;  // Hello
    cout << s2 << endl;  // World

    return 0;
}
```

### 3.2 赋值操作

```cpp
string s;

// 直接赋值（不像 char 数组那样麻烦）
s = "Hello";
cout << s << endl;  // Hello

s = "New value";    // 可以随时重新赋值
cout << s << endl;  // New value

// 用另一个字符串赋值
string s2 = s;      // s2 的内容和 s 一样
```

### 3.3 拼接操作

```cpp
string s1 = "Hello";
string s2 = " World";

// 方式1：用 + 运算符
string s3 = s1 + s2;
cout << s3 << endl;  // Hello World

// 方式2：用 += 运算符
s1 += s2;
cout << s1 << endl;  // Hello World

// 可以和 C 风格字符串混合使用
string s4 = "Hello";
s4 += " C++";
cout << s4 << endl;  // Hello C++

// 也可以和单个字符拼接
s4 += '!';
cout << s4 << endl;  // Hello C++!
```

### 3.4 比较操作

```cpp
string a = "abc";
string b = "abd";

// 直接用比较运算符（不用 strcmp 那么麻烦）
if (a == b)  cout << "相等" << endl;
if (a != b)  cout << "不相等" << endl;
if (a < b)   cout << a << " < " << b << endl;   // abc < abd
if (a > b)   cout << a << " > " << b << endl;

// 比较规则：逐字符比较 ASCII 码
// "abc" vs "abd"：前两个字符相同，第三个字符 'c'(99) < 'd'(100)
// 所以 "abc" < "abd"
```

### 3.5 求长度

```cpp
string s = "Hello";

// 方式1：.length()
cout << "长度: " << s.length() << endl;  // 5

// 方式2：.size()（和 length() 完全一样，推荐用 size()）
cout << "大小: " << s.size() << endl;    // 5

// 判断是否为空
string empty = "";
if (empty.empty())  // 或者 empty.size() == 0
{
    cout << "字符串为空" << endl;
}
```

### 3.6 访问单个字符

```cpp
string s = "Hello";

// 方式1：用 [] 下标访问（从 0 开始）
cout << s[0] << endl;  // H
cout << s[1] << endl;  // e
cout << s[4] << endl;  // o

// 方式2：用 .at() 方法（更安全，会检查越界）
cout << s.at(0) << endl;  // H

// 区别：[] 不检查越界，at() 会检查并抛出异常
// cout << s[100];     // 不报错，但结果未定义（危险！）
// cout << s.at(100);  // 抛出 out_of_range 异常

// 修改单个字符
s[0] = 'h';  // 把 'H' 改成 'h'
cout << s << endl;  // hello
```

### 3.7 遍历字符串

```cpp
string s = "Hello";

// 方式1：for 循环 + 下标
for (int i = 0; i < s.size(); i++)
{
    cout << s[i] << " ";
}
cout << endl;
// 输出: H e l l o

// 方式2：范围 for（C++11，推荐）
for (char ch : s)
{
    cout << ch << " ";
}
cout << endl;
// 输出: H e l l o
```

### 3.8 子串和查找

```cpp
string s = "Hello C++ World";

// 截取子串：substr(起始位置, 长度)
string sub = s.substr(6, 3);  // 从位置6开始，取3个字符
cout << sub << endl;  // C++

// 查找：find() 返回第一次出现的位置，找不到返回 string::npos
int pos = s.find("C++");
if (pos != string::npos)
{
    cout << "找到了，在位置: " << pos << endl;  // 6
}

// 替换：replace(起始位置, 长度, 新字符串)
s.replace(6, 3, "Java");
cout << s << endl;  // Hello Java World

// 插入：insert(位置, 字符串)
s.insert(6, "Beautiful ");
cout << s << endl;  // Hello Beautiful Java World

// 删除：erase(起始位置, 长度)
s.erase(6, 10);
cout << s << endl;  // Hello Java World
```

### 3.9 输入字符串

```cpp
string s;

// 方式1：cin（遇到空格就停止）
cout << "请输入（cin）: ";
cin >> s;
cout << "你输入了: " << s << endl;
// 输入 "Hello World"，只读到 "Hello"

// 方式2：getline（读取整行，包括空格）
cout << "请输入（getline）: ";
getline(cin, s);
cout << "你输入了: " << s << endl;
// 输入 "Hello World"，读到完整的 "Hello World"
```

## 四、C 风格字符串 vs C++ string 对比

| 对比项 | char 数组 | string |
|--------|-----------|--------|
| 头文件 | 不需要 | `#include <string>` |
| 赋值 | 需要 `strcpy` | 直接 `=` |
| 拼接 | 需要 `strcat` | 直接 `+` 或 `+=` |
| 比较 | 需要 `strcmp` | 直接 `==` `!=` `<` `>` |
| 求长度 | 需要 `strlen` | `.size()` 或 `.length()` |
| 安全性 | 容易越界 | 自动管理内存 |
| 读取整行 | `cin.getline()` | `getline(cin, s)` |
| 推荐度 | 不推荐 | 强烈推荐 |

## 五、C 风格字符串与 string 的转换

```cpp
// C 风格字符串 -> string（自动转换）
const char* cstr = "Hello";
string s = cstr;        // 直接赋值即可
string s2(cstr);        // 构造函数方式

// string -> C 风格字符串
string s3 = "World";
const char* cstr2 = s3.c_str();  // 返回 const char* 指针
cout << cstr2 << endl;

// 注意：c_str() 返回的指针指向 string 内部的内存
// 如果 string 被修改或销毁，指针就失效了
```

## 六、综合示例

```cpp
#include <iostream>
#include <string>
using namespace std;

int main()
{
    cout << "===== 字符串综合示例 =====" << endl;

    // 1. 基本操作
    string firstName = "张";
    string lastName = "三";
    string fullName = lastName + firstName;  // 拼接
    cout << "姓名: " << fullName << endl;
    cout << "姓名长度: " << fullName.size() << endl;

    // 2. 输入输出
    string input;
    cout << "\n请输入一句话: ";
    getline(cin, input);  // 读取整行
    cout << "你输入了: " << input << endl;
    cout << "长度: " << input.size() << " 个字符" << endl;

    // 3. 查找和替换
    string text = "I love C++ and C++ is great!";
    int pos = text.find("C++");
    if (pos != string::npos)
    {
        cout << "\n第一次出现 C++ 的位置: " << pos << endl;
    }

    // 替换所有的 C++ 为 Java
    while ((pos = text.find("C++")) != string::npos)
    {
        text.replace(pos, 3, "Java");
    }
    cout << "替换后: " << text << endl;

    // 4. 遍历并统计字符
    string str = "Hello World";
    int letterCount = 0;
    int spaceCount = 0;
    for (char ch : str)
    {
        if (ch == ' ')
            spaceCount++;
        else
            letterCount++;
    }
    cout << "\n字符串 \"" << str << "\" 中:" << endl;
    cout << "字母个数: " << letterCount << endl;
    cout << "空格个数: " << spaceCount << endl;

    return 0;
}
```

## 七、总结

| 要点 | 说明 |
|------|------|
| C 风格 | `char str[] = "Hello"`，以 `'\0'` 结尾，坑多，不推荐 |
| C++ 风格 | `string s = "Hello"`，操作方便安全，强烈推荐 |
| 头文件 | 使用 string 必须 `#include <string>` |
| 输入整行 | 用 `getline(cin, s)` 而不是 `cin >> s` |
| 核心操作 | 赋值 `=`、拼接 `+`、比较 `==`、长度 `.size()`、访问 `[]` |
