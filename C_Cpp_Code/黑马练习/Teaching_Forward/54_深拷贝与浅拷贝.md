# 深拷贝与浅拷贝

## 一、什么是浅拷贝

**浅拷贝**是编译器默认提供的拷贝构造函数所做的操作：**逐字节复制**。

- 值类型（int、double 等）：直接复制值
- 指针类型：只复制**地址值**，不复制指向的数据

```cpp
class Person {
public:
    string name;
    int age;
    int *height;    // 指针成员，指向堆区

    Person(string n, int a, int h) {
        name = n;
        age = a;
        height = new int(h);    // 在堆区分配内存
    }

    ~Person() {
        delete height;          // 释放堆区内存
        height = nullptr;
    }
};

int main() {
    Person p1("张三", 25, 175);
    Person p2(p1);   // 调用编译器默认的拷贝构造（浅拷贝）
    // 问题来了...
    return 0;
}
```

## 二、浅拷贝的问题

### 问题图解

```
浅拷贝后：

p1:  name="张三"  age=25  height=0x1000 ──→ [175] (堆区)
p2:  name="张三"  age=25  height=0x1000 ──→ [175] (堆区，同一块！)

两个指针指向同一块堆区内存！
```

### 后果：同一块内存被释放两次

```cpp
#include <iostream>
using namespace std;

class Person {
public:
    string name;
    int *height;    // 指向堆区的指针

    Person(string n, int h) {
        name = n;
        height = new int(h);    // 堆区分配内存
        cout << "[构造] " << name << "，height地址：" << height << endl;
    }

    ~Person() {
        cout << "[析构] " << name << "，释放地址：" << height << endl;
        delete height;
        height = nullptr;
    }
};

int main() {
    Person p1("张三", 175);
    Person p2(p1);   // 浅拷贝：p2.height 和 p1.height 指向同一块内存

    cout << "p1.height 地址：" << p1.height << endl;
    cout << "p2.height 地址：" << p2.height << endl;
    // 两个地址相同！

    return 0;
    // p2 先析构：释放了堆区内存
    // p1 再析构：又试图释放同一块内存 → 程序崩溃！
}
```

**运行结果（可能崩溃）：**
```
[构造] 张三，height地址：0x6e1b70
p1.height 地址：0x6e1b70
p2.height 地址：0x6e1b70     ← 两个指针指向同一块内存！
[析构] 张三，释放地址：0x6e1b70   ← p2 先析构，释放成功
[析构] 张三，释放地址：0x6e1b70   ← p1 再析构，同一块内存被释放两次 → 崩溃！
```

## 三、什么是深拷贝

**深拷贝**是手动实现的拷贝构造函数：**重新在堆区开辟新内存，复制数据**。

两个对象各自管理自己的堆区内存，互不影响。

### 深拷贝图解

```
深拷贝后：

p1:  name="张三"  age=25  height=0x1000 ──→ [175] (堆区)
p2:  name="张三"  age=25  height=0x2000 ──→ [175] (堆区，新开辟的！)

两个指针指向不同的堆区内存，各自独立！
```

## 四、深拷贝的代码实现

```cpp
#include <iostream>
using namespace std;

class Person {
public:
    string name;
    int age;
    int *height;    // 指向堆区的指针

    // 有参构造
    Person(string n, int a, int h) {
        name = n;
        age = a;
        height = new int(h);    // 在堆区分配内存
        cout << "[构造] " << name << endl;
    }

    // ========== 深拷贝构造函数 ==========
    Person(const Person &p) {
        name = p.name;
        age = p.age;
        // 关键：重新在堆区分配内存，复制数据
        height = new int(*p.height);
        cout << "[深拷贝] 复制了 " << name << endl;
    }

    // 析构函数
    ~Person() {
        delete height;
        height = nullptr;
        cout << "[析构] " << name << endl;
    }

    void show() {
        cout << "  " << name << "，年龄" << age << "，身高" << *height << "cm" << endl;
    }
};

int main() {
    cout << "===== 创建 p1 =====" << endl;
    Person p1("张三", 25, 175);

    cout << "\n===== 深拷贝创建 p2 =====" << endl;
    Person p2(p1);   // 深拷贝

    cout << "\n===== 验证地址不同 =====" << endl;
    cout << "p1.height 地址：" << p1.height << endl;
    cout << "p2.height 地址：" << p2.height << endl;

    cout << "\n===== 修改 p2 的数据 =====" << endl;
    *p2.height = 180;    // 修改 p2 的身高
    p1.show();           // p1 身高不变
    p2.show();           // p2 身高变了

    cout << "\n===== main 结束 =====" << endl;
    return 0;
    // p2 先析构：释放 p2 的堆区内存
    // p1 再析构：释放 p1 的堆区内存（不同的内存，不会崩溃）
}
```

**输出：**
```
===== 创建 p1 =====
[构造] 张三

===== 深拷贝创建 p2 =====
[深拷贝] 复制了 张三

===== 验证地址不同 =====
p1.height 地址：0x721b70
p2.height 地址：0x721b90    ← 地址不同！

===== 修改 p2 的数据 =====
  张三，年龄25，身高175cm    ← p1 不受影响
  张三，年龄25，身高180cm    ← p2 改了

===== main 结束 =====
[析构] 张三                  ← p2 先析构
[析构] 张三                  ← p1 后析构，不会崩溃
```

## 五、浅拷贝 vs 深拷贝 对比

```cpp
#include <iostream>
using namespace std;

class Array {
public:
    int *data;      // 指向堆区数组
    int size;

    // 构造函数
    Array(int s) {
        size = s;
        data = new int[size];   // 堆区分配数组
        for (int i = 0; i < size; i++) {
            data[i] = 0;        // 初始化为0
        }
        cout << "[构造] 分配了 " << size << " 个int" << endl;
    }

    // 深拷贝构造函数
    Array(const Array &other) {
        size = other.size;
        data = new int[size];   // 重新分配内存
        for (int i = 0; i < size; i++) {
            data[i] = other.data[i];    // 复制数据
        }
        cout << "[深拷贝] 复制了 " << size << " 个int" << endl;
    }

    // 析构函数
    ~Array() {
        delete[] data;
        data = nullptr;
        cout << "[析构] 释放内存" << endl;
    }

    // 设置值
    void set(int index, int value) {
        if (index >= 0 && index < size) {
            data[index] = value;
        }
    }

    // 打印数组
    void print() {
        cout << "  [";
        for (int i = 0; i < size; i++) {
            cout << data[i];
            if (i < size - 1) cout << ", ";
        }
        cout << "]" << endl;
    }
};

int main() {
    Array a1(5);
    a1.set(0, 10);
    a1.set(1, 20);
    a1.set(2, 30);

    cout << "a1: ";
    a1.print();

    Array a2(a1);   // 深拷贝

    cout << "a2: ";
    a2.print();

    // 修改 a2，不影响 a1
    a2.set(0, 999);
    cout << "\n修改 a2[0]=999 后：" << endl;
    cout << "a1: ";
    a1.print();     // a1 不受影响
    cout << "a2: ";
    a2.print();     // a2 被修改

    return 0;
}
```

## 六、什么时候需要写深拷贝

**当类中有指针成员指向堆区时，必须手写深拷贝构造函数。**

```cpp
// 需要写深拷贝的情况：
class NeedDeepCopy {
public:
    int *data;              // 指针指向堆区
    string *name;           // 指针指向堆区
    int *array;             // 指针指向堆区数组

    // 必须手写拷贝构造，进行深拷贝
    NeedDeepCopy(const NeedDeepCopy &other) {
        data = new int(*other.data);
        name = new string(*other.name);
        array = new int[10];
        for (int i = 0; i < 10; i++) {
            array[i] = other.array[i];
        }
    }
};

// 不需要写深拷贝的情况（默认浅拷贝就够了）：
class NoNeedDeepCopy {
public:
    int age;                // 值类型
    double score;           // 值类型
    string name;            // string 类内部自己管理内存
    // 没有指针指向堆区，默认浅拷贝足够
};
```

## 七、常见错误

### 错误1：忘记写深拷贝，导致崩溃

```cpp
class Bad {
public:
    int *data;

    Bad(int val) {
        data = new int(val);
    }

    // 忘记写拷贝构造！编译器用默认的浅拷贝
    // ~Bad() { delete data; }  // 析构时会 double free

    ~Bad() {
        delete data;
    }
};

int main() {
    Bad b1(10);
    Bad b2(b1);     // 浅拷贝，两个对象的 data 指向同一块内存
    return 0;
    // 崩溃！同一块内存被释放两次
}
```

### 错误2：拷贝构造中忘记释放旧内存

```cpp
class BadCopy {
public:
    int *data;

    BadCopy(int val) {
        data = new int(val);
    }

    BadCopy(const BadCopy &other) {
        // ❌ 错误：如果 data 已经有值，应该先释放
        // delete data;   // 如果需要，先释放旧内存
        data = new int(*other.data);
    }

    ~BadCopy() {
        delete data;
    }
};
```

### 错误3：拷贝构造中只复制了指针

```cpp
class BadCopy {
public:
    int *data;

    BadCopy(const BadCopy &other) {
        data = other.data;          // ❌ 只复制了指针（浅拷贝）
        // data = new int(*other.data);  // ✅ 应该深拷贝
    }
};
```

## 八、完整深拷贝模板

```cpp
#include <iostream>
using namespace std;

class MyString {
private:
    char *m_str;    // 指向堆区的字符串
    int m_length;

public:
    // 构造函数
    MyString(const char *str) {
        m_length = strlen(str);
        m_str = new char[m_length + 1];    // +1 给 '\0'
        strcpy(m_str, str);
        cout << "[构造] " << m_str << endl;
    }

    // 深拷贝构造函数
    MyString(const MyString &other) {
        m_length = other.m_length;
        m_str = new char[m_length + 1];    // 重新分配内存
        strcpy(m_str, other.m_str);         // 复制数据
        cout << "[深拷贝] " << m_str << endl;
    }

    // 析构函数
    ~MyString() {
        cout << "[析构] " << (m_str ? m_str : "nullptr") << endl;
        delete[] m_str;
        m_str = nullptr;
    }

    // 显示
    void show() {
        cout << "  内容：" << m_str << "，长度：" << m_length << endl;
    }
};

int main() {
    MyString s1("Hello");
    s1.show();

    MyString s2(s1);    // 深拷贝
    s2.show();

    cout << "\ns1 地址：" << (void*)s1.getStr() << endl;
    cout << "s2 地址：" << (void*)s2.getStr() << endl;

    return 0;
}
```

## 九、总结

| 对比项 | 浅拷贝 | 深拷贝 |
|--------|--------|--------|
| 实现方式 | 编译器默认提供 | 手动实现拷贝构造 |
| 指针处理 | 只复制地址值 | 重新分配内存，复制数据 |
| 堆区内存 | 两个对象共享同一块 | 两个对象各自独立 |
| 析构问题 | 同一块内存释放两次→崩溃 | 各自释放各自的，不会崩溃 |
| 使用场景 | 类中没有指针成员时 | 类中有指针指向堆区时 |

**一句话总结：类中有指针指向堆区，就必须手写深拷贝构造函数。**
