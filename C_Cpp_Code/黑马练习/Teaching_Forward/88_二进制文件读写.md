# 二进制文件读写

## 1. 概述

文件分为两种类型：**文本文件**和**二进制文件**。

- **文本文件**：存储的是字符的ASCII码，可以用记事本打开查看。比如数字100会存储为字符'1''0''0'三个字节。
- **二进制文件**：存储的是数据的原始二进制格式，不能直接用记事本查看。比如数字100会存储为4个字节的二进制表示。

---

## 2. 文本文件 vs 二进制文件

### 2.1 存储方式对比

```cpp
#include <iostream>
using namespace std;

int main() {
    int num = 100;

    // 文本方式存储：
    // 数字100 -> 字符 '1', '0', '0' -> ASCII码 49, 48, 48 -> 占3个字节

    // 二进制方式存储：
    // 数字100 -> 二进制 00000000 00000000 00000000 01100100 -> 占4个字节

    cout << "int 类型占 " << sizeof(int) << " 个字节" << endl;

    return 0;
}
```

### 2.2 两种文件的对比

| 对比项 | 文本文件 | 二进制文件 |
|--------|----------|------------|
| 可读性 | 可以用记事本查看 | 记事本打开是乱码 |
| 存储方式 | 字符形式 | 原始二进制形式 |
| 读写速度 | 较慢（需要转换） | 较快（直接读写） |
| 精度 | 可能有精度损失 | 保持原始精度 |
| 跨平台 | 兼容性好 | 可能有兼容性问题 |
| 用途 | 配置文件、日志 | 图片、音视频、数据库 |

## 3. 二进制写入

### 3.1 基本语法

```cpp
#include <iostream>
#include <fstream>
using namespace std;

int main() {
    // 以二进制模式打开文件进行写入
    ofstream ofs("data.dat", ios::out | ios::binary);

    if (!ofs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    int num = 100;

    // write 函数的参数：
    // 第一个参数：要写入数据的地址（需要强制转换为 const char*）
    // 第二个参数：要写入的字节数
    ofs.write((const char *)&num, sizeof(num));

    ofs.close();
    cout << "写入完成！" << endl;

    return 0;
}
```

### 3.2 写入结构体

```cpp
#include <iostream>
#include <fstream>
#include <cstring>  // 用于 strcpy
using namespace std;

// 定义一个结构体
struct Student {
    char name[50];  // 注意：要用 char 数组，不能用 string
    int age;
    double score;
};

int main() {
    // 创建学生数据
    Student stu1;
    strcpy(stu1.name, "张三");  // 用 strcpy 复制字符串
    stu1.age = 20;
    stu1.score = 95.5;

    // 以二进制模式写入
    ofstream ofs("student.dat", ios::out | ios::binary);

    if (!ofs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    // 写入结构体
    ofs.write((const char *)&stu1, sizeof(Student));

    ofs.close();
    cout << "学生信息写入完成！" << endl;

    return 0;
}
```

### 3.3 写入多个数据


## 4. 二进制读取

### 4.1 基本语法

```cpp
#include <iostream>
#include <fstream>
using namespace std;

int main() {
    // 假设之前已经写入了 data.dat

    // 以二进制模式打开文件进行读取
    ifstream ifs("data.dat", ios::in | ios::binary);

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    int num;

    // read 函数的参数：
    // 第一个参数：存放数据的地址（需要强制转换为 char*）
    // 第二个参数：要读取的字节数
    ifs.read((char *)&num, sizeof(num));

    cout << "读取到的数字：" << num << endl;

    ifs.close();
    return 0;
}
```

### 4.2 读取结构体

```cpp
#include <iostream>
#include <fstream>
using namespace std;

struct Student {
    char name[50];
    int age;
    double score;
};

int main() {
    // 以二进制模式读取
    ifstream ifs("student.dat", ios::in | ios::binary);

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    Student stu;

    // 读取结构体
    ifs.read((char *)&stu, sizeof(Student));

    cout << "姓名：" << stu.name << endl;
    cout << "年龄：" << stu.age << endl;
    cout << "分数：" << stu.score << endl;

    ifs.close();
    return 0;
}
```

### 4.3 读取多个数据

```cpp
#include <iostream>

struct Student {

        cout << "  年龄：" << students[i].age << endl;
        cout << "  分数：" << students[i].score << endl;
    }

    return 0;

```

## 5. 重要注意事项

### 5.1 不要使用 string 类型

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

// 错误示范：使用 string 类型
struct BadStudent {
    string name;  // 错误！string 内部有指针，不能直接二进制写入
    int age;
};

// 正确示范：使用 char 数组
struct GoodStudent {
    char name[50];  // 正确！char 数组是连续的内存
    int age;
};

int main() {
    // 错误示范会导致：
    // 1. 写入的是指针地址，不是字符串内容
    // 2. 读取时指针已经失效，会导致程序崩溃
    // 3. 不同机器上指针地址不同，无法跨机器使用

    GoodStudent stu;
    strcpy(stu.name, "张三");
    stu.age = 20;

    // 正确的二进制写入
    ofstream ofs("good.dat", ios::out | ios::binary);
    ofs.write((const char *)&stu, sizeof(GoodStudent));
    ofs.close();

    return 0;
}
```

### 5.2 结构体的内存对齐

```cpp
#include <iostream>
#include <fstream>
using namespace std;

// 结构体可能有内存对齐，导致实际大小大于成员大小之和
struct Example {
    char a;     // 1字节
    // 可能有3字节的填充（对齐到4字节边界）
    int b;      // 4字节
    char c;     // 1字节
    // 可能有3字节的填充（对齐到4字节边界）
};

int main() {
    cout << "char 大小：" << sizeof(char) << endl;   // 1
    cout << "int 大小：" << sizeof(int) << endl;     // 4
    cout << "Example 大小：" << sizeof(Example) << endl;  // 可能是12，不是6

    // 建议：在读写二进制文件前，先用 sizeof 确认结构体大小
    // 写入和读取时使用相同的结构体定义

    return 0;
}
```

### 5.3 读写要匹配

```cpp
#include <iostream>
#include <fstream>
using namespace std;

struct Data {
    int a;
    double b;
};

int main() {
    // 写入一个 Data
    Data d1 = {100, 3.14};
    ofstream ofs("match.dat", ios::out | ios::binary);
    ofs.write((const char *)&d1, sizeof(Data));
    ofs.close();

    // 读取时必须用相同的类型和大小
    Data d2;
    ifstream ifs("match.dat", ios::in | ios::binary);
    ifs.read((char *)&d2, sizeof(Data));  // 正确

    // 错误示范：
    // int temp;
    // ifs.read((char *)&temp, sizeof(int));  // 只读了部分数据

    cout << "a = " << d2.a << endl;
    cout << "b = " << d2.b << endl;

    ifs.close();
    return 0;
}
```

## 6. 完整的实际应用示例

### 6.1 学生信息管理系统（二进制版）

```cpp
#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

struct Student {
    int id;
    char name[50];
    int age;
    double score;
};

// 写入学生信息
void writeStudents(const char *filename) {
    ofstream ofs(filename, ios::out | ios::binary);

    if (!ofs.is_open()) {
        cout << "文件打开失败！" << endl;
        return;
    }

    Student stu;

    // 输入学生1
    stu.id = 1;
    strcpy(stu.name, "张三");
    stu.age = 20;
    stu.score = 95.5;
    ofs.write((const char *)&stu, sizeof(Student));

    // 输入学生2
    stu.id = 2;
    strcpy(stu.name, "李四");
    stu.age = 21;
    stu.score = 88.0;
    ofs.write((const char *)&stu, sizeof(Student));

    // 输入学生3
    stu.id = 3;
    strcpy(stu.name, "王五");
    stu.age = 22;
    stu.score = 92.5;
    ofs.write((const char *)&stu, sizeof(Student));

    ofs.close();
    cout << "学生信息保存成功！" << endl;
}

// 读取学生信息
void readStudents(const char *filename) {
    ifstream ifs(filename, ios::in | ios::binary);

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return;
    }

    Student stu;

    cout << "================================" << endl;
    cout << "学号\t姓名\t年龄\t分数" << endl;
    cout << "================================" << endl;

    // 循环读取所有学生
    while (ifs.read((char *)&stu, sizeof(Student))) {
        cout << stu.id << "\t"
             << stu.name << "\t"
             << stu.age << "\t"
             << stu.score << endl;
    }

    cout << "================================" << endl;

    ifs.close();
}

int main() {
    const char *filename = "students_binary.dat";

    // 写入学生信息
    writeStudents(filename);

    // 读取并显示
    readStudents(filename);

    return 0;
}
```

### 6.2 追加写入二进制文件

```cpp
#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

struct Record {
    char time[20];
    char message[100];
};

int main() {
    // 追加模式写入
    ofstream ofs("log.dat", ios::out | ios::app | ios::binary);

    if (!ofs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    Record rec;

    strcpy(rec.time, "2024-01-01 10:00");
    strcpy(rec.message, "程序启动");
    ofs.write((const char *)&rec, sizeof(Record));

    strcpy(rec.time, "2024-01-01 10:05");
    strcpy(rec.message, "用户登录成功");
    ofs.write((const char *)&rec, sizeof(Record));

    ofs.close();
    cout << "日志写入完成！" << endl;

    // 读取日志
    ifstream ifs("log.dat", ios::in | ios::binary);

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    cout << "\n日志内容：" << endl;
    while (ifs.read((char *)&rec, sizeof(Record))) {
        cout << "[" << rec.time << "] " << rec.message << endl;
    }

    ifs.close();
    return 0;
}
```

## 7. 二进制文件的优势

```cpp
// 优势1：速度快
// 文本方式需要将数字转换为字符串，再写入
// 二进制方式直接写入内存中的数据，无需转换

// 优势2：保持精度
// 文本方式写入 3.14159265358979，可能丢失精度
// 二进制方式完全保持 double 的原始精度

// 优势3：节省空间
// 文本方式存储数字 1000000 需要7个字节
// 二进制方式存储数字 1000000 只需要4个字节（int类型）
```

## 8. 二进制文件的注意事项

### 8.1 跨平台兼容性

```cpp
// 不同平台可能有不同的：
// 1. 字节序（大端/小端）
//    - x86/x64 是小端序
//    - 某些ARM、网络协议是大端序
//
// 2. 数据类型大小
//    - long 在32位系统是4字节，在64位Linux是8字节
//    - 指针大小因平台而异
//
// 3. 结构体对齐方式
//    - 不同编译器可能有不同的对齐规则

// 解决方案：
// 1. 使用固定大小的类型（int32_t, uint64_t 等）
// 2. 避免使用指针和 string
// 3. 在同一平台上读写
// 4. 需要跨平台时，使用序列化库
```

### 8.2 版本兼容性

```cpp
#include <iostream>
#include <fstream>
using namespace std;

// 版本1：结构体定义
struct StudentV1 {
    char name[50];
    int age;
};

// 版本2：增加了字段
struct StudentV2 {
    char name[50];
    int age;
    double score;  // 新增字段
};

// 问题：用V1写入的文件，用V2读取会出错
// 解决方案：
// 1. 在文件头记录版本号
// 2. 根据版本号选择不同的读取方式
// 3. 设计时预留扩展空间

struct FileHeader {
    int version;      // 版本号
    int recordCount;  // 记录数量
};
```

## 9. 总结

| 要点 | 说明 |
|------|------|
| 写入函数 | `ofs.write((const char *)&obj, sizeof(obj))` |
| 读取函数 | `ifs.read((char *)&obj, sizeof(obj))` |
| 打开模式 | `ios::out \| ios::binary` 或 `ios::in \| ios::binary` |
| 重要规则 | 结构体中不要用 string，要用 char 数组 |
| 适用场景 | 需要高效存储和读取结构化数据 |

二进制文件读写的优点：
1. 读写速度快
2. 保持数据精度
3. 适合存储结构体/类对象

注意事项：
1. 不要使用 string 类型
2. 写入和读取的数据类型要匹配
3. 注意跨平台兼容性
