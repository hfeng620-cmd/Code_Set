# 异常处理详解

## 1. 概述

程序在运行时可能会出现各种错误，比如：
- 除以0
- 数组越界
- 内存不足
- 文件不存在
- 网络连接失败

**异常处理**是一种处理运行时错误的机制，让程序在出错时能够优雅地处理，而不是直接崩溃。

---

## 2. 为什么需要异常处理

### 2.1 没有异常处理的问题

```cpp
#include <iostream>
using namespace std;

// 没有异常处理，程序可能直接崩溃
int divide(int a, int b) {
    return a / b;  // 如果 b 是 0，程序会崩溃！
}

int main() {
    int result = divide(10, 0);  // 程序崩溃！
    cout << result << endl;
    return 0;
}
```

### 2.2 使用错误码的方式（不推荐）

```cpp
#include <iostream>
using namespace std;

// 使用返回值表示错误，但调用者可能忘记检查
int divide(int a, int b, bool &success) {
    if (b == 0) {
        success = false;
        return 0;
    }
    success = true;
    return a / b;
}

int main() {
    bool success;
    int result = divide(10, 0, success);

    // 调用者必须记得检查 success
    if (success) {
        cout << "结果：" << result << endl;
    } else {
        cout << "除数不能为0！" << endl;
    }

    return 0;
}
```

### 2.3 使用异常处理（推荐）

```cpp
#include <iostream>
using namespace std;

int divide(int a, int b) {
    if (b == 0) {
        throw "除数不能为0！";  // 抛出异常
    }
    return a / b;
}

int main() {
    try {
        int result = divide(10, 0);  // 这里会抛出异常
        cout << "结果：" << result << endl;
    } catch (const char *msg) {
        cout << "捕获到异常：" << msg << endl;
    }

    return 0;
}
```

## 3. try-catch-throw 基本语法

```cpp
#include <iostream>
using namespace std;

int main() {
    try {
        // try 块：包裹可能抛出异常的代码
        int num = 10;
        int den = 0;

        if (den == 0) {
            throw "除数不能为零";  // throw 抛出异常
        }

        cout << num / den << endl;
    }
    catch (const char *msg) {
        // catch 块：捕获并处理异常
        cout << "错误：" << msg << endl;
    }

    cout << "程序继续运行..." << endl;

    return 0;
}
```

## 5. catch 捕获异常的多种写法

### 5.1 捕获整型异常

```cpp
#include <iostream>
using namespace std;

int main() {
    try {
        throw 100;  // 抛出整数
    } catch (int e) {
        cout << "捕获到整型异常：" << e << endl;
    }

    return 0;
}
```

### 5.2 捕获字符串异常

```cpp
#include <iostream>
using namespace std;

int main() {
    try {
        throw "这是一个错误";  // 抛出 const char*
    } catch (const char *msg) {
        cout << "捕获到字符串异常：" << msg << endl;
    }

    return 0;
}
```

### 5.3 捕获 string 对象异常

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    try {
        throw string("这是一个错误");  // 抛出 string 对象
    } catch (string &msg) {
        cout << "捕获到string异常：" << msg << endl;
    }

    return 0;
}
```

### 5.4 捕获标准异常（推荐）

```cpp
#include <iostream>
#include <stdexcept>  // 标准异常类
using namespace std;

int main() {
    try {
        throw runtime_error("运行时错误");
    } catch (runtime_error &e) {
        cout << "捕获到运行时错误：" << e.what() << endl;
    }

    return 0;
}
```

### 5.5 捕获所有异常

```cpp
#include <iostream>
using namespace std;

int main() {
    try {
        throw 3.14;  // 抛出浮点数
    } catch (...) {
        // catch(...) 可以捕获任何类型的异常
        // 但是无法获取异常的具体信息
        cout << "捕获到未知类型的异常" << endl;
    }

    return 0;
}
```

## 6. 多级 catch 的匹配规则

```cpp
#include <iostream>
#include <stdexcept>
using namespace std;

int main() {
    try {
        throw runtime_error("测试错误");
    }
    catch (int e) {
        cout << "捕获 int：" << e << endl;
    }
    catch (const char *msg) {
        cout << "捕获 const char*：" << msg << endl;
    }
    catch (runtime_error &e) {
        // 匹配到了这个 catch
        cout << "捕获 runtime_error：" << e.what() << endl;
    }
    catch (exception &e) {
        // 不会执行，因为上面已经匹配到了
        cout << "捕获 exception：" << e.what() << endl;
    }
    catch (...) {
        // 不会执行，因为上面已经匹配到了
        cout << "捕获未知异常" << endl;
    }

    return 0;
}
```

```cpp
// 匹配规则：
// 1. 从上到下依次匹配
// 2. 匹配到第一个就停止，不会继续往下匹配
// 3. 子类异常要放在父类异常前面
// 4. catch(...) 要放在最后，捕获所有未匹配的异常
```

## 7. 异常的重新抛出

```cpp
#include <iostream>
#include <stdexcept>
using namespace std;

void func1() {
    try {
        throw runtime_error("func1 中的错误");
    } catch (runtime_error &e) {
        cout << "func1 捕获到异常：" << e.what() << endl;
        throw;  // 重新抛出异常，让上层处理
    }
}

void func2() {
    try {
        func1();
    } catch (runtime_error &e) {
        cout << "func2 捕获到异常：" << e.what() << endl;
        // 这里可以处理异常，也可以继续抛出
    }
}

int main() {
    func2();
    return 0;
}
```

输出：
```
func1 捕获到异常：func1 中的错误
func2 捕获到异常：func1 中的错误
```

## 8. 自定义异常类

### 8.1 继承 exception 类（推荐）

```cpp
#include <iostream>
#include <exception>
#include <string>
using namespace std;

// 自定义异常类，继承 exception
class MyException : public exception {
private:
    string message;
    int errorCode;

public:
    MyException(const string &msg, int code) : message(msg), errorCode(code) {}

    // 重写 what() 方法
    const char *what() const noexcept override {
        return message.c_str();
    }

    int getErrorCode() const {
        return errorCode;
    }
};

void doSomething(int value) {
    if (value < 0) {
        throw MyException("值不能为负数", 1001);
    }
    if (value > 100) {
        throw MyException("值不能超过100", 1002);
    }
    cout << "值为：" << value << endl;
}

int main() {
    try {
        doSomething(-1);
    } catch (MyException &e) {
        cout << "错误信息：" << e.what() << endl;
        cout << "错误代码：" << e.getErrorCode() << endl;
    }

    return 0;
}
```

### 8.2 不继承 exception 的自定义异常

```cpp
#include <iostream>
#include <string>
using namespace std;

// 简单的自定义异常类
class FileError {
private:
    string filename;
    string message;

public:
    FileError(const string &file, const string &msg)
        : filename(file), message(msg) {}

    void showError() const {
        cout << "文件错误：" << filename << " - " << message << endl;
    }
};

void openFile(const string &filename) {
    // 模拟文件打开失败
    if (filename == "不存在.txt") {
        throw FileError(filename, "文件不存在");
    }
    cout << "文件打开成功：" << filename << endl;
}

int main() {
    try {
        openFile("不存在.txt");
    } catch (FileError &e) {
        e.showError();
    }

    return 0;
}
```

## 9. 标准异常类

### 9.1 标准异常类层次结构

```
exception
├── logic_error
│   ├── invalid_argument    // 无效参数
│   ├── domain_error        // 域错误
│   ├── length_error        // 长度错误
│   └── out_of_range        // 越界错误
├── runtime_error
│   ├── range_error         // 范围错误
│   ├── overflow_error      // 上溢错误
│   └── underflow_error     // 下溢错误
└── bad_alloc               // 内存分配失败
```

### 9.2 常用标准异常示例

```cpp
#include <iostream>
#include <stdexcept>
#include <vector>
using namespace std;

int main() {
    // runtime_error 示例
    try {
        throw runtime_error("运行时发生了错误");
    } catch (runtime_error &e) {
        cout << "runtime_error: " << e.what() << endl;
    }

    // logic_error 示例
    try {
        throw logic_error("逻辑错误");
    } catch (logic_error &e) {
        cout << "logic_error: " << e.what() << endl;
    }

    // invalid_argument 示例
    try {
        int age = -1;
        if (age < 0) {
            throw invalid_argument("年龄不能为负数");
        }
    } catch (invalid_argument &e) {
        cout << "invalid_argument: " << e.what() << endl;
    }

    // out_of_range 示例
    try {
        vector<int> v = {1, 2, 3};
        cout << v.at(10) << endl;  // 会抛出 out_of_range
    } catch (out_of_range &e) {
        cout << "out_of_range: " << e.what() << endl;
    }

    // 捕获所有标准异常（用 exception 基类）
    try {
        throw runtime_error("测试");
    } catch (exception &e) {
        cout << "exception: " << e.what() << endl;
    }

    return 0;
}
```

## 10. noexcept 关键字

### 10.1 noexcept 的含义

```cpp
#include <iostream>
using namespace std;

// noexcept 表示这个函数不会抛出异常
// 如果函数中真的抛出了异常，程序会直接调用 terminate() 终止
void safeFunction() noexcept {
    cout << "这个函数保证不抛出异常" << endl;
    // throw "错误";  // 如果取消注释，程序会直接崩溃
}

// 没有 noexcept 的函数可能抛出异常
void riskyFunction() {
    throw "可能发生的错误";
}

int main() {
    safeFunction();

    try {
        riskyFunction();
    } catch (const char *msg) {
        cout << "捕获：" << msg << endl;
    }

    return 0;
}
```

### 10.2 noexcept 的使用场景

```cpp
#include <iostream>
using namespace std;

class MyClass {
public:
    // 析构函数默认就是 noexcept 的
    ~MyClass() {
        cout << "析构函数" << endl;
    }

    // 移动构造函数建议标记为 noexcept
    MyClass(MyClass &&other) noexcept {
        cout << "移动构造函数" << endl;
    }

    // 某些函数可以标记为 noexcept
    int getValue() const noexcept {
        return 42;
    }
};

int main() {
    MyClass obj;
    cout << obj.getValue() << endl;

    return 0;
}
```

## 11. 异常处理的注意事项

### 11.1 不要滥用异常

```cpp
#include <iostream>
using namespace std;

// 不好的做法：用异常控制正常流程
int findIndex(int arr[], int size, int target) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == target) {
            return i;  // 找到了返回索引
        }
    }
    throw "未找到";  // 不应该用异常来表示"没找到"这种情况
}

// 好的做法：用返回值或特殊标记
int findIndexBetter(int arr[], int size, int target) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == target) {
            return i;
        }
    }
    return -1;  // 用 -1 表示未找到
}

int main() {
    int arr[] = {1, 2, 3, 4, 5};

    // 好的做法
    int index = findIndexBetter(arr, 5, 3);
    if (index != -1) {
        cout << "找到，索引为：" << index << endl;
    } else {
        cout << "未找到" << endl;
    }

    return 0;
}
```

```cpp
// 异常应该用于：
// 1. 不可恢复的错误（如内存不足）
// 2. 不应该发生的错误（如数组越界）
// 3. 外部错误（如文件不存在、网络失败）

// 异常不应该用于：
// 1. 正常的控制流程
// 2. 可以预见的情况
// 3. 频繁发生的情况（异常有性能开销）
```

### 11.2 析构函数中不要抛出异常

```cpp
#include <iostream>
using namespace std;

// 错误示范：析构函数中抛出异常
class BadClass {
public:
    ~BadClass() {
        // throw "析构函数中的错误";  // 危险！
        // 如果在处理其他异常时析构，会导致程序终止
    }
};

// 正确做法：析构函数中捕获所有异常
class GoodClass {
public:
    ~GoodClass() {
        try {
            // 可能抛出异常的代码
        } catch (...) {
            // 记录日志，但不继续抛出
            cout << "析构函数中发生错误，已忽略" << endl;
        }
    }
};

int main() {
    GoodClass obj;
    return 0;
}
```

### 11.3 资源泄漏问题

```cpp
#include <iostream>
using namespace std;

// 问题：异常可能导致资源泄漏
void problematic() {
    int *ptr = new int(100);

    // 如果这里抛出异常，delete 不会执行，内存泄漏！
    throw "错误";

    delete ptr;  // 永远不会执行
}

// 解决方案1：使用 try-catch
void solution1() {
    int *ptr = new int(100);

    try {
        throw "错误";
    } catch (...) {
        delete ptr;  // 在 catch 中释放资源
        throw;       // 重新抛出异常
    }

    delete ptr;
}

// 解决方案2：使用智能指针（推荐）
#include <memory>
void solution2() {
    unique_ptr<int> ptr(new int(100));

    throw "错误";

    // 不需要手动 delete，智能指针会自动释放
}

int main() {
    try {
        solution2();
    } catch (const char *msg) {
        cout << "错误：" << msg << endl;
    }

    return 0;
}
```

## 12. 完整的实际应用示例

### 12.1 输入验证

```cpp
#include <iostream>
#include <stdexcept>
#include <string>
using namespace std;

class InputValidator {
public:
    static int getAge() {
        int age;
        cout << "请输入年龄：";
        cin >> age;

        if (cin.fail()) {
            cin.clear();  // 清除错误标志
            cin.ignore(1000, '\n');  // 清空输入缓冲区
            throw invalid_argument("输入的不是数字");
        }

        if (age < 0 || age > 150) {
            throw out_of_range("年龄必须在0-150之间");
        }

        return age;
    }

    static string getNonEmptyString(const string &prompt) {
        string input;
        cout << prompt;
        getline(cin, input);

        if (input.empty()) {
            throw invalid_argument("输入不能为空");
        }

        return input;
    }
};

int main() {
    try {
        string name = InputValidator::getNonEmptyString("请输入姓名：");
        int age = InputValidator::getAge();

        cout << "姓名：" << name << endl;
        cout << "年龄：" << age << endl;
    } catch (invalid_argument &e) {
        cout << "输入错误：" << e.what() << endl;
    } catch (out_of_range &e) {
        cout << "范围错误：" << e.what() << endl;
    } catch (exception &e) {
        cout << "其他错误：" << e.what() << endl;
    }

    return 0;
}
```

### 12.2 文件操作异常处理

```cpp
#include <iostream>
#include <fstream>
#include <stdexcept>
#include <string>
using namespace std;

class FileManager {
public:
    static string readFile(const string &filename) {
        ifstream ifs(filename);

        if (!ifs.is_open()) {
            throw runtime_error("无法打开文件：" + filename);
        }

        string content;
        string line;
        while (getline(ifs, line)) {
            content += line + "\n";
        }

        if (ifs.bad()) {
            throw runtime_error("读取文件时发生错误");
        }

        ifs.close();
        return content;
    }

    static void writeFile(const string &filename, const string &content) {
        ofstream ofs(filename);

        if (!ofs.is_open()) {
            throw runtime_error("无法创建文件：" + filename);
        }

        ofs << content;

        if (ofs.bad()) {
            throw runtime_error("写入文件时发生错误");
        }

        ofs.close();
    }
};

int main() {
    try {
        // 写入文件
        FileManager::writeFile("test.txt", "Hello, World!");
        cout << "文件写入成功" << endl;

        // 读取文件
        string content = FileManager::readFile("test.txt");
        cout << "文件内容：" << content;

        // 尝试读取不存在的文件
        string content2 = FileManager::readFile("不存在.txt");
    } catch (runtime_error &e) {
        cout << "运行时错误：" << e.what() << endl;
    } catch (exception &e) {
        cout << "错误：" << e.what() << endl;
    }

    return 0;
}
```

### 12.3 数学计算异常

```cpp
#include <iostream>
#include <stdexcept>
#include <cmath>
using namespace std;

class MathUtils {
public:
    static double divide(double a, double b) {
        if (b == 0) {
            throw invalid_argument("除数不能为零");
        }
        return a / b;
    }

    static double squareRoot(double x) {
        if (x < 0) {
            throw invalid_argument("不能对负数求平方根");
        }
        return sqrt(x);
    }

    static double logarithm(double x) {
        if (x <= 0) {
            throw invalid_argument("对数的参数必须大于0");
        }
        return log(x);
    }

    static int factorial(int n) {
        if (n < 0) {
            throw invalid_argument("阶乘的参数不能为负数");
        }
        if (n > 20) {
            throw overflow_error("结果会溢出");
        }

        int result = 1;
        for (int i = 2; i <= n; i++) {
            result *= i;
        }
        return result;
    }
};

int main() {
    try {
        cout << "10 / 3 = " << MathUtils::divide(10, 3) << endl;
        cout << "sqrt(16) = " << MathUtils::squareRoot(16) << endl;
        cout << "ln(e) = " << MathUtils::logarithm(2.71828) << endl;
        cout << "5! = " << MathUtils::factorial(5) << endl;

        // 这会抛出异常
        cout << "10 / 0 = " << MathUtils::divide(10, 0) << endl;
    } catch (invalid_argument &e) {
        cout << "参数错误：" << e.what() << endl;
    } catch (overflow_error &e) {
        cout << "溢出错误：" << e.what() << endl;
    }

    return 0;
}
```

## 13. 总结

| 要点 | 说明 |
|------|------|
| throw | 抛出异常对象 |
| try | 包裹可能抛出异常的代码 |
| catch | 捕获并处理异常 |
| catch(...) | 捕获所有类型的异常 |
| exception | 标准异常基类 |
| what() | 获取异常的错误信息 |
| noexcept | 承诺函数不抛出异常 |

异常处理的基本原则：
1. 只在真正异常的情况下使用异常
2. 子类异常要放在父类异常前面捕获
3. 析构函数中不要抛出异常
4. 使用智能指针避免资源泄漏
5. 不要滥用异常处理正常流程
