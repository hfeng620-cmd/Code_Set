# 32 - 函数基础与定义

## 一、为什么需要函数？

### 1.1 代码复用
函数可以将重复的代码封装起来，避免写同样的代码多次。

```cpp
// 没有函数：每次都要写重复的代码
#include <iostream>
using namespace std;

int main() {
    // 第一次计算圆的面积
    double r1 = 5.0;
    double area1 = 3.14 * r1 * r1;
    cout << "第一个圆的面积: " << area1 << endl;

    // 第二次计算圆的面积（重复代码！）
    double r2 = 10.0;
    double area2 = 3.14 * r2 * r2;
    cout << "第二个圆的面积: " << area2 << endl;

    // 第三次计算圆的面积（又是重复代码！）
    double r3 = 3.0;
    double area3 = 3.14 * r3 * r3;
    cout << "第三个圆的面积: " << area3 << endl;

    return 0;
}
```

```cpp
// 使用函数：代码复用，简洁清晰
#include <iostream>
using namespace std;

// 定义一个计算圆面积的函数
double circleArea(double radius) {
    return 3.14 * radius * radius;
}

int main() {
    // 调用函数，代码简洁明了
    cout << "第一个圆的面积: " << circleArea(5.0) << endl;
    cout << "第二个圆的面积: " << circleArea(10.0) << endl;
    cout << "第三个圆的面积: " << circleArea(3.0) << endl;

    return 0;
}
```

### 1.2 逻辑清晰
函数让程序结构更清晰，每个函数负责一个特定的功能。

```cpp
#include <iostream>
using namespace std;

// 每个函数负责一个明确的功能
void printWelcome() {
    cout << "========================" << endl;
    cout << "  欢迎使用本程序！" << endl;
    cout << "========================" << endl;
}

void printMenu() {
    cout << "1. 添加学生" << endl;
    cout << "2. 查找学生" << endl;
    cout << "3. 退出程序" << endl;
}

void printGoodbye() {
    cout << "感谢使用，再见！" << endl;
}

int main() {
    printWelcome();   // 打印欢迎信息
    printMenu();      // 打印菜单
    printGoodbye();   // 打印再见信息
    return 0;
}
```

### 1.3 方便维护
如果需要修改功能，只需要修改函数定义，所有调用处都会自动更新。

```cpp
#include <iostream>
using namespace std;

// 如果需要修改圆周率精度，只需要改这一处
double circleArea(double radius) {
    return 3.14159265358979 * radius * radius;  // 修改了精度
}

int main() {
    // 所有调用处自动使用新的精度
    cout << circleArea(5.0) << endl;
    cout << circleArea(10.0) << endl;
    return 0;
}
```

## 二、函数的定义语法

### 2.1 基本语法

```
返回值类型 函数名(参数列表) {
    函数体
    return 返回值;
}
```

各部分说明：
- **返回值类型**：函数执行完毕后返回的数据类型（int、double、void等）
- **函数名**：给函数起的名字，遵循标识符命名规则
- **参数列表**：函数接收的输入，可以有多个参数，用逗号分隔
- **函数体**：函数要执行的代码
- **return**：返回结果，void函数可以省略

### 2.2 语法示例

```cpp
#include <iostream>
using namespace std;

// 示例1：计算两个数的和
// 返回值类型：int
// 函数名：add
// 参数列表：int a, int b
// 函数体：计算并返回结果
int add(int a, int b) {
    int sum = a + b;
    return sum;  // 返回计算结果
}

// 示例2：打印分隔线
// 返回值类型：void（不需要返回值）
// 函数名：printLine
// 参数列表：无
void printLine() {
    cout << "--------------------" << endl;
    // void函数可以省略return，函数执行完自动返回
}

// 示例3：判断是否为正数
// 返回值类型：bool
// 函数名：isPositive
// 参数列表：double num
bool isPositive(double num) {
    if (num > 0) {
        return true;   // 返回true表示是正数
    } else {
        return false;  // 返回false表示不是正数
    }
}

int main() {
    int result = add(3, 5);    // 调用函数，result = 8
    cout << "3 + 5 = " << result << endl;

    printLine();               // 调用打印分隔线函数

    bool check = isPositive(-3.14);  // 调用判断函数
    cout << "-3.14是正数吗？" << (check ? "是" : "否") << endl;

    return 0;
}
```

## 三、函数的四种形式

### 3.1 无参无返（void 函数名()）

没有参数，没有返回值，通常用于执行某个固定的操作。

```cpp
#include <iostream>
using namespace std;

// 无参无返：打印欢迎信息
void printWelcome() {
    cout << "========================" << endl;
    cout << "  欢迎来到C++世界！" << endl;
    cout << "========================" << endl;
}

// 无参无返：打印分割线
void printSeparator() {
    cout << "----------------------------------------" << endl;
}

// 无参无返：执行某个固定操作
void doSomething() {
    int x = 10;  // 局部变量，只在函数内有效
    x += 5;
    cout << "函数内部 x = " << x << endl;
    // 这个函数没有返回值，不能用变量接收
}

int main() {
    printWelcome();      // 直接调用，不需要接收返回值
    printSeparator();
    doSomething();

    // 错误写法！void函数没有返回值
    // int result = printWelcome();  // 编译错误！

    return 0;
}
```

### 3.2 有参无返（void 函数名(参数)）

有参数但没有返回值，通常用于根据参数执行操作。

```cpp
#include <iostream>
using namespace std;

// 有参无返：打印n个星号
void printStars(int n) {
    for (int i = 0; i < n; i++) {
        cout << "*";
    }
    cout << endl;
}

// 有参无返：打印两个数的和（直接输出，不返回）
void printSum(int a, int b) {
    int sum = a + b;
    cout << a << " + " << b << " = " << sum << endl;
}

// 有参无返：打印指定次数的消息
void printMessage(string msg, int times) {
    for (int i = 0; i < times; i++) {
        cout << msg << endl;
    }
}

int main() {
    printStars(5);           // 打印5个星号：*****
    printStars(10);          // 打印10个星号：**********

    printSum(3, 5);          // 输出：3 + 5 = 8
    printSum(100, 200);      // 输出：100 + 200 = 300

    printMessage("Hello", 3);  // 打印3次Hello

    return 0;
}
```

### 3.3 无参有返（返回值类型 函数名()）

没有参数但有返回值，通常用于获取某个值。

```cpp
#include <iostream>
using namespace std;

// 无参有返：获取用户输入的整数
int getNumber() {
    int num;
    cout << "请输入一个整数: ";
    cin >> num;
    return num;  // 返回用户输入的值
}

// 无参有返：获取当前时间（模拟）
int getCurrentHour() {
    // 这里模拟返回一个固定值
    return 14;  // 假设当前是14点
}

// 无参有返：计算1到100的和
int getSum1to100() {
    int sum = 0;
    for (int i = 1; i <= 100; i++) {
        sum += i;
    }
    return sum;  // 返回计算结果
}

int main() {
    // 方式1：用变量接收返回值
    int num = getNumber();
    cout << "你输入的是: " << num << endl;

    // 方式2：直接使用返回值
    cout << "当前时间: " << getCurrentHour() << "点" << endl;

    // 方式3：将返回值用于表达式
    int result = getSum1to100() + 1000;
    cout << "1到100的和 + 1000 = " << result << endl;

    return 0;
}
```

### 3.4 有参有返（返回值类型 函数名(参数)）

既有参数又有返回值，这是最常用的函数形式。

```cpp
#include <iostream>
using namespace std;

// 有参有返：计算两个数的最大值
int getMax(int a, int b) {
    if (a > b) {
        return a;
    } else {
        return b;
    }
    // 也可以简写为：return (a > b) ? a : b;
}

// 有参有返：计算圆的面积
double circleArea(double radius) {
    return 3.14159 * radius * radius;
}

// 有参有返：计算阶乘
long long factorial(int n) {
    long long result = 1;
    for (int i = 1; i <= n; i++) {
        result *= i;
    }
    return result;
}

// 有参有返：判断是否为偶数
bool isEven(int num) {
    return num % 2 == 0;
}

// 有参有返：计算字符串长度（模拟）
int getStringLength(string str) {
    return str.length();
}

int main() {
    // 调用有参有返的函数
    int maxVal = getMax(10, 20);
    cout << "10和20的最大值是: " << maxVal << endl;

    double area = circleArea(5.0);
    cout << "半径为5的圆面积: " << area << endl;

    long long fact = factorial(10);
    cout << "10的阶乘: " << fact << endl;

    cout << "7是偶数吗？" << (isEven(7) ? "是" : "否") << endl;
    cout << "8是偶数吗？" << (isEven(8) ? "是" : "否") << endl;

    cout << "字符串\"Hello\"的长度: " << getStringLength("Hello") << endl;

    return 0;
}
```

## 四、函数的调用方式

### 4.1 直接调用

直接调用函数执行，不使用返回值（即使函数有返回值）。

```cpp
#include <iostream>
using namespace std;

int add(int a, int b) {
    return a + b;
}

void printHello() {
    cout << "Hello!" << endl;
}

int main() {
    // 直接调用有返回值的函数（不使用返回值）
    add(3, 5);  // 计算了但没用结果，通常不推荐

    // 直接调用无返回值的函数
    printHello();  // 正常用法

    // 将函数调用放在输出语句中
    cout << add(10, 20) << endl;  // 直接输出返回值

    return 0;
}
```

### 4.2 赋值调用

将函数的返回值赋给一个变量。

```cpp
#include <iostream>
using namespace std;

double calculateArea(double radius) {
    return 3.14 * radius * radius;
}

int main() {
    // 赋值调用：用变量接收返回值
    double area1 = calculateArea(5.0);
    double area2 = calculateArea(10.0);

    cout << "面积1: " << area1 << endl;
    cout << "面积2: " << area2 << endl;

    // 赋值调用：将返回值用于表达式
    double totalArea = calculateArea(3.0) + calculateArea(4.0);
    cout << "总面积: " << totalArea << endl;

    // 赋值调用：将返回值作为另一个函数的参数
    cout << "面积差: " << calculateArea(10.0) - calculateArea(5.0) << endl;

    return 0;
}
```

## 五、函数的返回值

### 5.1 return 的作用

```cpp
#include <iostream>
using namespace std;

// return 的两个作用：
// 1. 返回一个值给调用者
// 2. 结束函数的执行

int divide(int a, int b) {
    if (b == 0) {
        cout << "错误：除数不能为0！" << endl;
        return -1;  // 提前返回，后面的代码不会执行
    }
    return a / b;   // 正常返回计算结果
}

int main() {
    int result = divide(10, 0);  // 会打印错误信息，返回-1
    cout << "结果: " << result << endl;

    result = divide(10, 2);      // 正常计算，返回5
    cout << "结果: " << result << endl;

    return 0;
}
```

### 5.2 void函数可以省略return

```cpp
#include <iostream>
using namespace std;

// void函数可以省略return，函数执行完自动返回
void printNumbers(int n) {
    for (int i = 1; i <= n; i++) {
        cout << i << " ";
    }
    cout << endl;
    // 不需要写return，函数自然结束
}

// void函数也可以用return提前结束
void processNumber(int num) {
    if (num < 0) {
        cout << "负数，不处理" << endl;
        return;  // 提前结束函数
    }
    cout << "处理数字: " << num * 2 << endl;
}

int main() {
    printNumbers(5);      // 输出：1 2 3 4 5
    processNumber(-5);    // 输出：负数，不处理
    processNumber(10);    // 输出：处理数字: 20

    return 0;
}
```

### 5.3 return后代码不执行

```cpp
#include <iostream>
using namespace std;

int findFirstEven(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        if (arr[i] % 2 == 0) {
            return arr[i];  // 找到第一个偶数就返回
            // 下面的cout永远不会执行
            cout << "这行代码不会执行！" << endl;
        }
    }
    return -1;  // 没找到偶数
}

int main() {
    int arr[] = {1, 3, 4, 6, 8};
    int result = findFirstEven(arr, 5);
    cout << "第一个偶数是: " << result << endl;  // 输出4

    return 0;
}
```

## 六、参数传递的概念

### 6.1 形参和实参的区别

```cpp
#include <iostream>
using namespace std;

// 形式参数（形参）：定义函数时的参数，相当于占位符
// 形参在函数调用时才会分配内存
void add(int a, int b) {  // a和b是形参
    cout << "函数内 a = " << a << ", b = " << b << endl;
    int sum = a + b;
    cout << "和 = " << sum << endl;
}

int main() {
    int x = 10;
    int y = 20;

    // 实际参数（实参）：调用函数时传入的实际值
    // 实参可以是常量、变量、表达式
    add(x, y);        // x和y是实参
    add(100, 200);    // 100和200是实参
    add(x + 5, y * 2);  // x+5和y*2是实参（表达式）

    return 0;
}
```

### 6.2 形参和实参的关系

```cpp
#include <iostream>
using namespace std;

// 值传递：形参是实参的拷贝
// 修改形参不会影响实参
void modifyValue(int num) {
    num = 100;  // 修改的是形参，不影响实参
    cout << "函数内 num = " << num << endl;
}

int main() {
    int x = 10;
    cout << "调用前 x = " << x << endl;

    modifyValue(x);  // 传递x的值给形参num
    cout << "调用后 x = " << x << endl;  // x仍然是10

    return 0;
}
```

## 七、常见错误

### 7.1 忘记return

```cpp
#include <iostream>
using namespace std;

// 错误示例：有返回值的函数忘记return
int add(int a, int b) {
    int sum = a + b;
    // 忘记写 return sum; 这是严重错误！
    // 编译器可能会警告，但某些编译器可能不会报错
    // 结果是未定义的
}

// 正确写法
int addCorrect(int a, int b) {
    int sum = a + b;
    return sum;  // 必须返回值
}

// 还有一种错误：不是所有路径都有return
int getValue(bool flag) {
    if (flag) {
        return 10;
    }
    // 如果flag为false，没有return，这是错误！
    // 应该加上 else { return 0; }
}

int main() {
    // 使用忘记return的函数，结果是未定义的
    int result = add(3, 5);
    cout << "结果: " << result << endl;  // 输出什么？不确定！

    return 0;
}
```

### 7.2 返回值类型不匹配

```cpp
#include <iostream>
using namespace std;

// 函数声明返回int，但返回了double
int getDouble() {
    return 3.14;  // double会被截断为int，小数部分丢失
}

// 函数声明返回double，但返回了int
double getInt() {
    return 42;  // int会自动转换为double，没问题
}

// 函数声明返回bool，但返回了int
bool check(int x) {
    return x;  // 非0为true，0为false
}

int main() {
    cout << getDouble() << endl;  // 输出3，小数部分丢失
    cout << getInt() << endl;     // 输出42，自动转换
    cout << check(0) << endl;     // 输出0（false）
    cout << check(5) << endl;     // 输出1（true）

    return 0;
}
```

### 7.3 函数嵌套定义（不允许）

```cpp
#include <iostream>
using namespace std;

// C++不允许函数嵌套定义！
void outerFunction() {
    cout << "这是外部函数" << endl;

    // 错误！不能在函数内部定义另一个函数
    // void innerFunction() {
    //     cout << "这是内部函数" << endl;
    // }

    // 正确做法：在外部定义函数
}

// 正确：在外部单独定义
void innerFunction() {
    cout << "这是内部函数" << endl;
}

void outerFunctionCorrect() {
    cout << "这是外部函数" << endl;
    innerFunction();  // 调用在外部定义的函数
}

int main() {
    outerFunctionCorrect();
    return 0;
}
```

### 7.4 参数个数不匹配

```cpp
#include <iostream>
using namespace std;

int add(int a, int b) {
    return a + b;
}

int main() {
    // 错误：参数个数不匹配
    // int result1 = add(1);        // 错误！少了一个参数
    // int result2 = add(1, 2, 3);  // 错误！多了一个参数

    // 正确：参数个数匹配
    int result = add(1, 2);
    cout << result << endl;

    return 0;
}
```

### 7.5 参数类型不匹配

```cpp
#include <iostream>
using namespace std;

int add(int a, int b) {
    return a + b;
}

int main() {
    // 类型不匹配时，编译器会尝试自动转换
    // 但如果转换可能丢失数据，会警告
    int result = add(3.7, 5.2);  // double转换为int，丢失小数
    cout << result << endl;      // 输出8，而不是8.9

    // 显式转换更清晰
    int result2 = add((int)3.7, (int)5.2);
    cout << result2 << endl;

    return 0;
}
```

## 八、综合练习示例

### 8.1 计算器程序

```cpp
#include <iostream>
using namespace std;

// 函数声明
double add(double a, double b);
double subtract(double a, double b);
double multiply(double a, double b);
double divide(double a, double b);
void printMenu();
char getOperator();

int main() {
    double num1, num2, result;
    char op;

    printMenu();

    cout << "请输入第一个数: ";
    cin >> num1;

    cout << "请输入运算符 (+, -, *, /): ";
    cin >> op;

    cout << "请输入第二个数: ";
    cin >> num2;

    // 根据运算符调用不同的函数
    switch (op) {
        case '+':
            result = add(num1, num2);
            break;
        case '-':
            result = subtract(num1, num2);
            break;
        case '*':
            result = multiply(num1, num2);
            break;
        case '/':
            result = divide(num1, num2);
            break;
        default:
            cout << "无效的运算符！" << endl;
            return 1;
    }

    cout << num1 << " " << op << " " << num2 << " = " << result << endl;

    return 0;
}

// 函数定义
double add(double a, double b) {
    return a + b;
}

double subtract(double a, double b) {
    return a - b;
}

double multiply(double a, double b) {
    return a * b;
}

double divide(double a, double b) {
    if (b == 0) {
        cout << "警告：除数为0！" << endl;
        return 0;
    }
    return a / b;
}

void printMenu() {
    cout << "========================" << endl;
    cout << "  简单计算器程序" << endl;
    cout << "========================" << endl;
}
```

### 8.2 数组处理函数

```cpp
#include <iostream>
using namespace std;

// 打印数组
void printArray(int arr[], int size) {
    cout << "[ ";
    for (int i = 0; i < size; i++) {
        cout << arr[i];
        if (i < size - 1) {
            cout << ", ";
        }
    }
    cout << " ]" << endl;
}

// 计算数组平均值
double getAverage(int arr[], int size) {
    int sum = 0;
    for (int i = 0; i < size; i++) {
        sum += arr[i];
    }
    return (double)sum / size;
}

// 查找数组中的最大值
int findMax(int arr[], int size) {
    int maxVal = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] > maxVal) {
            maxVal = arr[i];
        }
    }
    return maxVal;
}

// 查找数组中的最小值
int findMin(int arr[], int size) {
    int minVal = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] < minVal) {
            minVal = arr[i];
        }
    }
    return minVal;
}

int main() {
    int numbers[] = {64, 25, 12, 22, 11, 90, 45};
    int size = sizeof(numbers) / sizeof(numbers[0]);

    cout << "数组内容: ";
    printArray(numbers, size);

    cout << "平均值: " << getAverage(numbers, size) << endl;
    cout << "最大值: " << findMax(numbers, size) << endl;
    cout << "最小值: " << findMin(numbers, size) << endl;

    return 0;
}
```

## 九、注意事项总结

1. **函数定义在调用之前**：如果函数定义在main之后，需要提前声明（下一节详解）
2. **返回值类型要匹配**：return的值应该与函数声明的返回值类型一致
3. **void函数不需要return**：但可以用return提前结束函数
4. **参数传递是值传递**：形参的修改不会影响实参（指针和引用除外）
5. **不能嵌套定义函数**：所有函数必须在全局作用域定义
6. **函数名不能重复**：同一作用域内不能有同名函数（函数重载除外）
7. **建议**：一个函数只做一件事，函数名要能描述函数的功能
