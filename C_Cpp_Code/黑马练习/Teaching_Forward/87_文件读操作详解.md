# 文件读操作详解

## 1. 概述

读文件是把文件中的内容读取到程序中。C++使用 `ifstream` 类来读取文件，用法和 `cin` 非常相似。

---

## 2. 需要包含的头文件

```cpp
#include <fstream>  // 文件操作必须包含
#include <string>   // 使用 string 类时需要包含
```

---

## 3. ifstream 类简介

`ifstream` 是 **Input File Stream** 的缩写，意思是"输入文件流"。
它专门用于**从文件读取数据**。

类的继承关系：
```
ios_base -> ios -> istream -> ifstream
```

---

## 4. 读文件的基本步骤

读文件一共5个步骤：
1. 创建 `ifstream` 对象
2. 打开文件
3. 判断是否打开成功
4. 读取内容
5. 关闭文件

### 4.1 最简单的读文件示例

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    // 第1步：创建 ifstream 对象
    ifstream ifs;

    // 第2步：打开文件
    ifs.open("test.txt");

    // 第3步：判断是否打开成功
    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    // 第4步：读取内容（下面会详细讲解多种读取方式）
    string buf;
    while (getline(ifs, buf)) {  // 逐行读取
        cout << buf << endl;
    }

    // 第5步：关闭文件
    ifs.close();

    return 0;
}
```

### 4.2 用构造函数直接打开文件


## 5. 读取的三种方式

### 5.1 方式一：逐单词读取（ifs >> 变量）

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    // 假设 test.txt 内容为：
    // Hello World
    // 你好 世界

    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string buf;

    // 用 >> 运算符逐单词读取
    // 特点：遇到空格或换行就停止
    while (ifs >> buf) {
        cout << buf << endl;
    }
    // 输出：
    // Hello
    // World
    // 你好
    // 世界

    ifs.close();
    return 0;
}
```

```cpp
// 逐单词读取的详细说明
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    // 假设 data.txt 内容为：
    // 张三 20 95.5
    // 李四 21 88.0

    ifstream ifs("data.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string name;
    int age;
    double score;

    // 可以直接读取到不同类型的变量中
    while (ifs >> name >> age >> score) {
        cout << "姓名：" << name << "，年龄：" << age << "，分数：" << score << endl;
    }

    ifs.close();
    return 0;
}
```

### 5.2 方式二：逐行读取（getline）【推荐】

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    // 假设 test.txt 内容为：
    // Hello World
    // 你好 世界
    // 这是第三行

    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string line;

    // getline(流对象, 字符串变量) 逐行读取
    // 特点：遇到换行才停止，空格也会读入
    // 返回值：成功返回true，读到文件末尾返回false
    while (getline(ifs, line)) {
        cout << line << endl;
    }
    // 输出：
    // Hello World
    // 你好 世界
    // 这是第三行

    ifs.close();
    return 0;
}
```

```cpp
// getline 读取带空格的整行内容
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    // 假设 info.txt 内容为：
    // 张三 北京市海淀区
    // 李四 上海市浦东新区

    ifstream ifs("info.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string line;
    int lineNum = 0;

    while (getline(ifs, line)) {
        lineNum++;
        cout << "第" << lineNum << "行：" << line << endl;
    }

    cout << "共读取了 " << lineNum << " 行" << endl;

    ifs.close();
    return 0;
}
```

### 5.3 方式三：逐字符读取（ifs.get()）

```cpp
#include <iostream>
#include <fstream>
using namespace std;

int main() {
    // 假设 test.txt 内容为：
    // AB
    // CD

    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    char ch;
    int count = 0;

    // get() 方法每次读取一个字符（包括空格和换行）
    // 读到文件末尾返回 EOF
    while ((ch = ifs.get()) != EOF) {
        cout << ch;
        count++;
    }
    // 输出：AB\nCD\n（原样输出所有字符）

    cout << "\n共读取了 " << count << " 个字符" << endl;

    ifs.close();
    return 0;
}
```

```cpp
// 逐字符读取并统计特定字符
#include <iostream>
#include <fstream>
using namespace std;

int main() {
    // 创建测试文件
    ofstream ofs("test_char.txt");
    ofs << "Hello World" << endl;
    ofs << "C++ Programming" << endl;
    ofs.close();

    ifstream ifs("test_char.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    char ch;
    int spaceCount = 0;
    int letterCount = 0;
    int totalCount = 0;

    while ((ch = ifs.get()) != EOF) {
        totalCount++;
        if (ch == ' ') {
            spaceCount++;
        }
        if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')) {
            letterCount++;
        }
    }

    cout << "总字符数：" << totalCount << endl;
    cout << "空格数：" << spaceCount << endl;
    cout << "字母数：" << letterCount << endl;

    ifs.close();
    return 0;
}
```

## 6. 三种读取方式的对比

| 方式 | 语法 | 特点 | 适用场景 |
|------|------|------|----------|
| 逐单词 | `ifs >> buf` | 遇到空格/换行停止 | 读取格式化数据 |
| 逐行 | `getline(ifs, line)` | 遇到换行停止 | 读取整行文本【推荐】 |
| 逐字符 | `ifs.get()` | 每次一个字符 | 需要逐字符处理 |

## 7. 文件结束的判断

### 7.1 使用 EOF 判断

```cpp
#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    char ch;

    // EOF (End Of File) 是文件结束标志
    // 当读取到文件末尾时，get() 返回 EOF
    while ((ch = ifs.get()) != EOF) {
        cout << ch;
    }

    ifs.close();
    return 0;
}
```

### 7.2 使用 getline 返回值判断

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string line;

    // getline 返回流对象本身，流对象在成功时转为true，失败时转为false
    while (getline(ifs, line)) {
        cout << line << endl;
    }

    ifs.close();
    return 0;
}
```

### 7.3 使用 >> 运算符判断

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string word;

    // >> 运算符返回流对象本身，读取失败时转为false
    while (ifs >> word) {
        cout << word << endl;
    }

    ifs.close();
    return 0;
}
```

## 9. 读取打开模式

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    // ios::in 是默认模式，表示以输入（读取）方式打开
    ifstream ifs("test.txt", ios::in);

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string line;
    while (getline(ifs, line)) {
        cout << line << endl;
    }

    ifs.close();

    // 以二进制模式读取
    ifstream ifs2("data.dat", ios::in | ios::binary);
    // ... 读取二进制数据
    ifs2.close();

    return 0;
}
```

## 10. 完整的实际应用示例

### 10.1 统计文件行数

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    int lineCount = 0;
    string line;

    while (getline(ifs, line)) {
        lineCount++;
    }

    cout << "文件共有 " << lineCount << " 行" << endl;

    ifs.close();
    return 0;
}
```

### 10.2 读取并解析CSV文件

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    // 假设 data.csv 内容为：
    // 姓名,年龄,城市
    // 张三,20,北京
    // 李四,21,上海

    ifstream ifs("data.csv");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string line;
    int lineNum = 0;

    while (getline(ifs, line)) {
        lineNum++;
        if (lineNum == 1) {
            cout << "表头：" << line << endl;
            cout << "-------------------" << endl;
        } else {
            cout << "第" << lineNum - 1 << "条数据：" << line << endl;
        }
    }

    ifs.close();
    return 0;
}
```

### 10.3 在文件中搜索特定内容

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string keyword;
    cout << "请输入要搜索的关键词：";
    cin >> keyword;

    string line;
    int lineNum = 0;
    bool found = false;

    while (getline(ifs, line)) {
        lineNum++;

        // string::find 查找子字符串
        // 如果找到，返回位置；找不到返回 string::npos
        if (line.find(keyword) != string::npos) {
            cout << "第" << lineNum << "行找到：" << line << endl;
            found = true;
        }
    }

    if (!found) {
        cout << "未找到包含 \"" << keyword << "\" 的内容" << endl;
    }

    ifs.close();
    return 0;
}
```

### 10.4 读取文件并显示内容（带行号）

```cpp
#include <iostream>
#include <fstream>
#include <string>
#include <iomanip>  // 用于 setw 设置宽度
using namespace std;

int main() {
    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string line;
    int lineNum = 0;

    cout << "文件内容：" << endl;
    cout << "================================" << endl;

    while (getline(ifs, line)) {
        lineNum++;
        // setw(4) 设置宽度为4，right 右对齐

    }

    cout << "================================" << endl;
    cout << "共 " << lineNum << " 行" << endl;

    ifs.close();
    return 0;
}
```

## 11. 常见错误和注意事项

### 11.1 忘记检查文件是否打开成功

```cpp
#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream ifs("test.txt");

    // 错误：没有检查就直接读取
    // string line;
    // while (getline(ifs, line)) {  // 如果文件不存在，循环不会执行，但不会报错
    //     cout << line << endl;
    // }

    // 正确：先检查
    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    string line;
    while (getline(ifs, line)) {
        cout << line << endl;
    }

    ifs.close();
    return 0;
}
```

### 11.2 读取类型不匹配

```cpp
#include <iostream>
#include <fstream>
using namespace std;

int main() {
    // 假设 test.txt 内容为：
    // Hello 123

    ifstream ifs("test.txt");

    if (!ifs.is_open()) {
        cout << "文件打开失败！" << endl;
        return -1;
    }

    // 错误示范：文件内容是 "Hello"，但试图读入 int
    int num;
    ifs >> num;  // 读取失败！Hello 不是数字
    cout << "num = " << num << endl;  // 输出：num = 0（失败时变量不变或为0）

    // 正确做法：先读入字符串，再转换
    ifs.clear();  // 清除错误标志
    ifs.seekg(0); // 回到文件开头

    string str;
    ifs >> str;
    cout << "str = " << str << endl;  // 输出：str = Hello

    ifs.close();
    return 0;
}
```

### 11.3 文件路径问题

```cpp
#include <iostream>
#include <fstream>
using namespace std;

int main() {
    // 问题1：相对路径依赖于程序运行的目录
    ifstream ifs1("test.txt");  // 在哪里找 test.txt？

    // 问题2：路径分隔符
    // ifstream ifs2("C:\new\test.txt");  // 错误！\n 和 \t 是转义字符
    ifstream ifs3("C:\\new\\test.txt");   // 正确
    ifstream ifs4("C:/new/test.txt");     // 正确（推荐）

    return 0;
}
```

## 12. 总结

| 要点 | 说明 |
|------|------|
| 头文件 | `#include <fstream>` |
| 读文件类 | `ifstream` |
| 打开文件 | `ifs.open("路径")` 或构造函数传参 |
| 判断打开 | `ifs.is_open()` |
| 逐单词读 | `ifs >> 变量` |
| 逐行读 | `getline(ifs, 变量)`【推荐】 |
| 逐字符读 | `ifs.get()` |
| 判断结束 | EOF 或 getline 返回 false |
| 关闭文件 | `ifs.close()` |

记住这5个步骤：**创建对象 -> 打开文件 -> 判断成功 -> 读取内容 -> 关闭文件**

推荐使用 `getline` 逐行读取，这是最常用、最安全的方式。
