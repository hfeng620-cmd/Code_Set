# 加号与左移运算符重载

## 一、加号 + 运算符重载

加号 `+` 运算符重载是最常见的运算符重载之一，用于让两个自定义类型的对象可以相加。

---

### 1.1 成员函数重载加号

**核心思想：** 左操作数是 `this`，右操作数是函数参数。

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
public:
    string m_Name;
    int m_Age;

    Person() : m_Name(""), m_Age(0) {}
    Person(string name, int age) : m_Name(name), m_Age(age) {}

    // 成员函数重载 + 号
    // this 指向左操作数（p1），p 是右操作数（p2）
    // 调用方式：p1.operator+(p2)  等价于  p1 + p2
    Person operator+(const Person &p) {
        Person temp;
        temp.m_Name = this->m_Name + "和" + p.m_Name;
        temp.m_Age = this->m_Age + p.m_Age;
        return temp;
    }

    void showInfo() {
        cout << "姓名: " << m_Name << ", 年龄: " << m_Age << endl;
    }
};

int main() {
    Person p1("张三", 18);
    Person p2("李四", 20);

    // 方式1：直接用 + 号（推荐）
    Person p3 = p1 + p2;
    p3.showInfo();  // 输出：姓名: 张三和李四, 年龄: 38

    // 方式2：显式调用成员函数（理解原理用）
    Person p4 = p1.operator+(p2);
    p4.showInfo();  // 输出：姓名: 张三和李四, 年龄: 38

    return 0;
}
```

**编译器的转换过程：**

```
p1 + p2
    ↓ 编译器转换
p1.operator+(p2)
    ↓ 调用成员函数
传入 p2 作为参数，this 指向 p1
    ↓ 函数内部
创建 temp，计算结果，返回 temp
```

---

### 1.2 全局函数重载加号

**核心思想：** 两个操作数都作为函数参数传入。

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
public:
    string m_Name;
    int m_Age;

    Person() : m_Name(""), m_Age(0) {}
    Person(string name, int age) : m_Name(name), m_Age(age) {}

    void showInfo() {
        cout << "姓名: " << m_Name << ", 年龄: " << m_Age << endl;
    }
};

// 全局函数重载 + 号
// 两个操作数都作为参数传入
// 调用方式：operator+(p1, p2)  等价于  p1 + p2
Person operator+(const Person &p1, const Person &p2) {
    Person temp;
    temp.m_Name = p1.m_Name + "和" + p2.m_Name;
    temp.m_Age = p1.m_Age + p2.m_Age;
    return temp;
}

int main() {
    Person p1("张三", 18);
    Person p2("李四", 20);

    // p1 + p2 编译器会转换为 operator+(p1, p2)
    Person p3 = p1 + p2;
    p3.showInfo();  // 输出：姓名: 张三和李四, 年龄: 38

    return 0;
}
```

**编译器的转换过程：**

```
p1 + p2
    ↓ 编译器转换
operator+(p1, p2)
    ↓ 调用全局函数
p1 传给第一个参数，p2 传给第二个参数
    ↓ 函数内部
创建 temp，计算结果，返回 temp
```

### 1.3 不同类型之间的加法运算

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
public:
    string m_Name;
    int m_Age;

    Person() : m_Name(""), m_Age(0) {}
    Person(string name, int age) : m_Name(name), m_Age(age) {}

    // 成员函数重载：Person + Person
    Person operator+(const Person &p) {
        Person temp;
        temp.m_Name = this->m_Name + "和" + p.m_Name;
        temp.m_Age = this->m_Age + p.m_Age;
        return temp;
    }

    // 成员函数重载：Person + int
    // 让 Person 对象可以和 int 类型相加
    Person operator+(int age) {
        Person temp;
        temp.m_Name = this->m_Name;
        temp.m_Age = this->m_Age + age;
        return temp;
    }

    void showInfo() {
        cout << "姓名: " << m_Name << ", 年龄: " << m_Age << endl;
    }
};

// 全局函数重载：int + Person
// 注意：如果用成员函数重载，左操作数必须是 Person
// 所以 int + Person 必须用全局函数重载
Person operator+(int age, const Person &p) {
    Person temp;
    temp.m_Name = p.m_Name;
    temp.m_Age = age + p.m_Age;
    return temp;
}

int main() {
    Person p1("张三", 18);

    // Person + int（成员函数重载）
    Person p2 = p1 + 2;  // 等价于 p1.operator+(2)
    p2.showInfo();        // 输出：姓名: 张三, 年龄: 20

    // int + Person（全局函数重载）
    Person p3 = 5 + p1;  // 等价于 operator+(5, p1)
    p3.showInfo();        // 输出：姓名: 张三, 年龄: 23

    // Person + Person（成员函数重载）
    Person p4("李四", 20);
    Person p5 = p1 + p4;
    p5.showInfo();        // 输出：姓名: 张三和李四, 年龄: 38

    return 0;
}
```

**注意：** 如果左操作数不是当前类的对象（如 `int + Person`），必须使用**全局函数重载**，因为成员函数的 `this` 指针只能指向当前类的对象。

### 1.4 加号重载的注意事项

```cpp
#include <iostream>
using namespace std;

class MyInt {
public:
    int m_Value;

    MyInt(int v = 0) : m_Value(v) {}

    // 重载 + 号
    MyInt operator+(const MyInt &other) {
        return MyInt(this->m_Value + other.m_Value);
    }

    void show() {
        cout << "值: " << m_Value << endl;
    }
};

int main() {
    MyInt a(10);
    MyInt b(20);

    // 注意1：加号重载不会改变原对象
    MyInt c = a + b;
    a.show();  // 输出：10（a 没有被修改）
    b.show();  // 输出：20（b 没有被修改）
    c.show();  // 输出：30（c 是新对象）

    // 注意2：支持链式运算（因为返回的是对象）
    MyInt d(30);
    MyInt e = a + b + d;  // 等价于 (a + b) + d
    e.show();  // 输出：60

    // 注意3：可以和基本类型混合运算（如果有对应的重载或构造函数）
    MyInt f = a + 100;  // 100 通过构造函数转为 MyInt(100)
    f.show();  // 输出：110

    return 0;
}
```

## 二、左移 << 运算符重载

左移运算符 `<<` 的重载主要用于配合 `cout` 实现**自定义类型的输出**。

---

### 2.1 为什么不能用成员函数重载 << ？

```cpp
#include <iostream>
using namespace std;

class Person {
public:
    string m_Name;
    int m_Age;

    Person(string name, int age) : m_Name(name), m_Age(age) {}

    // 如果用成员函数重载 << 运算符
    // this 指向左操作数，p 是右操作数
    // 那么调用方式是：p1.operator<<(cout)
    // 也就是：p1 << cout
    // 这和我们想要的 cout << p1 完全相反！

    // 错误示例（不要这样做）：
    // void operator<<(ostream &out) {
    //     out << "姓名: " << m_Name << ", 年龄: " << m_Age;
    // }

    // 这样调用的话：
    // p1 << cout;  // 语法上可以，但语义完全不对！
    // 我们想要的是：cout << p1;
};

int main() {
    Person p("张三", 18);

    // 我们想要的调用方式：cout << p;
    // 但是成员函数重载会让它变成：p << cout;
    // 所以成员函数重载 << 不适用于 cout 的情况

    return 0;
}
```

**结论：** `<<` 运算符重载**不能用成员函数**，因为 `cout` 在左边，而成员函数的 `this` 在左边，两者冲突。必须使用**全局函数重载**。

---

### 2.2 全局函数重载 << 运算符

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
public:
    string m_Name;
    int m_Age;

    Person(string name, int age) : m_Name(name), m_Age(age) {}
};

// 全局函数重载 << 运算符
// 参数1：ostream &out  ——  左操作数（cout），用引用传递
// 参数2：const Person &p  ——  右操作数（Person 对象），用 const 引用
// 返回值：ostream&  ——  返回输出流的引用
ostream& operator<<(ostream &out, const Person &p) {
    out << "姓名: " << p.m_Name << ", 年龄: " << p.m_Age;
    return out;  // 返回 out 的引用，支持链式输出
}

int main() {
    Person p("张三", 18);

    // 使用重载的 << 运算符
    cout << p << endl;
    // 输出：姓名: 张三, 年龄: 18

    return 0;
}
```

**编译器的转换过程：**

```
cout << p
    ↓ 编译器转换
operator<<(cout, p)
    ↓ 调用全局函数
cout 传给第一个参数 out，p 传给第二个参数 p
    ↓ 函数内部
out << "姓名: " << p.m_Name << ...
    ↓ 返回
返回 out 的引用（就是 cout 本身）
```

---

### 2.3 返回 ostream& 的意义（链式输出）

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
public:
    string m_Name;
    int m_Age;

    Person(string name, int age) : m_Name(name), m_Age(age) {}
};

// 返回 ostream& 的意义：支持链式输出
// 如果返回 void，就无法链式输出
ostream& operator<<(ostream &out, const Person &p) {
    out << "姓名: " << p.m_Name << ", 年龄: " << p.m_Age;
    return out;  // 关键：返回 out 的引用
}

int main() {
    Person p1("张三", 18);
    Person p2("李四", 20);
    Person p3("王五", 25);

    // 链式输出：cout << p1 << p2 << p3 << endl;
    // 编译器解析过程：
    // 第1步：operator<<(cout, p1)  → 返回 cout 引用
    // 第2步：operator<<(cout, p2)  → 返回 cout 引用
    // 第3步：operator<<(cout, p3)  → 返回 cout 引用
    // 第4步：cout << endl

    cout << p1 << " | " << p2 << " | " << p3 << endl;
    // 输出：姓名: 张三, 年龄: 18 | 姓名: 李四, 年龄: 20 | 姓名: 王五, 年龄: 25

    return 0;
}
```

**如果不返回引用会怎样？**

```cpp
// 错误示例：返回 void
void operator<<(ostream &out, const Person &p) {
    out << "姓名: " << p.m_Name << ", 年龄: " << p.m_Age;
    // 没有返回值
}

int main() {
    Person p1("张三", 18);
    Person p2("李四", 20);

    cout << p1 << endl;       // OK：单次输出没问题

    // cout << p1 << p2;      // 错误！因为 cout << p1 返回 void，void << p2 不合法
    // cout << p1 << endl;    // 错误！因为 cout << p1 返回 void，void << endl 不合法

    // 如果要输出多个，只能分开写：
    cout << p1;    // OK
    cout << p2;    // OK

    return 0;
}
```

### 2.4 需要声明为友元的情况

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
    // 如果成员变量是私有的，全局函数无法直接访问
    // 需要在类内声明该函数为友元函数
    friend ostream& operator<<(ostream &out, const Person &p);

private:  // 私有成员
    string m_Name;
    int m_Age;

public:
    Person(string name, int age) : m_Name(name), m_Age(age) {}
};

// 全局函数重载 << 运算符
// 因为是友元函数，所以可以访问 Person 的私有成员
ostream& operator<<(ostream &out, const Person &p) {
    out << "姓名: " << p.m_Name << ", 年龄: " << p.m_Age;
    return out;
}

int main() {
    Person p("张三", 18);

    // cout << p.m_Name;  // 错误！m_Name 是私有的，外部不能直接访问
    cout << p << endl;     // 正确！通过重载的 << 运算符访问

    return 0;
}
```

### 2.5 完整的加号和左移运算符重载示例

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
    // 声明友元函数
    friend ostream& operator<<(ostream &out, const Person &p);

public:
    string m_Name;
    int m_Age;
    int m_Score;

    // 构造函数
    Person() : m_Name(""), m_Age(0), m_Score(0) {}
    Person(string name, int age, int score)
        : m_Name(name), m_Age(age), m_Score(score) {}

    // 成员函数重载 + 号
    // 两个 Person 相加：名字拼接，年龄和分数相加
    Person operator+(const Person &p) {
        Person temp;
        temp.m_Name = this->m_Name + "和" + p.m_Name;
        temp.m_Age = this->m_Age + p.m_Age;
        temp.m_Score = this->m_Score + p.m_Score;
        return temp;
    }

    // 成员函数重载 + 号：Person + int（加分）
    Person operator+(int score) {
        Person temp;
        temp.m_Name = this->m_Name;
        temp.m_Age = this->m_Age;
        temp.m_Score = this->m_Score + score;
        return temp;
    }
};

// 全局函数重载 << 运算符
ostream& operator<<(ostream &out, const Person &p) {
    out << "[姓名: " << p.m_Name
        << ", 年龄: " << p.m_Age
        << ", 分数: " << p.m_Score << "]";
    return out;
}

// 全局函数重载：int + Person
Person operator+(int score, const Person &p) {
    Person temp;
    temp.m_Name = p.m_Name;
    temp.m_Age = p.m_Age;
    temp.m_Score = score + p.m_Score;
    return temp;
}

int main() {
    Person p1("张三", 18, 85);
    Person p2("李四", 20, 90);

    // 测试 + 号重载：Person + Person
    cout << "=== Person + Person ===" << endl;
    Person p3 = p1 + p2;
    cout << "p1 = " << p1 << endl;  // [姓名: 张三, 年龄: 18, 分数: 85]
    cout << "p2 = " << p2 << endl;  // [姓名: 李四, 年龄: 20, 分数: 90]
    cout << "p1 + p2 = " << p3 << endl;  // [姓名: 张三和李四, 年龄: 38, 分数: 175]

    // 测试 + 号重载：Person + int
    cout << "\n=== Person + int ===" << endl;
    Person p4 = p1 + 15;  // 张三分数加15
    cout << "p1 + 15 = " << p4 << endl;  // [姓名: 张三, 年龄: 18, 分数: 100]

    // 测试 + 号重载：int + Person
    cout << "\n=== int + Person ===" << endl;
    Person p5 = 200 + p1;  // 200 + 张三的分数
    cout << "200 + p1 = " << p5 << endl;  // [姓名: 张三, 年龄: 18, 分数: 285]

    // 测试链式输出
    cout << "\n=== 链式输出 ===" << endl;
    cout << p1 << " | " << p2 << " | " << p3 << endl;

    return 0;
}
```

## 三、常见错误和注意事项

### 错误1：忘记返回 ostream&

```cpp
// 错误：返回 void，无法链式输出
void operator<<(ostream &out, const Person &p) {
    out << p.m_Name;
    // 没有 return out;
}

// 正确：返回 ostream&，支持链式输出
ostream& operator<<(ostream &out, const Person &p) {
    out << p.m_Name;
    return out;
}
```

### 错误2：ostream 参数没有用引用

```cpp
// 错误：ostream 不能拷贝，必须用引用
// ostream operator<<(ostream out, const Person &p) { }  // 编译错误

// 正确：用引用
ostream& operator<<(ostream &out, const Person &p) { }
```

### 错误3：Person 参数没有用 const

```cpp
// 不推荐：没有用 const，可能会意外修改 Person 对象
ostream& operator<<(ostream &out, Person &p) {
    p.m_Name = "被修改了";  // 不应该在输出函数里修改对象
    out << p.m_Name;
    return out;
}

// 推荐：用 const 保护 Person 对象
ostream& operator<<(ostream &out, const Person &p) {
    // p.m_Name = "被修改了";  // 编译错误！const 对象不能被修改
    out << p.m_Name;
    return out;
}
```

### 错误4：加号重载修改了原对象

```cpp
// 错误：加号重载不应该修改原对象
Person operator+(const Person &p) {
    this->m_Age += p.m_Age;  // 修改了原对象！
    return *this;
}

// 正确：创建新对象返回，不修改原对象
Person operator+(const Person &p) {
    Person temp;
    temp.m_Name = this->m_Name + "和" + p.m_Name;
    temp.m_Age = this->m_Age + p.m_Age;
    return temp;
}
```

## 四、总结

| 重载方式 | 适用场景 | 要点 |
|---------|---------|------|
| 成员函数重载 + | 左操作数是当前类的对象 | this 是左操作数，参数是右操作数 |
| 全局函数重载 + | 左操作数不是当前类的对象 | 所有操作数都是参数 |
| 全局函数重载 << | 配合 cout 输出自定义类型 | 必须返回 ostream&，支持链式输出 |

**关键记忆点：**
1. `<<` 重载**必须用全局函数**（因为 cout 在左边）
2. `<<` 重载**必须返回 `ostream&`**（支持链式输出）
3. 如果成员变量是私有的，需要声明**友元函数**
4. `+` 重载**不要修改原对象**，创建新对象返回
5. 参数尽量用 **const 引用**传递，避免不必要的拷贝
