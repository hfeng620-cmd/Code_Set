# 类模板详解

## 一、类模板的基本语法

类模板允许我们将类中的数据类型参数化，创建一个通用的类。
使用时再指定具体的数据类型。

```cpp
template <typename T1, typename T2>
// 或者写成 template <class T1, class T2>  两者完全等价
class 类名 {
    // 类成员定义
};
```

### 1.1 基本示例

```cpp
#include <iostream>
#include <string>
using namespace std;

// 类模板：通用的配对类
template <class T1, class T2>
class Pair {
public:
    T1 m_First;     // 第一个值，类型为 T1
    T2 m_Second;    // 第二个值，类型为 T2

    Pair(T1 first, T2 second) : m_First(first), m_Second(second) {}

    void show() {
        cout << "第一个值：" << m_First << "，第二个值：" << m_Second << endl;
    }
};

int main() {
    // 使用类模板时必须显式指定类型
    Pair<string, int> p1("张三", 20);
    p1.show();   // 第一个值：张三，第二个值：20

    Pair<int, double> p2(100, 3.14);
    p2.show();   // 第一个值：100，第二个值：3.14

    Pair<string, string> p3("Hello", "World");
    p3.show();   // 第一个值：Hello，第二个值：World

    return 0;
}
```

### 1.2 多个模板参数

```cpp
#include <iostream>
#include <string>
using namespace std;

// 三个模板参数
template <class T1, class T2, class T3>
class Triple {
public:
    T1 m_First;
    T2 m_Second;
    T3 m_Third;

    Triple(T1 a, T2 b, T3 c) : m_First(a), m_Second(b), m_Third(c) {}

    void show() {
        cout << m_First << " - " << m_Second << " - " << m_Third << endl;
    }
};

int main() {
    Triple<string, int, double> t("张三", 20, 95.5);
    t.show();   // 张三 - 20 - 95.5

    return 0;
}
```

## 二、类模板必须显式指定类型

与函数模板不同，类模板 **不能自动推导类型**，必须在使用时显式指定。

```cpp
#include <iostream>
using namespace std;

template <class T>
class MyClass {
public:
    T m_Data;
    MyClass(T data) : m_Data(data) {}
};

// 函数模板可以自动推导
template <class T>
void func(T val) {
    cout << val << endl;
}

int main() {
    // 函数模板：可以自动推导
    func(10);           // 推导 T = int
    func(3.14);         // 推导 T = double

    // 类模板：必须显式指定类型
    MyClass<int> obj1(10);          // 必须写 <int>
    MyClass<double> obj2(3.14);     // 必须写 <double>
    MyClass<string> obj3("Hello");  // 必须写 <string>

    // 错误：不能省略类型
    // MyClass obj4(10);   // 编译错误！

    return 0;
}
```

## 三、类模板与函数模板的区别

| 特性 | 函数模板 | 类模板 |
|------|---------|--------|
| 类型推导 | 可以自动推导 | 不能自动推导，必须显式指定 |
| 默认类型参数 | C++11 支持 | 一直支持 |
| 使用场景 | 函数 | 类 |

### 3.1 类模板可以有默认类型参数

```cpp
#include <iostream>
#include <string>
using namespace std;

// 类模板可以设置默认类型参数
template <class T1, class T2 = int>     // T2 默认为 int
class Person {
public:
    T1 m_Name;
    T2 m_Age;

    Person(T1 name, T2 age) : m_Name(name), m_Age(age) {}

    void show() {
        cout << "姓名：" << m_Name << "，年龄：" << m_Age << endl;
    }
};

int main() {
    // 使用默认类型 T2 = int
    Person<string> p1("张三", 20);
    p1.show();   // 姓名：张三，年龄：20

    // 不使用默认类型，手动指定
    Person<string, double> p2("李四", 20.5);
    p2.show();   // 姓名：李四，年龄：20.5

    return 0;
}
```

## 四、类模板成员函数的创建时机

类模板中的成员函数，在 **调用时才创建**，而不是在类实例化时。
这意味着如果某个成员函数没有被调用，即使它对某种类型不合法，也不会报错。

```cpp
#include <iostream>
#include <string>
using namespace std;

class PersonA {
public:
    void showA() { cout << "PersonA showA" << endl; }
};

class PersonB {
public:
    void showB() { cout << "PersonB showB" << endl; }
};

template <class T>
class MyClass {
public:
    T m_Obj;

    // 这个函数调用 T 的 showA()
    void funcA() {
        m_Obj.showA();
    }

    // 这个函数调用 T 的 showB()
    void funcB() {
        m_Obj.showB();
    }
};

int main() {
    // PersonA 有 showA，没有 showB
    MyClass<PersonA> objA;
    objA.funcA();       // OK：调用时创建 funcA，PersonA 有 showA
    // objA.funcB();    // 如果取消注释会报错！因为 PersonA 没有 showB

    // PersonB 有 showB，没有 showA
    MyClass<PersonB> objB;
    objB.funcB();       // OK：调用时创建 funcB，PersonB 有 showB
    // objB.funcA();    // 如果取消注释会报错！因为 PersonB 没有 showA

    return 0;
}
```

## 五、类模板做函数参数

类模板对象作为函数参数传递时，有三种方式。

```cpp
#include <iostream>
#include <string>
using namespace std;

template <class T1, class T2>
class Person {
public:
    T1 m_Name;
    T2 m_Age;

    Person(T1 name, T2 age) : m_Name(name), m_Age(age) {}
};
```

### 5.1 方式一：指定传入类型（最常用）

```cpp
// 直接指定 Person 的模板参数类型
void printPerson1(Person<string, int>& p) {
    cout << "方式一：" << p.m_Name << "，" << p.m_Age << endl;
}

int main() {
    Person<string, int> p("张三", 20);
    printPerson1(p);    // 方式一：张三，20
    return 0;
}
```

### 5.2 方式二：参数模板化（函数模板配合）

```cpp
// 将函数模板化，让编译器自动推导 Person 的类型
template <class T1, class T2>
void printPerson2(Person<T1, T2>& p) {
    cout << "方式二：" << p.m_Name << "，" << p.m_Age << endl;
}

int main() {
    Person<string, int> p("张三", 20);
    printPerson2(p);    // 方式二：张三，20
    return 0;
}
```

### 5.3 方式三：整个类模板化

```cpp
// 把整个 Person 当作一个类型 T
template <class T>
void printPerson3(T& p) {
    cout << "方式三：" << p.m_Name << "，" << p.m_Age << endl;
}

int main() {
    Person<string, int> p("张三", 20);
    printPerson3(p);    // 方式三：张三，20
    return 0;
}
```

### 5.4 三种方式对比

```cpp
/*
    方式一：指定传入类型
        void func(Person<string, int>& p)
        优点：直观，类型明确
        缺点：不够灵活，只能传固定类型

    方式二：参数模板化
        template<class T1, class T2>
        void func(Person<T1, T2>& p)
        优点：灵活，自动推导
        缺点：稍微复杂

    方式三：整个类模板化
        template<class T>
        void func(T& p)
        优点：最灵活，任何类型都能传
        缺点：类型不明确，容易传错
*/
```

## 六、类模板与继承

### 6.1 子类必须指明父类的 T 类型

```cpp
#include <iostream>
using namespace std;

template <class T>
class Base {
public:
    T m_Data;
    Base(T data) : m_Data(data) {}
};

// 错误写法：编译器不知道 T 是什么类型，无法分配内存
// class Son : public Base {
// };

// 正确写法1：子类明确指定父类的类型
class Son1 : public Base<int> {
public:
    Son1(int data) : Base<int>(data) {}
};

// 正确写法2：子类也是模板，用子类的 T 传递给父类
template <class T>
class Son2 : public Base<T> {
public:
    Son2(T data) : Base<T>(data) {}
};

int main() {
    Son1 s1(100);
    cout << s1.m_Data << endl;   // 100

    Son2<double> s2(3.14);
    cout << s2.m_Data << endl;   // 3.14

    Son2<string> s3("Hello");
    cout << s3.m_Data << endl;   // Hello

    return 0;
}
```

### 6.2 子类模板指定固定类型给父类

```cpp
#include <iostream>
using namespace std;

template <class T>
class Base {
public:
    T m_Data;
    Base(T data) : m_Data(data) {}
    void show() { cout << "Base data: " << m_Data << endl; }
};

// 子类模板：用自己的模板类型 U 传给父类
template <class U>
class Derived : public Base<U> {
public:
    U m_Extra;

    Derived(U data, U extra) : Base<U>(data), m_Extra(extra) {}

    void showAll() {
        this->show();   // 注意：模板类继承时调用父类函数要用 this->
        cout << "Derived extra: " << m_Extra << endl;
    }
};

int main() {
    Derived<int> d(100, 200);
    d.showAll();
    // Base data: 100
    // Derived extra: 200

    return 0;
}
```

**注意：** 模板类继承中，如果子类要调用父类的成员函数，需要用 `this->` 前缀，
否则编译器可能找不到（因为模板的两阶段编译机制）。

## 七、类模板分文件问题与 .hpp 解决方案

### 7.1 问题描述

普通类可以将声明放在 `.h` 文件，实现放在 `.cpp` 文件。
但类模板不行！因为类模板的成员函数在调用时才创建，
编译器需要同时看到声明和实现才能正确编译。

```
// person.h        - 只有声明
// person.cpp       - 有实现
// main.cpp         - #include "person.h"
// 结果：链接错误！编译器在 person.cpp 中看不到模板实现
```

### 7.2 解决方案：使用 .hpp 文件

将类模板的声明和实现都写在一个 `.hpp` 文件中。

```cpp
// ========== person.hpp ==========
#ifndef PERSON_HPP
#define PERSON_HPP

#include <iostream>
#include <string>
using namespace std;

template <class T1, class T2>
class Person {
public:
    T1 m_Name;
    T2 m_Age;

    Person(T1 name, T2 age);   // 构造函数声明
    void show();                // 成员函数声明
};

// 构造函数实现（注意模板语法）
template <class T1, class T2>
Person<T1, T2>::Person(T1 name, T2 age) : m_Name(name), m_Age(age) {}

// show 函数实现
template <class T1, class T2>
void Person<T1, T2>::show() {
    cout << "姓名：" << m_Name << "，年龄：" << m_Age << endl;
}

#endif
```

```cpp
// ========== main.cpp ==========
#include <iostream>
#include "person.hpp"   // 包含 .hpp 文件

int main() {
    Person<string, int> p("张三", 20);
    p.show();   // 姓名：张三，年龄：20

    return 0;
}
```

### 7.3 另一种方案：全部写在 .h 中

```cpp
// ========== person.h ==========
#ifndef PERSON_H
#define PERSON_H

#include <iostream>
#include <string>
using namespace std;

template <class T1, class T2>
class Person {
public:
    T1 m_Name;
    T2 m_Age;

    // 直接在类内实现（适合较短的函数）
    Person(T1 name, T2 age) : m_Name(name), m_Age(age) {}

    void show() {
        cout << "姓名：" << m_Name << "，年龄：" << m_Age << endl;
    }
};

#endif
```

## 八、类模板与友元

### 8.1 类内实现友元（简单方式）

```cpp
#include <iostream>
#include <string>
using namespace std;

template <class T1, class T2>
class Person {
    // 类内声明友元函数（类内实现）
    friend void printPerson(Person<T1, T2>& p) {
        cout << "姓名：" << p.m_Name << "，年龄：" << p.m_Age << endl;
    }

public:
    T1 m_Name;
    T2 m_Age;

    Person(T1 name, T2 age) : m_Name(name), m_Age(age) {}
};

int main() {
    Person<string, int> p("张三", 20);
    printPerson(p);     // 姓名：张三，年龄：20
    return 0;
}
```

### 8.2 类外实现友元（复杂方式）

```cpp
#include <iostream>
#include <string>
using namespace std;

// 前置声明：类模板
template <class T1, class T2>
class Person;

// 前置声明：友元函数模板
template <class T1, class T2>
void printPerson(Person<T1, T2>& p);

// 类模板定义
template <class T1, class T2>
class Person {
    // 类内声明友元
    friend void printPerson<T1, T2>(Person<T1, T2>& p);

public:
    T1 m_Name;
    T2 m_Age;

    Person(T1 name, T2 age) : m_Name(name), m_Age(age) {}
};

// 类外实现友元函数（注意 <> 模板语法）
template <class T1, class T2>
void printPerson(Person<T1, T2>& p) {
    cout << "姓名：" << p.m_Name << "，年龄：" << p.m_Age << endl;
}

int main() {
    Person<string, int> p("张三", 20);
    printPerson(p);     // 姓名：张三，年龄：20

    Person<string, string> p2("李四", "二十五");
    printPerson(p2);    // 姓名：李四，年龄：二十五

    return 0;
}
```

## 九、综合示例：通用数组类模板

```cpp
#include <iostream>
#include <string>
using namespace std;

// 通用数组类模板
template <class T>
class MyArray {
private:
    T* m_Data;      // 指向堆区数组的指针
    int m_Capacity; // 数组容量
    int m_Size;     // 当前元素个数

public:
    // 构造函数
    MyArray(int capacity) : m_Capacity(capacity), m_Size(0) {
        m_Data = new T[m_Capacity];
    }

    // 拷贝构造（深拷贝）
    MyArray(const MyArray& other) : m_Capacity(other.m_Capacity), m_Size(other.m_Size) {
        m_Data = new T[m_Capacity];
        for (int i = 0; i < m_Size; i++) {
            m_Data[i] = other.m_Data[i];
        }
    }

    // 赋值运算符重载（深拷贝）
    MyArray& operator=(const MyArray& other) {
        if (this != &other) {
            delete[] m_Data;
            m_Capacity = other.m_Capacity;
            m_Size = other.m_Size;
            m_Data = new T[m_Capacity];
            for (int i = 0; i < m_Size; i++) {
                m_Data[i] = other.m_Data[i];
            }
        }
        return *this;
    }

    // 析构函数
    ~MyArray() {
        delete[] m_Data;
        m_Data = nullptr;
    }

    // 尾部添加元素
    void pushBack(const T& value) {
        if (m_Size >= m_Capacity) {
            cout << "数组已满！" << endl;
            return;
        }
        m_Data[m_Size++] = value;
    }

    // 尾部删除元素
    void popBack() {
        if (m_Size <= 0) {
            cout << "数组为空！" << endl;
            return;
        }
        m_Size--;
    }

    // 通过下标访问元素
    T& operator[](int index) {
        return m_Data[index];
    }

    // 获取当前大小
    int getSize() const {
        return m_Size;
    }

    // 获取容量
    int getCapacity() const {
        return m_Capacity;
    }
};

int main() {
    // 使用 int 类型的数组
    MyArray<int> arr(5);
    arr.pushBack(10);
    arr.pushBack(20);
    arr.pushBack(30);

    cout << "int 数组：" << endl;
    for (int i = 0; i < arr.getSize(); i++) {
        cout << arr[i] << " ";
    }
    cout << endl;   // 输出：10 20 30

    // 使用 string 类型的数组
    MyArray<string> strArr(3);
    strArr.pushBack("Hello");
    strArr.pushBack("World");

    cout << "string 数组：" << endl;
    for (int i = 0; i < strArr.getSize(); i++) {
        cout << strArr[i] << " ";
    }
    cout << endl;   // 输出：Hello World

    return 0;
}
```

## 十、总结

| 知识点 | 要点 |
|--------|------|
| 语法 | `template<class T1, class T2> class 类名` |
| 显式指定类型 | 类模板使用时必须 `<类型>` |
| 默认类型参数 | `template<class T = int>` 支持默认类型 |
| 成员函数创建时机 | 调用时才创建，不是实例化时 |
| 函数参数三种方式 | 指定类型、参数模板化、整个类模板化 |
| 与继承 | 子类必须指明父类的 T 类型，或子类也是模板 |
| 分文件问题 | 声明和实现要放在同一个 `.hpp` 文件中 |
| 友元 | 类内实现简单，类外需要前置声明 |
