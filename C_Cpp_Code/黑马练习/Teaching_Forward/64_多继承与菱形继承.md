# 多继承与菱形继承

## 一、多继承语法

C++ 允许一个子类同时继承多个父类，这就是多继承。

```cpp
class 子类名 : 继承方式1 父类1, 继承方式2 父类2 {
    // 子类新增的成员
};
```

### 1.1 多继承基本示例

```cpp
#include <iostream>
#include <string>
using namespace std;

// 父类1：知识
class Knowledge {
public:
    string m_Language;      // 会的语言

    Knowledge() : m_Language("C++") {}

    void study() {
        cout << "学习知识，会的编程语言：" << m_Language << endl;
    }
};

// 父类2：体能
class Physical {
public:
    int m_Strength;     // 力量值

    Physical() : m_Strength(80) {}

    void exercise() {
        cout << "体能训练，力量值：" << m_Strength << endl;
    }
};

// 子类同时继承两个父类
class Student : public Knowledge, public Physical {
public:
    string m_Name;

    Student(string name) : m_Name(name) {}

    void show() {
        cout << "姓名:" << m_Name << endl;
        cout << "编程语言:" << m_Language << endl;   // 来自 Knowledge
        cout << "力量值:" << m_Strength << endl;      // 来自 Physical
    }
};

int main() {
    Student s("张三");
    s.show();
    cout << endl;
    s.study();       // 来自 Knowledge
    s.exercise();    // 来自 Physical

    return 0;
}
```

输出：
```
姓名:张三
编程语言:C++
力量值:80

学习知识，会的编程语言：C++
体能训练，力量值：80
```

### 1.2 多继承的构造顺序

```cpp
#include <iostream>
using namespace std;

class A {
public:
    A() { cout << "A 构造" << endl; }
    ~A() { cout << "A 析构" << endl; }
};

class B {
public:
    B() { cout << "B 构造" << endl; }
    ~B() { cout << "B 析构" << endl; }
};

class C : public A, public B {
public:
    // 构造顺序：按继承列表中的声明顺序（A先，B后）
    C() { cout << "C 构造" << endl; }
    ~C() { cout << "C 析构" << endl; }
};

int main() {
    C c;
    return 0;
}
```

输出：
```
A 构造
B 构造
C 构造
C 析构
B 析构
A 析构
```

**构造顺序：** 按照继承列表中的声明顺序（`A` 先于 `B`）。
**析构顺序：** 与构造顺序相反。

## 二、多继承的同名二义性

当多个父类有同名成员时，子类直接访问会产生二义性，必须使用作用域分辨符。

### 2.1 同名成员变量的二义性

```cpp
#include <iostream>
using namespace std;

class Base1 {
public:
    int m_A = 10;
};

class Base2 {
public:
    int m_A = 20;
};

class Son : public Base1, public Base2 {
public:
    int m_C = 30;
};

int main() {
    Son s;

    // 错误：二义性，编译器不知道访问哪个 m_A
    // cout << s.m_A << endl;

    // 正确：必须用作用域分辨符指明来自哪个父类
    cout << "Base1 的 m_A = " << s.Base1::m_A << endl;   // 10
    cout << "Base2 的 m_A = " << s.Base2::m_A << endl;   // 20
    cout << "自己的 m_C = " << s.m_C << endl;             // 30

    return 0;
}
```

### 2.2 同名成员函数的二义性

```cpp
#include <iostream>
using namespace std;

class Base1 {
public:
    void func() {
        cout << "Base1 的 func" << endl;
    }
};

class Base2 {
public:
    void func() {
        cout << "Base2 的 func" << endl;
    }
};

class Son : public Base1, public Base2 {
public:
    // 子类也可以定义自己的同名函数
    void func() {
        cout << "Son 的 func" << endl;
    }
};

int main() {
    Son s;

    // 调用子类自己的
    s.func();               // Son 的 func

    // 调用父类的必须指明作用域
    s.Base1::func();        // Base1 的 func
    s.Base2::func();        // Base2 的 func

    return 0;
}
```

## 三、菱形继承

### 3.1 什么是菱形继承

菱形继承（Diamond Problem）是多继承带来的经典问题：

```
        Animal          （最顶层基类）
       /      \
    Sheep    Tuo        （两个中间子类）
       \      /
       SheepTuo         （最底层子类）
```

`Sheep` 和 `Tuo` 都继承自 `Animal`，而 `SheepTuo` 同时继承 `Sheep` 和 `Tuo`。
这就形成了菱形（钻石形）的继承结构。

### 3.2 菱形继承的问题

```cpp
#include <iostream>
#include <string>
using namespace std;

// 最顶层基类
class Animal {
public:
    int m_Age;          // 年龄
    string m_Name;      // 名字

    void eat() {
        cout << m_Name << " 正在吃东西，年龄：" << m_Age << endl;
    }
};

// 继承 Animal
class Sheep : public Animal {
public:
    void graze() {
        cout << m_Name << " 在吃草" << endl;
    }
};

// 继承 Animal
class Tuo : public Animal {
public:
    void carry() {
        cout << m_Name << " 在驼东西" << endl;
    }
};

// 同时继承 Sheep 和 Tuo
class SheepTuo : public Sheep, public Tuo {
public:
    void show() {
        // 问题：SheepTuo 中有两份 Animal 的成员！
        // 编译器不知道该用哪一份
        // cout << m_Name << endl;   // 错误：二义性！
        // cout << m_Age << endl;    // 错误：二义性！
    }
};

int main() {
    SheepTuo st;

    // 必须指明作用域，但这样会有两份数据
    st.Sheep::m_Name = "羊驼";
    st.Sheep::m_Age = 3;
    st.Tuo::m_Name = "驼羊";     // 这是另一份！
    st.Tuo::m_Age = 5;           // 这是另一份！

    cout << "Sheep 中的：" << st.Sheep::m_Name << ", " << st.Sheep::m_Age << endl;
    cout << "Tuo 中的：" << st.Tuo::m_Name << ", " << st.Tuo::m_Age << endl;

    return 0;
}
```

输出：
```
Sheep 中的：羊驼, 3
Tuo 中的：驼羊, 5
```

**问题分析：**
1. **数据冗余：** `SheepTuo` 中有两份 `Animal` 的成员，浪费内存
2. **二义性：** 访问 `m_Name` 时不知道用哪份

## 四、虚继承解决菱形继承

### 4.1 虚继承语法

使用关键字 `virtual` 进行虚继承，可以保证最底层子类只保留一份基类成员。

```cpp
#include <iostream>
#include <string>
using namespace std;

// 最顶层基类（虚基类）
class Animal {
public:
    int m_Age;
    string m_Name;

    void eat() {
        cout << m_Name << " 正在吃东西，年龄：" << m_Age << endl;
    }
};

// 使用 virtual 虚继承
class Sheep : virtual public Animal {
public:
    void graze() {
        cout << m_Name << " 在吃草" << endl;
    }
};

// 使用 virtual 虚继承
class Tuo : virtual public Animal {
public:
    void carry() {
        cout << m_Name << " 在驼东西" << endl;
    }
};

class SheepTuo : public Sheep, public Tuo {
public:
    void show() {
        // 现在只有一份 Animal 的成员，不再有二义性
        cout << m_Name << "，年龄：" << m_Age << endl;
    }
};

int main() {
    SheepTuo st;

    // 不需要作用域分辨符了，只有一份数据
    st.m_Name = "羊驼";
    st.m_Age = 3;

    st.show();           // 羊驼，年龄：3
    st.eat();            // 羊驼 正在吃东西，年龄：3
    st.graze();          // 羊驼 在吃草
    st.carry();          // 羊驼 在驼东西

    cout << "sizeof(SheepTuo) = " << sizeof(SheepTuo) << endl;

    return 0;
}
```

### 4.2 虚继承的原理：vbptr（虚基类指针）

```
普通继承（两份 Animal 数据）：
┌─────────────┐     ┌─────────────┐
│  Sheep 对象  │     │   Tuo 对象   │
│ ┌─────────┐ │     │ ┌─────────┐ │
│ │ Animal  │ │     │ │ Animal  │ │    <-- 两份 Animal
│ │ m_Age   │ │     │ │ m_Age   │ │
│ │ m_Name  │ │     │ │ m_Name  │ │
│ └─────────┘ │     │ └─────────┘ │
└─────────────┘     └─────────────┘

虚继承（一份 Animal 数据）：
┌───────────────┐
│  SheepTuo 对象  │
│ ┌────────────┐ │
│ │ vbptr(Sheep)│─┼──┐
│ └────────────┘ │  │
│ ┌────────────┐ │  │   ┌──────────────┐
│ │ vbptr(Tuo) │─┼──┼──>│ Animal 共享   │
│ └────────────┘ │  │   │ m_Age        │
└───────────────┘  │   │ m_Name       │
                   └──>│ (只有一份)     │
                       └──────────────┘
```

**vbptr（virtual base pointer）虚基类指针：**
- 每个虚继承的子类都会生成一个 vbptr
- vbptr 指向一个虚基类表（vbtable）
- 表中记录了虚基类成员相对于当前位置的偏移量
- 通过偏移量找到共享的虚基类成员

### 4.3 构造函数调用问题

虚继承中，最底层子类需要直接调用虚基类的构造函数。

```cpp
#include <iostream>
using namespace std;

class Animal {
public:
    int m_Age;

    Animal(int age) : m_Age(age) {
        cout << "Animal 构造，age = " << m_Age << endl;
    }

    ~Animal() {
        cout << "Animal 析构" << endl;
    }
};

class Sheep : virtual public Animal {
public:
    Sheep(int age) : Animal(age) {
        cout << "Sheep 构造" << endl;
    }
};

class Tuo : virtual public Animal {
public:
    Tuo(int age) : Animal(age) {
        cout << "Tuo 构造" << endl;
    }
};

class SheepTuo : public Sheep, public Tuo {
public:
    // 虚继承时，最底层子类必须直接调用虚基类的构造函数
    // 中间类的构造函数调用会被忽略
    SheepTuo(int age) : Animal(age), Sheep(age), Tuo(age) {
        cout << "SheepTuo 构造" << endl;
    }
};

int main() {
    SheepTuo st(3);

    cout << "m_Age = " << st.m_Age << endl;

    return 0;
}
```

输出：
```
Animal 构造，age = 3
Sheep 构造
Tuo 构造
SheepTuo 构造
m_Age = 3
Animal 析构
```

注意：Animal 只被构造了一次。虚继承中，虚基类的构造由最底层子类负责。

## 五、虚继承与非虚继承的对比

```cpp
#include <iostream>
using namespace std;

// ========== 不用虚继承 ==========
class Base1 {
public:
    int m_Data = 10;
};

class Left1 : public Base1 {};
class Right1 : public Base1 {};
class Diamond1 : public Left1, public Right1 {};

// ========== 使用虚继承 ==========
class Base2 {
public:
    int m_Data = 20;
};

class Left2 : virtual public Base2 {};
class Right2 : virtual public Base2 {};
class Diamond2 : public Left2, public Right2 {};

int main() {
    // 不用虚继承：有两份 m_Data
    Diamond1 d1;
    cout << "非虚继承 sizeof(Diamond1) = " << sizeof(Diamond1) << endl;
    // 左边继承的 m_Data 和右边继承的 m_Data 各一份
    // sizeof 可能是 8 (两个int) + vbptr开销

    // 用虚继承：只有一份 m_Data
    Diamond2 d2;
    cout << "虚继承 sizeof(Diamond2) = " << sizeof(Diamond2) << endl;
    // 只有一份 m_Data，但有 vbptr 的额外开销

    // 非虚继承需要作用域分辨
    d1.Left1::m_Data = 100;
    d1.Right1::m_Data = 200;
    cout << "d1.Left1::m_Data = " << d1.Left1::m_Data << endl;    // 100
    cout << "d1.Right1::m_Data = " << d1.Right1::m_Data << endl;  // 200

    // 虚继承不需要作用域，只有一份
    d2.m_Data = 300;
    cout << "d2.m_Data = " << d2.m_Data << endl;   // 300

    return 0;
}
```

## 六、综合练习

```cpp
#include <iostream>
#include <string>
using namespace std;

// 菱形继承的实际例子：家具设计

// 基类：家具
class Furniture {
public:
    string m_Material;      // 材质
    double m_Price;         // 价格

    Furniture(string material, double price)
        : m_Material(material), m_Price(price) {}

    void showBase() {
        cout << "材质：" << m_Material << "，价格：" << m_Price << "元" << endl;
    }
};

// 虚继承：桌子
class Desk : virtual public Furniture {
public:
    double m_Length;    // 长
    double m_Width;     // 宽

    Desk(string material, double price, double len, double wid)
        : Furniture(material, price), m_Length(len), m_Width(wid) {}

    void showDesk() {
        cout << "桌子 - ";
        showBase();
        cout << "尺寸：" << m_Length << " x " << m_Width << endl;
    }
};

// 虚继承：椅子
class Chair : virtual public Furniture {
public:
    bool m_HasArmrest;  // 是否有扶手

    Chair(string material, double price, bool armrest)
        : Furniture(material, price), m_HasArmrest(armrest) {}

    void showChair() {
        cout << "椅子 - ";
        showBase();
        cout << "有扶手：" << (m_HasArmrest ? "是" : "否") << endl;
    }
};

// 桌椅套装
class DeskChairSet : public Desk, public Chair {
public:
    string m_Style;     // 风格

    DeskChairSet(string material, double price, double len, double wid, bool armrest, string style)
        : Furniture(material, price),          // 直接调用虚基类构造
          Desk(material, price, len, wid),
          Chair(material, price, armrest),
          m_Style(style) {}

    void show() {
        cout << "===== 桌椅套装 =====" << endl;
        cout << "风格：" << m_Style << endl;
        showBase();             // 只有一份，无二义性
        cout << "桌子尺寸：" << m_Length << " x " << m_Width << endl;
        cout << "椅子扶手：" << (m_HasArmrest ? "有" : "无") << endl;
    }
};

int main() {
    DeskChairSet set("实木", 3999, 120, 60, true, "北欧简约");
    set.show();

    return 0;
}
```

## 七、总结

| 知识点 | 要点 |
|--------|------|
| 多继承语法 | `class C : public A, public B` |
| 同名二义性 | 必须使用 `类名::` 指明来自哪个父类 |
| 菱形继承问题 | 数据冗余 + 二义性 |
| 虚继承语法 | `virtual public` 关键字 |
| 虚继承原理 | vbptr 虚基类指针，指向虚基类表 |
| 虚继承效果 | 只保留一份基类成员 |
| 虚继承构造 | 最底层子类直接调用虚基类构造函数 |
| 建议 | 尽量避免菱形继承，如必须使用则用虚继承 |
