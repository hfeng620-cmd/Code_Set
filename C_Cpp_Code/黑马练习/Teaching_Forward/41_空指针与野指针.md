# 空指针与野指针详解

## 一、为什么要学习空指针和野指针

指针是C++中最强大的工具之一，但也是最危险的。如果指针使用不当，会导致：
- 程序崩溃（最常见的后果）
- 数据被意外修改
- 安全漏洞（被黑客利用）

**空指针**和**野指针**是两种最常见的指针问题，理解它们能帮助你避免大部分指针相关的bug。

---

## 二、空指针（Null Pointer）

### 2.1 什么是空指针

空指针就是**不指向任何有效内存地址的指针**。它是一个"明确表示什么都没指向"的状态。

```cpp
#include <iostream>
using namespace std;

int main() {
    // 空指针的定义方式
    int *p1 = nullptr;   // C++11 推荐写法
    int *p2 = NULL;      // C语言风格，不推荐
    int *p3 = 0;         // 也能编译通过，但不推荐

    // 三种写法效果相同：指针不指向任何有效地址
    // 推荐使用 nullptr，它是C++11引入的关键字，类型安全

    return 0;
}
```

### 2.2 nullptr vs NULL 的区别

```cpp
#include <iostream>
using namespace std;

void func(int x) {
    cout << "调用了 func(int)" << endl;
}

void func(int *p) {
    cout << "调用了 func(int*)" << endl;
}

int main() {
    // NULL 的问题：它本质上是整数 0
    func(NULL);      // 调用 func(int)！不是 func(int*)！
    func(nullptr);   // 调用 func(int*) -- 正确！

    // 这就是推荐使用 nullptr 的原因
    // nullptr 是真正的空指针常量，类型是 std::nullptr_t
    // NULL 可能被定义为 0，在重载时会产生歧义

    return 0;
}
```

### 2.3 为什么不能解引用空指针

```cpp
#include <iostream>
using namespace std;

int main() {
    int *p = nullptr;  // p 不指向任何有效地址

    // *p = 10;         // 严重错误！解引用空指针会导致程序崩溃！

    // 为什么？因为地址 0（nullptr）是操作系统占用的内存区域
    // 0 ~ 255 这段地址范围是系统保留的，程序不能访问
    // 尝试访问这些地址，操作系统会直接终止你的程序

    // 正确做法：使用前先检查指针是否为空
    if (p != nullptr) {
        *p = 10;  // 只有 p 不为空时才解引用
    } else {
        cout << "指针为空，不能解引用！" << endl;
    }

    return 0;
}
```

### 2.4 空指针的实际应用场景

```cpp
#include <iostream>
using namespace std;

// 场景1：函数参数的默认值
void processData(int *data = nullptr) {
    if (data == nullptr) {
        cout << "没有传入数据，使用默认行为" << endl;
        return;
    }
    cout << "处理数据：" << *data << endl;
}

// 场景2：函数返回值表示"没有找到"
int *findValue(int arr[], int size, int target) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == target) {
            return &arr[i];  // 找到了，返回地址
        }
    }
    return nullptr;  // 没找到，返回空指针
}

int main() {
    // 场景1测试
    processData();       // 不传参，使用默认行为
    int x = 42;
    processData(&x);     // 传入数据

    // 场景2测试
    int arr[] = {10, 20, 30, 40, 50};
    int *result = findValue(arr, 5, 30);
    if (result != nullptr) {
        cout << "找到了，值为：" << *result << endl;
    } else {
        cout << "没有找到" << endl;
    }

    result = findValue(arr, 5, 99);
    if (result != nullptr) {
        cout << "找到了，值为：" << *result << endl;
    } else {
        cout << "没有找到" << endl;
    }

    return 0;
}
```

## 三、野指针（Wild Pointer）

### 3.1 什么是野指针

野指针是指**指向不确定内存位置的指针**。它指向的地址是随机的、不可预测的，使用它会导致未定义行为。

```cpp
#include <iostream>
using namespace std;

int main() {
    // 情况1：未初始化的指针
    int *p1;           // p1 的值是随机的（垃圾值）
    // *p1 = 10;       // 严重错误！p1指向随机地址

    // 情况2：指针指向的变量已经被销毁
    int *p2;
    {
        int a = 10;
        p2 = &a;       // p2 指向 a
    }   // a 在这里被销毁了
    // *p2 = 20;       // 错误！a已经不存在了，p2是野指针

    // 情况3：指针越界
    int arr[3] = {1, 2, 3};
    int *p3 = arr;
    // p3 += 5;        // 越界！p3指向了数组之外的内存
    // *p3 = 100;      // 错误！

    return 0;
}
```

### 3.2 野指针的三种成因

#### 成因1：未初始化

```cpp
#include <iostream>
using namespace std;

int main() {
    // 错误示范
    int *p;         // 没有初始化，p的值是随机的
    // cout << *p;  // 未定义行为！

    // 正确做法：定义时就初始化
    int *p1 = nullptr;          // 初始化为空指针
    int a = 10;
    int *p2 = &a;               // 初始化为指向某个变量

    cout << "*p2 = " << *p2 << endl;  // 安全！p2指向有效的a

    return 0;
}
```

#### 成因2：指向已释放的内存

```cpp
#include <iostream>
using namespace std;

int main() {
    int *p;
    {
        int a = 10;
        p = &a;         // p 指向 a
        cout << "*p = " << *p << endl;  // 输出 10，此时是安全的
    }   // a 在这里离开作用域，内存被回收

    // 此时 p 成为野指针！
    // 虽然 p 的值（地址）没变，但那块内存已经不属于我们了
    // cout << *p;       // 未定义行为！可能输出垃圾值，也可能崩溃

    // 解决方案：离开作用域前将指针置空
    int *p2;
    {
        int b = 20;
        p2 = &b;
        cout << "*p2 = " << *p2 << endl;
        p2 = nullptr;   // 离开前置空
    }
    // p2 已经是 nullptr，不会误用

    return 0;
}
```

#### 成因3：指针越界

```cpp
#include <iostream>
using namespace std;

int main() {
    int arr[3] = {10, 20, 30};
    int *p = arr;    // p 指向 arr[0]

    // 合法的访问
    cout << "arr[0] = " << p[0] << endl;  // 10
    cout << "arr[1] = " << p[1] << endl;  // 20
    cout << "arr[2] = " << p[2] << endl;  // 30

    // 越界访问！p[3] 超出了数组范围
    // cout << p[3];   // 未定义行为！可能输出垃圾值

    // 指针算术越界
    int *p2 = arr + 10;  // p2 指向数组后面的某个位置
    // *p2 = 100;         // 严重错误！

    return 0;
}
```

### 3.3 野指针的危害

```cpp
#include <iostream>
using namespace std;

int main() {
    // 野指针的危害示例（不要真的运行这段代码！）
    int *p;  // 未初始化，野指针

    // 危害1：程序崩溃
    // *p = 10;   // 访问随机地址，操作系统直接终止程序

    // 危害2：数据被意外修改
    // 如果 p 恰好指向了其他变量的地址，*p = 10 会修改那个变量的值
    // 这种bug极难排查！

    // 危害3：安全漏洞
    // 黑客可以利用野指针写入恶意数据

    // 总结：野指针是C++中最危险的bug之一

    return 0;
}
```

## 四、如何避免野指针

### 4.1 规则1：定义时就初始化

```cpp
#include <iostream>
using namespace std;

int main() {
    // 好习惯：定义指针时就初始化
    int *p = nullptr;    // 初始化为空指针

    int a = 10;
    p = &a;              // 有具体指向时再赋值

    if (p != nullptr) {  // 使用前检查
        cout << "*p = " << *p << endl;
    }

    return 0;
}
```

### 4.2 规则2：释放后置空

```cpp
#include <iostream>
using namespace std;

int main() {
    // 使用 new 分配的内存
    int *p = new int(100);
    cout << "*p = " << *p << endl;  // 100

    // 释放内存后，立即置空
    delete p;
    p = nullptr;   // 关键一步！防止野指针

    // 之后再使用 p 前检查
    if (p != nullptr) {
        cout << "*p = " << *p << endl;
    } else {
        cout << "指针已为空，不能使用" << endl;
    }

    // 重复 delete nullptr 是安全的（不会崩溃）
    delete p;  // p 已经是 nullptr，这行不会出错

    return 0;
}
```

### 4.3 规则3：不要返回局部变量的地址

```cpp
#include <iostream>
using namespace std;

// 错误示范：返回局部变量的地址
int *badFunc() {
    int a = 10;
    return &a;   // a 在函数结束后就销毁了！
    // 返回的地址指向已销毁的变量，调用者拿到的是野指针！
}

// 正确做法1：返回静态变量的地址
int *goodFunc1() {
    static int a = 10;  // static 变量不会随函数结束而销毁
    return &a;
}

// 正确做法2：用 new 在堆上分配
int *goodFunc2() {
    int *p = new int(10);  // 堆上的内存不会自动释放
    return p;
    // 注意：调用者需要负责 delete
}

int main() {
    // int *p1 = badFunc();   // 危险！p1是野指针
    // cout << *p1;           // 未定义行为

    int *p2 = goodFunc1();
    cout << "*p2 = " << *p2 << endl;  // 安全！输出 10

    int *p3 = goodFunc2();
    cout << "*p3 = " << *p3 << endl;  // 安全！输出 10
    delete p3;                          // 记得释放
    p3 = nullptr;

    return 0;
}
```

### 4.4 规则4：注意作用域

```cpp
#include <iostream>
using namespace std;

int main() {
    int *p = nullptr;

    {
        int a = 10;
        p = &a;
        cout << "内部：*p = " << *p << endl;  // 安全，a还活着
    }   // a 在这里被销毁

    // p 现在是野指针！
    // cout << *p;   // 危险！

    // 正确做法：在作用域结束前处理好指针
    int *p2 = nullptr;
    {
        int b = 20;
        p2 = &b;
        cout << "内部：*p2 = " << *p2 << endl;
        // 如果需要在外部使用值，先保存
        int savedValue = *p2;
        p2 = nullptr;  // 置空，避免野指针
    }
    cout << "保存的值：" << 20 << endl;  // 用保存的值

    return 0;
}
```

## 五、悬空指针 vs 野指针

很多人会混淆**悬空指针（Dangling Pointer）**和**野指针（Wild Pointer）**，它们有区别：

```
野指针（Wild Pointer）：
  - 从未被初始化的指针
  - 指向随机的、不确定的地址
  - 例子：int *p;  // p的值是垃圾值

悬空指针（Dangling Pointer）：
  - 曾经指向有效地址，但后来那块内存被释放了
  - 指针的值（地址）还在，但那块内存已经不属于你了
  - 例子：delete p; 之后 p 就变成了悬空指针
```

```cpp
#include <iostream>
using namespace std;

int main() {
    // ====== 野指针 ======
    int *wild;          // 未初始化，是野指针
    // *wild = 10;      // 危险！

    // ====== 悬空指针 ======
    int *dangling = new int(100);
    cout << "释放前：" << *dangling << endl;  // 100
    delete dangling;    // 释放内存
    // dangling 仍然存储着那个地址，但那块内存已经还给系统了
    // dangling 现在是悬空指针
    // cout << *dangling;  // 未定义行为！

    // 解决方案：释放后置空
    dangling = nullptr;   // 现在是空指针，不再是悬空指针

    // ====== 另一种悬空指针 ======
    int *p;
    {
        int x = 42;
        p = &x;         // p 指向 x
    }   // x 被销毁
    // p 现在是悬空指针（x 的地址还在 p 里，但 x 已经不存在了）
    p = nullptr;         // 置空

    return 0;
}
```

### 对比总结

| 特征 | 野指针（Wild） | 悬空指针（Dangling） |
|------|---------------|---------------------|
| 原因 | 从未初始化 | 曾经有效，后来被释放 |
| 地址值 | 随机垃圾值 | 曾经有效的地址 |
| 危险程度 | 非常危险 | 非常危险 |
| 解决方案 | 定义时初始化 | 释放后置空 |

## 六、完整的安全指针使用流程

```cpp
#include <iostream>
using namespace std;

int main() {
    // 第1步：定义时初始化
    int *p = nullptr;

    // 第2步：有需要时再指向有效地址
    int a = 10;
    p = &a;

    // 第3步：使用前检查是否为空
    if (p != nullptr) {
        cout << "*p = " << *p << endl;  // 安全使用
    }

    // 第4步：如果用 new 分配了内存，用完要 delete
    p = new int(200);
    cout << "*p = " << *p << endl;      // 200

    // 第5步：delete 后立即置空
    delete p;
    p = nullptr;

    // 第6步：空指针可以安全地重复 delete
    delete p;   // 不会崩溃，因为 p 是 nullptr

    cout << "程序正常结束" << endl;
    return 0;
}
```

## 七、常见错误总结

```cpp
#include <iostream>
using namespace std;

int main() {
    // ====== 错误1：未初始化 ======
    int *p1;
    // *p1 = 10;      // 错误！

    // ====== 错误2：解引用空指针 ======
    int *p2 = nullptr;
    // *p2 = 10;      // 错误！

    // ====== 错误3：delete 后继续使用 ======
    int *p3 = new int(10);
    delete p3;
    // *p3 = 20;      // 错误！悬空指针

    // ====== 错误4：返回局部变量地址 ======
    // 见上面 badFunc() 的例子

    // ====== 错误5：数组越界 ======
    int arr[3] = {1, 2, 3};
    int *p4 = arr;
    // p4[5] = 100;   // 错误！越界

    return 0;
}
```

## 八、练习题

1. 编写代码演示空指针的定义和检查。
2. 编写代码演示野指针的成因（未初始化）。
3. 编写代码演示悬空指针的成因（delete后未置空）。
4. 编写一个安全的指针使用流程：定义、初始化、使用、释放、置空。

```cpp
// 练习4 参考答案：
#include <iostream>
using namespace std;

int main() {
    // 1. 定义并初始化
    int *p = nullptr;
    cout << "初始状态：p = " << p << endl;

    // 2. 指向有效地址
    p = new int(42);
    cout << "分配内存后：*p = " << *p << endl;

    // 3. 安全使用
    if (p != nullptr) {
        *p = 100;
        cout << "修改后：*p = " << *p << endl;
    }

    // 4. 释放内存
    delete p;

    // 5. 置空指针
    p = nullptr;

    // 6. 验证已安全
    if (p == nullptr) {
        cout << "指针已安全置空" << endl;
    }

    return 0;
}
```
