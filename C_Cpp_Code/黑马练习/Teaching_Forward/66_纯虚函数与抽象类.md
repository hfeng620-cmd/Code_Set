# 66 - 纯虚函数与抽象类

## 一、纯虚函数的语法

```cpp
virtual 返回值类型 函数名(参数列表) = 0;
```

**注意：**
- `= 0` 不是把函数赋值为 0，而是告诉编译器"这是一个纯虚函数，没有函数体"
- 纯虚函数**没有函数体**（在声明的类中不需要实现）
- `= 0` 只能写在类内部的声明中

```cpp
// 示例
class Animal {
public:
    // 纯虚函数：只有声明，没有实现（函数体）
    // = 0 表示这是一个纯虚函数
    virtual void speak() = 0;

    // 普通虚函数：有默认实现
    virtual void eat() {
        cout << "动物在吃东西" << endl;
    }
};
```

---

## 二、什么是抽象类？

> **抽象类：包含至少一个纯虚函数的类。**

```cpp
class Animal {
public:
    // 包含纯虚函数 → Animal 是抽象类
    virtual void speak() = 0;
};

class Shape {
public:
    // 包含纯虚函数 → Shape 是抽象类
    virtual double area() = 0;
    virtual double perimeter() = 0;
};
```

## 三、抽象类的特点

### 特点1：不能实例化对象

```cpp
#include <iostream>
using namespace std;

class Animal {
public:
    virtual void speak() = 0;   // 纯虚函数
};

int main() {
    // Animal a;              // 错误！抽象类不能创建对象
    // Animal *p = new Animal();  // 错误！也不能用 new 创建

    // 原因：抽象类不完整，speak() 没有实现，无法创建对象
    return 0;
}
```

### 特点2：子类必须重写所有纯虚函数，否则子类也是抽象类

```cpp
#include <iostream>
using namespace std;

class Animal {
public:
    virtual void speak() = 0;   // 纯虚函数1
    virtual void eat() = 0;     // 纯虚函数2
};

// 情况1：子类重写了所有纯虚函数 → 子类不是抽象类，可以创建对象
class Cat : public Animal {
public:
    void speak() override {       // 重写纯虚函数1
        cout << "猫说：喵喵喵" << endl;
    }

    void eat() override {         // 重写纯虚函数2
        cout << "猫在吃鱼" << endl;
    }
};

// 情况2：子类只重写了部分纯虚函数 → 子类仍然是抽象类，不能创建对象
class Dog : public Animal {
public:
    void speak() override {       // 只重写了 speak
        cout << "狗说：汪汪汪" << endl;
    }
    // 没有重写 eat() → Dog 仍然是抽象类！
    // Dog d;  // 错误！Dog 也是抽象类，不能创建对象
};

// 情况3：孙子类重写剩余的纯虚函数 → 可以创建对象
class Husky : public Dog {
public:
    void eat() override {         // 重写了 eat
        cout << "哈士奇在吃骨头" << endl;
    }
    // 现在所有纯虚函数都被重写了，Husky 不是抽象类
};

int main() {
    // Animal a;    // 错误！
    // Dog d;       // 错误！Dog 没有重写 eat()，也是抽象类
    Cat cat;        // 正确！Cat 重写了所有纯虚函数
    Husky husky;    // 正确！Husky 重写了所有纯虚函数

    cat.speak();    // 输出：猫说：喵喵喵
    husky.speak();  // 输出：狗说：汪汪汪
    husky.eat();    // 输出：哈士奇在吃骨头

    return 0;
}
```

### 特点3：可以创建抽象类的指针和引用

```cpp
#include <iostream>
using namespace std;

class Animal {
public:
    virtual void speak() = 0;
};

class Cat : public Animal {
public:
    void speak() override {
        cout << "猫说：喵喵喵" << endl;
    }
};

class Dog : public Animal {
public:
    void speak() override {
        cout << "狗说：汪汪汪" << endl;
    }
};

int main() {
    // Animal a;                  // 错误！不能创建抽象类的对象

    Animal *p = new Cat();        // 正确！可以创建抽象类的指针，指向子类对象
    p->speak();                   // 输出：猫说：喵喵喵
    delete p;

    Cat cat;
    Animal &ref = cat;            // 正确！可以创建抽象类的引用
    ref.speak();                  // 输出：猫说：喵喵喵

    // 通过抽象类指针数组管理不同的子类对象
    Animal *animals[] = {
        new Cat(),
        new Dog(),
        new Cat()
    };

    for (int i = 0; i < 3; i++) {
        animals[i]->speak();      // 多态调用
        delete animals[i];        // 释放内存
    }

    return 0;
}
```

## 四、纯虚函数 vs 虚函数 的区别

```cpp
class Animal {
public:
    // 普通虚函数：有默认实现
    // 子类可以选择性重写（不重写就用父类的默认实现）
    virtual void eat() {
        cout << "动物在吃东西" << endl;
    }

    // 纯虚函数：没有默认实现
    // 子类必须重写，否则子类也是抽象类
    virtual void speak() = 0;
};

class Cat : public Animal {
public:
    // 可以不重写 eat()——将使用父类的默认实现
    // 但必须重写 speak()——因为它是纯虚函数
    void speak() override {
        cout << "猫说：喵喵喵" << endl;
    }
};
```

| 对比项 | 普通虚函数 | 纯虚函数 |
|--------|-----------|----------|
| **语法** | `virtual void f() { ... }` | `virtual void f() = 0;` |
| **函数体** | 有默认实现 | 没有实现 |
| **子类是否必须重写** | 否（可以选择性重写） | 是（必须重写） |
| **所在类** | 普通类或抽象类 | 只能在抽象类中 |
| **用途** | 提供默认行为 | 定义接口规范 |

## 五、纯虚函数可以有函数体（进阶了解）

这是一个很少见但合法的特性：

```cpp
class Animal {
public:
    // 纯虚函数（有 = 0），但仍然可以提供函数体
    virtual void speak() = 0;
};

// 纯虚函数的函数体可以在类外提供
void Animal::speak() {
    cout << "Animal 的默认 speak" << endl;
}

class Cat : public Animal {
public:
    void speak() override {
        // 可以在子类中调用父类纯虚函数的默认实现
        Animal::speak();
        cout << "猫说：喵喵喵" << endl;
    }
};
```

> **注意：** 虽然语法上允许，但实际开发中很少这样做。大多数情况下，纯虚函数就是不提供实现的。

## 六、实际应用案例

### 案例1：图形面积计算（Shape 抽象类）

```cpp
#include <iostream>
#include <cmath>
using namespace std;

// ==================== 抽象基类：Shape ====================
class Shape {
public:
    // 纯虚函数：计算面积
    // 不同图形的面积公式不同，所以基类无法给出默认实现
    // 让子类各自实现 → 这就是多态的应用
    virtual double area() = 0;

    // 纯虚函数：计算周长
    virtual double perimeter() = 0;

    // 虚析构函数（后面章节详细讲）
    virtual ~Shape() {}
};

// ==================== 子类：圆形 ====================
class Circle : public Shape {
private:
    double radius;   // 半径

public:
    Circle(double r) : radius(r) {}

    // 重写 area()
    double area() override {
        return 3.14159265 * radius * radius;
    }

    // 重写 perimeter()
    double perimeter() override {
        return 2 * 3.14159265 * radius;
    }
};

// ==================== 子类：矩形 ====================
class Rectangle : public Shape {
private:
    double width;    // 宽
    double height;   // 高

public:
    Rectangle(double w, double h) : width(w), height(h) {}

    double area() override {
        return width * height;
    }

    double perimeter() override {
        return 2 * (width + height);
    }
};

// ==================== 子类：三角形 ====================
class Triangle : public Shape {
private:
    double a, b, c;  // 三条边

public:
    Triangle(double a, double b, double c) : a(a), b(b), c(c) {}

    double area() override {
        // 海伦公式
        double s = (a + b + c) / 2.0;
        return sqrt(s * (s - a) * (s - b) * (s - c));
    }

    double perimeter() override {
        return a + b + c;
    }
};

// ==================== 统一接口函数 ====================
// 通过抽象类指针，统一处理所有图形
void printShapeInfo(Shape *shape) {
    cout << "面积: " << shape->area() << endl;
    cout << "周长: " << shape->perimeter() << endl;
}

int main() {
    Circle circle(5.0);
    Rectangle rect(4.0, 6.0);
    Triangle tri(3.0, 4.0, 5.0);

    cout << "=== 圆形 ===" << endl;
    printShapeInfo(&circle);

    cout << "\n=== 矩形 ===" << endl;
    printShapeInfo(&rect);

    cout << "\n=== 三角形 ===" << endl;
    printShapeInfo(&tri);

    return 0;
}
```

### 案例2：员工薪资系统

```cpp
#include <iostream>
#include <string>
using namespace std;

// ==================== 抽象基类：Employee ====================
class Employee {
protected:
    string name;
    int id;

public:
    Employee(string name, int id) : name(name), id(id) {}

    // 纯虚函数：计算薪资
    // 不同类型员工的薪资计算方式不同
    virtual double calculateSalary() = 0;

    // 纯虚函数：显示信息
    virtual void display() = 0;

    // 虚析构函数
    virtual ~Employee() {}
};

// ==================== 子类：全职员工 ====================
class FullTimeEmployee : public Employee {
private:
    double monthlySalary;   // 月薪

public:
    FullTimeEmployee(string name, int id, double salary)
        : Employee(name, id), monthlySalary(salary) {}

    double calculateSalary() override {
        return monthlySalary;   // 全职员工固定月薪
    }

    void display() override {
        cout << "[全职] " << name << " (ID:" << id << ")"
             << " 月薪: " << calculateSalary() << " 元" << endl;
    }
};

// ==================== 子类：兼职员工 ====================
class PartTimeEmployee : public Employee {
private:
    double hourlyRate;      // 时薪
    int hoursPerMonth;      // 每月工时

public:
    PartTimeEmployee(string name, int id, double rate, int hours)
        : Employee(name, id), hourlyRate(rate), hoursPerMonth(hours) {}

    double calculateSalary() override {
        return hourlyRate * hoursPerMonth;   // 兼职按小时计算
    }

    void display() override {
        cout << "[兼职] " << name << " (ID:" << id << ")"
             << " 时薪: " << hourlyRate << " 工时: " << hoursPerMonth
             << "h 月薪: " << calculateSalary() << " 元" << endl;
    }
};

// ==================== 统一处理函数 ====================
void printEmployee(Employee *emp) {
    emp->display();
}

int main() {
    // 创建不同类型的员工
    FullTimeEmployee ft("张三", 1001, 8000);
    PartTimeEmployee pt("李四", 1002, 50, 80);

    // 用抽象类指针统一管理
    Employee *employees[] = { &ft, &pt };

    cout << "=== 员工信息 ===" << endl;
    for (int i = 0; i < 2; i++) {
        printEmployee(employees[i]);   // 多态调用
    }

    // 计算总薪资
    double total = 0;
    for (int i = 0; i < 2; i++) {
        total += employees[i]->calculateSalary();
    }
    cout << "\n总薪资支出: " << total << " 元" << endl;

    return 0;
}
```

## 七、常见错误与注意事项

### 错误1：忘记实现纯虚函数

```cpp
class Animal {
public:
    virtual void speak() = 0;
    virtual void eat() = 0;
};

class Cat : public Animal {
public:
    void speak() override {
        cout << "喵" << endl;
    }
    // 忘记实现 eat() → Cat 也是抽象类！
    // Cat cat;  // 编译错误！
};
```

### 错误2：纯虚函数语法写错

```cpp
class Animal {
public:
    // 错误写法：
    // virtual void speak() = 0 { ... }     // = 0 后面不能有函数体
    // virtual void speak() = 0.0;          // 不是 0.0，是 = 0
    // virtual void speak() == 0;           // 不是 ==，是 = 0
    // void speak() = 0;                    // 缺少 virtual

    // 正确写法：
    virtual void speak() = 0;
};
```

### 错误3：在构造函数或析构函数中调用纯虚函数

```cpp
class Base {
public:
    virtual void func() = 0;

    Base() {
        // func();  // 危险！在构造函数中调用纯虚函数，行为未定义
    }

    virtual ~Base() {
        // func();  // 危险！在析构函数中调用纯虚函数，行为未定义
    }
};
```

### 注意事项总结

```
1. 纯虚函数语法：virtual 返回值类型 函数名(参数) = 0;
2. 包含纯虚函数的类是抽象类，不能创建对象
3. 子类必须重写所有纯虚函数，否则子类也是抽象类
4. 抽象类可以创建指针和引用（这是它的主要用途）
5. 抽象类的意义：定义接口规范，强制子类实现特定功能
6. 构造函数不能是虚函数，析构函数可以是虚函数
7. 纯虚函数 = 0 只能出现在类内部的声明中
```

## 八、小结

```
纯虚函数：
  - 语法：virtual 返回值类型 函数名(参数) = 0;
  - 没有函数体（不需要实现）
  - 子类必须重写

抽象类：
  - 包含纯虚函数的类
  - 不能实例化对象
  - 可以创建指针和引用
  - 子类必须重写所有纯虚函数才能创建对象

抽象类的用途：
  - 定义接口规范（规定子类必须实现哪些功能）
  - 实现多态（通过抽象类指针/引用统一管理不同子类）
```
