# 65 - 多态基础与虚函数

## 一、什么是多态？

**多态（Polymorphism）** 是面向对象编程的三大特性之一（封装、继承、多态）。

> **核心概念：同一接口，不同实现。**
> 就像"动物都会叫"，但猫叫"喵喵"，狗叫"汪汪"——调用的都是 `speak()`，但行为不同。

---

## 二、静态多态 vs 动态多态

C++ 中的多态分为两种：

| 类型 | 实现方式 | 确定时机 | 又称 |
|------|----------|----------|------|
| **静态多态** | 函数重载、运算符重载 | 编译时 | 早绑定（Early Binding） |
| **动态多态** | 虚函数 | 运行时 | 晚绑定（Late Binding） |

### 2.1 静态多态——函数重载

```cpp
#include <iostream>
using namespace std;

// 静态多态：函数重载
// 编译器在编译阶段就根据参数类型/个数确定调用哪个函数
class Print {
public:
    // 重载函数1：打印 int
    void show(int x) {
        cout << "整数: " << x << endl;
    }

    // 重载函数2：打印 double（参数类型不同，构成重载）
    void show(double x) {
        cout << "浮点数: " << x << endl;
    }

    // 重载函数3：打印两个 int（参数个数不同，构成重载）
    void show(int a, int b) {
        cout << "两个整数: " << a << ", " << b << endl;
    }
};

int main() {
    Print p;
    p.show(10);         // 编译时就确定调用 show(int)
    p.show(3.14);       // 编译时就确定调用 show(double)
    p.show(1, 2);       // 编译时就确定调用 show(int, int)
    return 0;
}
```

### 2.2 动态多态——虚函数（本节重点）

```cpp
#include <iostream>
using namespace std;

// 父类
class Animal {
public:
    // 加上 virtual 关键字，变成虚函数
    virtual void speak() {
        cout << "动物在叫" << endl;
    }
};

// 子类：猫
class Cat : public Animal {
public:
    // 重写父类的虚函数（函数名、参数、返回值都相同）
    void speak() {
        cout << "猫说：喵喵喵" << endl;
    }
};

// 子类：狗
class Dog : public Animal {
public:
    void speak() {
        cout << "狗说：汪汪汪" << endl;
    }
};

// 统一接口：传入父类引用，根据实际对象类型调用不同版本
void doSpeak(Animal &animal) {
    animal.speak();  // 运行时才确定调用哪个 speak
}

int main() {
    Cat cat;
    Dog dog;

    doSpeak(cat);   // 输出：猫说：喵喵喵
    doSpeak(dog);   // 输出：狗说：汪汪汪

    return 0;
}
```

## 三、动态多态的三个必要条件

要实现动态多态，**必须同时满足以下三个条件**：

```
条件1：有继承关系（子类继承父类）
条件2：子类重写父类的虚函数
条件3：通过父类的指针或引用来调用虚函数
```

### 详细说明

```cpp
#include <iostream>
using namespace std;

// ========== 条件1：有继承关系 ==========
class Animal {
public:
    // ========== 条件中的虚函数 ==========
    // 在父类函数声明前加 virtual，使其成为虚函数
    virtual void speak() {
        cout << "动物在叫" << endl;
    }
};

class Cat : public Animal {
public:
    // ========== 条件2：子类重写父类的虚函数 ==========
    // 重写要求：函数名、参数列表、返回值类型 都必须相同
    void speak() {       // 注意：子类的 virtual 可加可不加，因为继承后自动是虚函数
        cout << "猫说：喵喵喵" << endl;
    }
};

// ========== 条件3：父类指针或引用指向子类对象 ==========
int main() {
    Cat cat;

    // 方式一：父类指针指向子类对象
    Animal *p = &cat;
    p->speak();      // 输出：猫说：喵喵喵  （动态多态生效）

    // 方式二：父类引用指向子类对象
    Animal &ref = cat;
    ref.speak();     // 输出：猫说：喵喵喵  （动态多态生效）

    return 0;
}
```

## 四、virtual 关键字详解

### 4.1 基本用法

```cpp
class Animal {
public:
    // 在父类函数声明前加 virtual
    // 作用：告诉编译器，这个函数要在运行时根据实际对象类型来决定调用哪个版本
    virtual void speak() {
        cout << "动物在叫" << endl;
    }

    // 普通函数（非虚函数）——不参与多态
    void eat() {
        cout << "动物在吃东西" << endl;
    }
};

class Cat : public Animal {
public:
    void speak() {    // 重写了虚函数 → 多态生效
        cout << "猫说：喵喵喵" << endl;
    }

    void eat() {      // 隐藏了父类的 eat（不是重写，因为父类 eat 不是虚函数）
        cout << "猫在吃鱼" << endl;
    }
};

int main() {
    Cat cat;
    Animal *p = &cat;

    p->speak();   // 输出：猫说：喵喵喵   ← 虚函数，多态生效，调用子类版本
    p->eat();     // 输出：动物在吃东西   ← 非虚函数，没有多态，调用父类版本

    return 0;
}
```

### 4.2 子类中 virtual 可加可不加

```cpp
class Animal {
public:
    virtual void speak() { cout << "Animal" << endl; }
};

class Cat : public Animal {
public:
    // 以下两种写法等效，都构成重写：
    // 写法1：不加 virtual（继承后自动具有虚函数特性）
    void speak() { cout << "Cat" << endl; }

    // 写法2：显式加 virtual（可读性更好，推荐）
    // virtual void speak() { cout << "Cat" << endl; }
};
```

## 五、没有 virtual 时的行为 vs 加上 virtual 后的行为

### 5.1 没有 virtual——始终调用父类版本

```cpp
#include <iostream>
using namespace std;

class Animal {
public:
    // 注意：没有 virtual！
    void speak() {
        cout << "动物在叫" << endl;
    }
};

class Cat : public Animal {
public:
    void speak() {
        cout << "猫说：喵喵喵" << endl;
    }
};

class Dog : public Animal {
public:
    void speak() {
        cout << "狗说：汪汪汪" << endl;
    }
};

void doSpeak(Animal &animal) {
    animal.speak();  // 没有 virtual，编译时就确定调用 Animal::speak()
}

int main() {
    Cat cat;
    Dog dog;

    doSpeak(cat);   // 输出：动物在叫  ← 不是"喵喵喵"！
    doSpeak(dog);   // 输出：动物在叫  ← 不是"汪汪汪"！

    // 原因：没有 virtual 时，编译器根据引用/指针的类型（Animal）来决定调用哪个函数
    // 这叫做"静态绑定"（早绑定），在编译阶段就确定了

    return 0;
}
```

### 5.2 加上 virtual——根据实际对象类型调用

```cpp
#include <iostream>
using namespace std;

class Animal {
public:
    // 加上 virtual！
    virtual void speak() {
        cout << "动物在叫" << endl;
    }
};

class Cat : public Animal {
public:
    void speak() {
        cout << "猫说：喵喵喵" << endl;
    }
};

class Dog : public Animal {
public:
    void speak() {
        cout << "狗说：汪汪汪" << endl;
    }
};

void doSpeak(Animal &animal) {
    animal.speak();  // 有 virtual，运行时根据实际对象类型决定调用哪个 speak
}

int main() {
    Cat cat;
    Dog dog;

    doSpeak(cat);   // 输出：猫说：喵喵喵  ← 多态生效！
    doSpeak(dog);   // 输出：狗说：汪汪汪  ← 多态生效！

    // 原因：有 virtual 时，编译器在运行时根据对象的实际类型来决定调用哪个函数
    // 这叫做"动态绑定"（晚绑定）

    return 0;
}
```

## 六、重写 vs 重载 的区别

这两个概念名字很像，但完全不同，初学者一定要分清！

| 对比项 | 重写（Override） | 重载（Overload） |
|--------|------------------|------------------|
| **发生关系** | 子类与父类之间 | 同一个类中 |
| **函数名** | 必须相同 | 必须相同 |
| **参数列表** | 必须相同 | 必须不同（类型、个数、顺序） |
| **返回值** | 必须相同（或协变） | 可以不同 |
| **virtual** | 父类必须有 virtual | 与 virtual 无关 |
| **多态类型** | 动态多态（运行时） | 静态多态（编译时） |

```cpp
#include <iostream>
using namespace std;

// ==================== 重载示例 ====================
class Calculator {
public:
    // 重载：同一个类中，函数名相同，参数不同
    int add(int a, int b) {
        return a + b;
    }

    double add(double a, double b) {   // 参数类型不同
        return a + b;
    }

    int add(int a, int b, int c) {     // 参数个数不同
        return a + b + c;
    }
};

// ==================== 重写示例 ====================
class Animal {
public:
    virtual void speak() {             // 父类有 virtual
        cout << "动物在叫" << endl;
    }
};

class Cat : public Animal {
public:
    void speak() {                     // 重写：函数名、参数、返回值都相同
        cout << "猫说：喵喵喵" << endl;
    }
};

int main() {
    // 重载演示
    Calculator calc;
    cout << calc.add(1, 2) << endl;        // 调用 int 版本
    cout << calc.add(1.5, 2.5) << endl;    // 调用 double 版本
    cout << calc.add(1, 2, 3) << endl;     // 调用三参数版本

    // 重写演示
    Cat cat;
    Animal *p = &cat;
    p->speak();   // 输出：猫说：喵喵喵（多态生效）

    return 0;
}
```

## 七、完整综合示例

```cpp
#include <iostream>
#include <string>
using namespace std;

// ==================== 父类：Animal ====================
class Animal {
public:
    string name;

    Animal(string name) : name(name) {}

    // 虚函数：speak
    // 加 virtual 后，子类重写这个函数，通过父类指针/引用调用时会调用子类版本
    virtual void speak() {
        cout << name << "在叫" << endl;
    }

    // 虚函数：getType
    virtual string getType() {
        return "动物";
    }

    // 普通函数（非虚函数）——不参与多态
    void info() {
        cout << "这是" << getType() << "类的动物" << endl;
    }
};

// ==================== 子类：Cat ====================
class Cat : public Animal {
public:
    Cat(string name) : Animal(name) {}

    // 重写父类的虚函数 speak
    void speak() override {    // override 关键字（C++11）：显式表明这是重写，防止拼写错误
        cout << name << "说：喵喵喵" << endl;
    }

    // 重写父类的虚函数 getType
    string getType() override {
        return "猫";
    }
};

// ==================== 子类：Dog ====================
class Dog : public Animal {
public:
    Dog(string name) : Animal(name) {}

    void speak() override {
        cout << name << "说：汪汪汪" << endl;
    }

    string getType() override {
        return "狗";
    }
};

// ==================== 子类：Pig ====================
class Pig : public Animal {
public:
    Pig(string name) : Animal(name) {}

    void speak() override {
        cout << name << "说：哼哼哼" << endl;
    }

    string getType() override {
        return "猪";
    }
};

// ==================== 统一接口函数 ====================
// 参数是父类引用，传入不同子类对象，行为不同 → 这就是多态
void doSpeak(Animal &animal) {
    animal.speak();
}

void showInfo(Animal &animal) {
    animal.info();       // 非虚函数，始终调用父类版本
    animal.speak();      // 虚函数，调用子类版本
}

// ==================== 主函数 ====================
int main() {
    Cat cat("Tom");
    Dog dog("Buddy");
    Pig pig("Peppa");

    cout << "=== 多态演示 ===" << endl;
    doSpeak(cat);        // 输出：Tom说：喵喵喵
    doSpeak(dog);        // 输出：Buddy说：汪汪汪
    doSpeak(pig);        // 输出：Peppa说：哼哼哼

    cout << "\n=== 信息展示 ===" << endl;
    showInfo(cat);       // info() 调用父类版本，speak() 调用子类版本
    showInfo(dog);

    cout << "\n=== 父类指针数组 ===" << endl;
    // 用父类指针数组管理多个子类对象
    Animal *animals[] = { &cat, &dog, &pig };
    for (int i = 0; i < 3; i++) {
        animals[i]->speak();  // 多态：根据实际对象类型调用不同版本
    }

    return 0;
}
```

## 八、常见错误与注意事项

### 错误1：忘记加 virtual

```cpp
class Animal {
public:
    void speak() { ... }  // 没有 virtual！
};

class Cat : public Animal {
public:
    void speak() { ... }  // 这是隐藏（hiding），不是重写（override）
};

Animal *p = new Cat();
p->speak();   // 调用的是 Animal::speak()，不是 Cat::speak()！
// 很多初学者在这里困惑，就是因为忘了加 virtual
```

### 错误2：参数列表不一致

```cpp
class Animal {
public:
    virtual void speak() { ... }
};

class Cat : public Animal {
public:
    void speak(int x) { ... }  // 参数不同！这不是重写，是重载/隐藏
};
// 正确写法：参数列表必须完全一致
```

### 错误3：返回值类型不同

```cpp
class Animal {
public:
    virtual void speak() { ... }
};

class Cat : public Animal {
public:
    int speak() { ... }  // 返回值不同！这不是合法的重写
};
// 正确写法：返回值类型也必须一致
```

### 错误4：不通过父类指针/引用调用

```cpp
Cat cat;
cat.speak();    // 直接用子类对象调用，不会触发多态（但也能调到子类版本）
// 多态的精髓在于：通过父类指针/引用调用，运行时决定调用哪个版本
```

### 注意事项总结

```
1. virtual 只需要在父类中加，子类中可加可不加（推荐加上，提高可读性）
2. C++11 可以在子类重写函数后加 override 关键字，编译器会帮你检查是否真的重写了
3. 多态只对虚函数有效，普通函数不参与多态
4. 构造函数不能是虚函数
5. 析构函数可以是虚函数（下一节详细讲）
6. 多态会有一点点性能开销（需要查虚函数表），但通常可以忽略
```

## 九、小结

```
多态 = 同一接口，不同实现

静态多态（编译时）：
  - 函数重载：同名不同参数
  - 运算符重载：自定义运算符行为

动态多态（运行时）：
  - 三个条件：继承 + 重写虚函数 + 父类指针/引用
  - virtual 关键字：在父类函数声明前加 virtual
  - 效果：通过父类指针/引用调用时，根据实际对象类型决定调用哪个版本
```
