# C++11 新特性详解

---

## 1. auto 自动类型推导

编译器根据初始化表达式自动推导变量类型。

```cpp
auto a = 10;              // int
auto b = 3.14;            // double
auto c = 'A';             // char
auto d = "hello";         // const char*
auto e = string("hello"); // string
auto f = true;            // bool

// 最常用于迭代器（避免写又长又复杂的类型）
vector<int> v = {1, 2, 3};
for (auto it = v.begin(); it != v.end(); it++) {
    cout << *it << " ";
}

// auto 的推导规则
auto x = 10;          // int
auto &y = x;          // int&（引用）
auto *z = &x;         // int*（指针）
const auto w = 10;    // const int
```

### auto 的限制
```cpp
// 不能用于函数参数
// void func(auto a) {}  // 错误！C++20 前不行

// 不能用于类成员变量
// class A { auto a = 10; };  // 错误！

// 不能用于声明（必须初始化）
// auto a;  // 错误！编译器无法推导类型

// 不能用于数组
// auto arr = {1, 2, 3};  // 这是 initializer_list，不是数组
```

## 2. decltype 类型推导

decltype 根据表达式推导类型，但**不求值**（不会执行表达式）。

```cpp
int a = 10;
double b = 3.14;

decltype(a) c = 20;          // int（和 a 同类型）
decltype(b) d = 2.71;        // double（和 b 同类型）
decltype(a + b) e = 1.5;     // double（int + double = double）

// auto vs decltype
auto x = a;      // int（根据值推导）
decltype(a) y;   // int（根据类型推导，不需要初始化）
```

### decltype 的特殊规则
```cpp
int a = 10;
int &r = a;

decltype(r) b = a;    // int&（r 是引用，推导出引用）
decltype(a + 0) c;    // int（a+0 是右值，推导出值）
decltype((a)) d = a;  // int&（(a) 是左值表达式，推导出引用）
```

## 3. nullptr 替代 NULL

```cpp
// NULL 的问题：实际是 0，在函数重载时有歧义
void func(int) { cout << "int" << endl; }
void func(char*) { cout << "char*" << endl; }

func(NULL);    // 调用 func(int)！不是 func(char*)，可能不是你想要的
func(nullptr); // 调用 func(char*)，正确！

// nullptr 的类型是 std::nullptr_t，可以隐式转换为任意指针类型
int *p = nullptr;      // 正确
char *q = nullptr;     // 正确
// int a = nullptr;    // 错误！不能转换为整型
```

## 4. 范围 for 循环

```cpp
vector<int> v = {1, 2, 3, 4, 5};

// 方式1：值拷贝（不能修改原容器）
for (auto item : v) {
    cout << item << " ";
}

// 方式2：引用（可以修改原容器）
for (auto &item : v) {
    item *= 2;  // 每个元素翻倍
}

// 方式3：const 引用（只读，避免拷贝，推荐）
for (const auto &item : v) {
    cout << item << " ";
}

// 使用条件：容器必须有 begin() 和 end() 方法
// 或者是数组
int arr[] = {1, 2, 3, 4, 5};
for (auto x : arr) {
    cout << x << " ";
}
```

## 5. 列表初始化

```cpp
// 用花括号 {} 初始化
int a{10};                // int a = 10
int b = {20};             // int b = 20
string s{"hello"};        // string s = "hello"

// 容器的列表初始化
vector<int> v{1, 2, 3, 4, 5};
map<string, int> m{{"a", 1}, {"b", 2}};
list<int> l{10, 20, 30};

// 防止窄化转换（数据丢失）
int x = 10;
// double y{x};     // 警告或错误！double→int 可能丢失精度
double y = {3.14};  // 但 int→double 是安全的
int z{3.14};         // 错误！防止窄化（3.14→3 丢失小数部分）
```

## 6. Lambda 表达式

### 基本语法
```cpp
[捕获列表](参数列表) mutable -> 返回类型 { 函数体 }

// 各部分都可以省略（除了捕获列表）
[]() { cout << "hello" << endl; }();  // 最简单的 Lambda，立即调用
```

### 捕获方式
```cpp
int x = 10;
int y = 20;

// [] 不捕获任何外部变量
auto f1 = []() { /* 不能访问 x, y */ };

// [x] 值捕获 x（拷贝一份，只读）
auto f2 = [x]() { cout << x << endl; };

// [&x] 引用捕获 x（可以修改原变量）
auto f3 = [&x]() { x++; };

// [=] 值捕获所有外部变量
auto f4 = [=]() { cout << x << y << endl; };

// [&] 引用捕获所有外部变量
auto f5 = [&]() { x++; y++; };

// [x, &y] 混合捕获
auto f6 = [x, &y]() { y += x; };

// [=, &x] 默认值捕获，x 引用捕获
auto f7 = [=, &x]() { x += y; };
```

### 使用示例
```cpp
// 在 sort 中使用 Lambda 自定义排序
vector<int> v = {5, 3, 1, 4, 2};
sort(v.begin(), v.end(), [](int a, int b) { return a > b; });
// 降序：5 4 3 2 1

// 在 for_each 中使用 Lambda
for_each(v.begin(), v.end(), [](int val) {
    cout << val << " ";
});

// 在 find_if 中使用 Lambda
auto it = find_if(v.begin(), v.end(), [](int val) {
    return val > 3;
});
if (it != v.end()) {
    cout << "找到: " << *it << endl;  // 5
}

// Lambda 作为变量保存和传递
function<int(int, int)> add = [](int a, int b) { return a + b; };
cout << add(1, 2) << endl;  // 3
```

### mutable 关键字
```cpp
int x = 10;
auto f = [x]() mutable {
    x++;  // 值捕获的变量默认不能修改，加 mutable 可以修改副本
    cout << x << endl;  // 11
};
f();
cout << x << endl;  // 10（原变量不变，修改的是副本）
```

### Lambda 的本质
```cpp
// Lambda 本质上是编译器生成的匿名类，重载了 () 运算符
// [x]() { cout << x; } 等价于：
class __Lambda {
    int x;  // 捕获的变量成为成员
public:
    __Lambda(int x) : x(x) {}
    void operator()() const { cout << x; }
};
```

## 7. 右值引用与移动语义

### 左值和右值
```cpp
// 左值：可以取地址的表达式（有名字的变量）
int a = 10;      // a 是左值
int &r = a;      // 左值引用

// 右值：不能取地址的表达式（临时值、字面量）
10;              // 右值（字面量）
a + 1;           // 右值（临时结果）
string("hello"); // 右值（临时对象）
```

### 右值引用
```cpp
int &&r1 = 10;          // 绑定到右值
// int &&r2 = a;        // 错误！不能绑定到左值

// 右值引用延长了临时对象的生命周期
string &&s = string("hello");  // 临时 string 的生命周期被延长
cout << s << endl;              // hello
```

### 移动语义
```cpp
class MyString {
public:
    // 普通构造
    MyString(const char *str = "") {
        m_size = strlen(str);
        m_data = new char[m_size + 1];
        strcpy(m_data, str);
        cout << "构造: " << m_data << endl;
    }

    // 拷贝构造（深拷贝，分配新内存，复制数据）
    MyString(const MyString &other) {
        m_size = other.m_size;
        m_data = new char[m_size + 1];
        strcpy(m_data, other.m_data);
        cout << "拷贝构造: " << m_data << endl;
    }

    // 移动构造（窃取资源，不分配新内存）
    MyString(MyString &&other) noexcept {
        m_data = other.m_data;      // 窃取指针
        m_size = other.m_size;
        other.m_data = nullptr;     // 原对象置空
        other.m_size = 0;
        cout << "移动构造" << endl;
    }

    ~MyString() { delete[] m_data; }

    char *m_data;
    int m_size;
};

int main() {
    MyString s1("hello");
    MyString s2(s1);               // 拷贝构造（s1 是左值）
    MyString s3(move(s1));         // 移动构造（move 将 s1 转为右值引用）
    // 此时 s1 的 m_data 已经被置空
}
```

### std::move
```cpp
// move 不移动任何东西，只是将左值转换为右值引用
int a = 10;
int &&r = move(a);  // 将 a 转换为右值引用
// 此后 a 的值不确定（不要继续使用）

// move 的实际用途：告诉编译器"我不再需要这个对象了，可以窃取它的资源"
vector<int> v1 = {1, 2, 3};
vector<int> v2 = move(v1);  // 移动 v1 的资源到 v2
// v1 此时为空（状态不确定）
```

## 8. C++11 其他常用特性

### 初始化列表 initializer_list
```cpp
#include <initializer_list>

class MyVector {
public:
    MyVector(initializer_list<int> list) {
        for (auto it = list.begin(); it != list.end(); it++) {
            push_back(*it);
        }
    }
};

MyVector v = {1, 2, 3, 4, 5};
```

### enum class（强类型枚举）
```cpp
// 传统枚举的问题：会隐式转换为 int，不同枚举可能有同名值
enum Color { Red, Green, Blue };
enum TrafficLight { Red, Yellow, Green };  // 错误！Red 和 Green 重名

// C++11 强类型枚举
enum class Color { Red, Green, Blue };
enum class TrafficLight { Red, Yellow, Green };  // 正确！不同作用域

Color c = Color::Red;  // 必须加作用域
// int x = c;          // 错误！不能隐式转换为 int
int x = static_cast<int>(c);  // 正确！显式转换
```

### 原始字符串
```cpp
// 普通字符串需要转义
string s1 = "C:\\Users\\test\\file.txt";
string s2 = "hello\nworld";

// 原始字符串（不处理转义）
string s3 = R"(C:\Users\test\file.txt)";
string s4 = R"(hello\nworld)";  // \n 不是换行，是两个字符
```

### std::function 和 std::bind
```cpp
#include <functional>

// function：通用函数包装器
function<int(int, int)> func;
func = [](int a, int b) { return a + b; };
cout << func(1, 2) << endl;  // 3

func = [](int a, int b) { return a * b; };
cout << func(1, 2) << endl;  // 2

// bind：绑定参数
int add(int a, int b) { return a + b; }
auto add5 = bind(add, 5, placeholders::_1);  // 绑定第一个参数为 5
cout << add5(10) << endl;  // 15
```
