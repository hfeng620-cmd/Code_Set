# 关系运算符与函数调用运算符重载

## 一、关系运算符重载概述

关系运算符用于比较两个对象的大小关系。在自定义类中，编译器不知道如何比较两个对象，
所以我们需要重载关系运算符，让类的对象支持比较操作。

常用的关系运算符：`==`、`!=`、`>`、`<`、`>=`、`<=`

**核心原则：** 关系运算符重载的返回值类型是 `bool`（真或假）。

## 二、== 运算符重载

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person {
public:
    string m_Name;
    int m_Age;

    Person(string name, int age) : m_Name(name), m_Age(age) {}
};

// 重载 == 运算符（全局函数方式）
bool operator==(const Person& p1, const Person& p2) {
    // 姓名和年龄都相同，才认为是同一个人
    if (p1.m_Name == p2.m_Name && p1.m_Age == p2.m_Age) {
        return true;
    }
    return false;
}

int main() {
    Person p1("张三", 20);
    Person p2("张三", 20);
    Person p3("李四", 25);

    if (p1 == p2) {
        cout << "p1 和 p2 是同一个人" << endl;   // 输出这行
    } else {
        cout << "p1 和 p2 不是同一个人" << endl;
    }

    if (p1 == p3) {
        cout << "p1 和 p3 是同一个人" << endl;
    } else {
        cout << "p1 和 p3 不是同一个人" << endl;   // 输出这行
    }

    return 0;
}
```

## 三、!= 运算符重载

```cpp
// 重载 != 运算符（成员函数方式）
class Person {
public:
    string m_Name;
    int m_Age;

    Person(string name, int age) : m_Name(name), m_Age(age) {}

    // 成员函数方式重载 !=
    bool operator!=(const Person& p) {
        if (this->m_Name == p.m_Name && this->m_Age == p.m_Age) {
            return false;   // 相同返回false
        }
        return true;        // 不同返回true
    }
};

int main() {
    Person p1("张三", 20);
    Person p2("李四", 25);

    if (p1 != p2) {
        cout << "p1 和 p2 不同" << endl;   // 输出这行
    }

    return 0;
}
```

## 四、> 和 < 运算符重载

```cpp
class Student {
public:
    string m_Name;
    int m_Score;    // 成绩

    Student(string name, int score) : m_Name(name), m_Score(score) {}
};

// 重载 < 按成绩升序
bool operator<(const Student& s1, const Student& s2) {
    return s1.m_Score < s2.m_Score;
}

// 重载 > 按成绩降序
bool operator>(const Student& s1, const Student& s2) {
    return s1.m_Score > s2.m_Score;
}
```

## 五、关系运算符在排序中的应用

STL 的 `sort` 函数默认使用 `<` 运算符来排序。重载 `<` 后可以直接用 `sort`。

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>    // sort 需要这个头文件
using namespace std;

class Student {
public:
    string m_Name;
    int m_Score;

    Student(string name, int score) : m_Name(name), m_Score(score) {}

    // 重载 < 运算符，按成绩升序排列
    bool operator<(const Student& other) const {
        return this->m_Score < other.m_Score;
    }
};

// 自定义比较函数：按成绩降序
bool compareDesc(const Student& s1, const Student& s2) {
    return s1.m_Score > s2.m_Score;
}

int main() {
    vector<Student> students;
    students.push_back(Student("张三", 85));
    students.push_back(Student("李四", 92));
    students.push_back(Student("王五", 78));
    students.push_back(Student("赵六", 95));
    students.push_back(Student("钱七", 88));

    // 方式1：使用重载的 < 运算符（默认升序）
    sort(students.begin(), students.end());

    cout << "按成绩升序：" << endl;
    for (int i = 0; i < students.size(); i++) {
        cout << students[i].m_Name << " : " << students[i].m_Score << endl;
    }
    // 输出：
    // 王五 : 78
    // 张三 : 85
    // 钱七 : 88
    // 李四 : 92
    // 赵六 : 95

    cout << endl;

    // 方式2：使用自定义比较函数（降序）
    sort(students.begin(), students.end(), compareDesc);

    cout << "按成绩降序：" << endl;
    for (int i = 0; i < students.size(); i++) {
        cout << students[i].m_Name << " : " << students[i].m_Score << endl;
    }

    return 0;
}
```

## 六、函数调用运算符重载（仿函数）

### 6.1 什么是仿函数

函数调用运算符就是 `()`。重载 `()` 运算符后，对象可以像函数一样被调用。
这种重载了 `()` 运算符的对象称为 **仿函数**（Functor）。

```cpp
#include <iostream>
using namespace std;

// 仿函数类
class MyAdd {
public:
    // 重载 () 运算符
    int operator()(int a, int b) {
        return a + b;
    }
};

int main() {
    MyAdd add;          // 创建仿函数对象

    // 像函数一样调用对象
    int result = add(10, 20);   // 实际调用 operator()(10, 20)
    cout << "10 + 20 = " << result << endl;   // 输出：10 + 20 = 30

    return 0;
}
```

### 6.2 仿函数可以有状态

普通函数没有成员变量，而仿函数是类，可以有成员变量来保存状态。

```cpp
#include <iostream>
using namespace std;

// 带状态的仿函数：累加器
class Accumulator {
private:
    int m_Sum;      // 保存累加结果（状态）

public:
    Accumulator() : m_Sum(0) {}

    // 每次调用，加上一个数
    int operator()(int value) {
        m_Sum += value;
        return m_Sum;
    }

    // 获取当前累加值
    int getSum() const {
        return m_Sum;
    }
};

int main() {
    Accumulator acc;

    cout << acc(10) << endl;    // 输出：10
    cout << acc(20) << endl;    // 输出：30  （10+20）
    cout << acc(30) << endl;    // 输出：60  （10+20+30）
    cout << acc.getSum() << endl;   // 输出：60

    return 0;
}
```

### 6.3 仿函数 vs 普通函数

```cpp
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

// ========== 普通函数方式 ==========
bool compareFunc(int a, int b) {
    return a > b;
}

// ========== 仿函数方式 ==========
class CompareClass {
private:
    bool m_Ascending;   // 状态：是否升序

public:
    CompareClass(bool asc = true) : m_Ascending(asc) {}

    bool operator()(int a, int b) {
        if (m_Ascending) {
            return a < b;   // 升序
        } else {
            return a > b;   // 降序
        }
    }
};

int main() {
    vector<int> v = {5, 2, 8, 1, 9, 3};

    // 用普通函数排序（降序）
    sort(v.begin(), v.end(), compareFunc);

    // 用仿函数排序（可灵活切换升序/降序）
    CompareClass comp(true);     // 升序
    sort(v.begin(), v.end(), comp);

    // 打印结果
    for (int i = 0; i < v.size(); i++) {
        cout << v[i] << " ";
    }
    cout << endl;   // 输出：1 2 3 5 8 9

    return 0;
}
```

**仿函数相比普通函数的优势：**

| 特性 | 普通函数 | 仿函数 |
|------|---------|--------|
| 有状态（成员变量） | 无 | 有 |
| 可作为模板参数 | 不可以 | 可以 |
| 内联优化 | 编译器不一定优化 | 编译器更容易内联 |
| 灵活性 | 固定行为 | 可通过构造参数改变行为 |

### 6.4 匿名函数对象

创建对象时不给名字，直接调用，称为匿名函数对象。

```cpp
#include <iostream>
using namespace std;

class MyPrint {
public:
    void operator()(string text) {
        cout << text << endl;
    }
};

int main() {
    // 方式1：有名对象
    MyPrint print;
    print("Hello");

    // 方式2：匿名对象，直接调用
    MyPrint()("World");     // 创建匿名对象并立即调用

    // 输出：
    // Hello
    // World

    return 0;
}
```

### 6.5 仿函数在 STL 中的应用

STL 中大量使用仿函数，例如 `sort`、`priority_queue`、`transform` 等。

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <functional>   // greater、less 等内置仿函数
using namespace std;

class PrintElement {
public:
    void operator()(int val) {
        cout << val << " ";
    }
};

int main() {
    vector<int> v = {5, 2, 8, 1, 9, 3};

    // 使用 greater<int>() 仿函数降序排序
    sort(v.begin(), v.end(), greater<int>());

    // 使用自定义仿函数遍历打印
    for_each(v.begin(), v.end(), PrintElement());
    cout << endl;   // 输出：9 8 5 3 2 1

    return 0;
}
```

### 6.6 返回值类型可以不同

仿函数的 `operator()` 返回值不限于 `bool`，可以是任意类型。

```cpp
#include <iostream>
using namespace std;

// 计算阶乘的仿函数
class Factorial {
public:
    // 返回 long long，不是 bool
    long long operator()(int n) {
        long long result = 1;
        for (int i = 2; i <= n; i++) {
            result *= i;
        }
        return result;
    }
};

int main() {
    Factorial fact;

    cout << "5! = " << fact(5) << endl;     // 输出：5! = 120
    cout << "10! = " << fact(10) << endl;   // 输出：10! = 3628800
    cout << "15! = " << fact(15) << endl;   // 输出：15! = 1307674368000

    return 0;
}
```

## 七、综合练习

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

class Person {
public:
    string m_Name;
    int m_Age;
    int m_Score;

    Person(string name, int age, int score)
        : m_Name(name), m_Age(age), m_Score(score) {}

    // 重载 == 判断是否为同一人
    bool operator==(const Person& other) const {
        return m_Name == other.m_Name && m_Age == other.m_Age;
    }

    // 重载 < 用于排序（先按分数降序，分数相同按年龄升序）
    bool operator<(const Person& other) const {
        if (m_Score != other.m_Score) {
            return m_Score > other.m_Score;     // 分数降序
        }
        return m_Age < other.m_Age;             // 年龄升序
    }
};

// 仿函数：打印 Person 信息
class PrintPerson {
public:
    void operator()(const Person& p) {
        cout << p.m_Name << "  年龄:" << p.m_Age << "  分数:" << p.m_Score << endl;
    }
};

int main() {
    vector<Person> people;
    people.push_back(Person("张三", 20, 85));
    people.push_back(Person("李四", 22, 92));
    people.push_back(Person("王五", 21, 85));
    people.push_back(Person("赵六", 23, 95));

    // 排序（利用重载的 < 运算符）
    sort(people.begin(), people.end());

    cout << "排序结果：" << endl;
    for_each(people.begin(), people.end(), PrintPerson());

    // 比较（利用重载的 == 运算符）
    if (people[0] == people[1]) {
        cout << "前两个人是同一人" << endl;
    } else {
        cout << "前两个人不是同一人" << endl;
    }

    return 0;
}
```

## 八、总结

| 知识点 | 要点 |
|--------|------|
| 关系运算符重载 | 返回 `bool` 类型 |
| `==` 和 `!=` | 判断对象是否相等/不等 |
| `>` 和 `<` | 比较对象大小，常配合 `sort` 使用 |
| 仿函数 | 重载 `()` 运算符，对象可以像函数一样调用 |
| 仿函数优势 | 有状态、可作模板参数、更灵活 |
| 匿名函数对象 | 不创建变量名，直接 `类名()(参数)` 调用 |
| STL 中的应用 | `sort`、`for_each`、`priority_queue` 等大量使用仿函数 |
